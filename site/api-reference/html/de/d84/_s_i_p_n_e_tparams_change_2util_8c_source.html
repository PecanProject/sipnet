65/* Dynamically allocate space for a 2-d array of doubles of given size,
66 return pointer to start of array
67
68 Ensures that array elements are continuous to increase efficiency
69
70 To free array, first free(arr[0]), then free(arr)
71*/
72double **make2DArray(int nrows, int ncols) {
73 double **arr;
74 int i;
75
76 arr = (double **)malloc(nrows * sizeof(double *));
77
78 // assure that all space is allocated continuously:
79 arr[0] = (double *)malloc(nrows * ncols * sizeof(double));
80 for (i = 1; i < nrows; i++)
81 arr[i] = arr[0] + i*ncols;
82
83 return arr;
84}
