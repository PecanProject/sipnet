110event_type_t getEventType(const char *eventTypeStr) {
111 if (strcmp(eventTypeStr, "irrig") == 0) {
112 return IRRIGATION;
113 } else if (strcmp(eventTypeStr, "fert") == 0) {
114 return FERTILIZATION;

125EventNode **readEventData(char *eventFile, int numLocs) {
126 int loc, year, day, eventType;
127 int currLoc, currYear, currDay;
128 int EVENT_LINE_SIZE = 1024;
129 int numBytes;
130 char *eventParamsStr;
131 char eventTypeStr[20];
132 char line[EVENT_LINE_SIZE];
133 EventNode *curr, *next;
134
135 printf("Begin reading event data from file %s\n", eventFile);
136
137 EventNode **events =
138 (EventNode **)calloc(sizeof(EventNode *), numLocs * sizeof(EventNode *));
139 // status of the read
140 FILE *in = openFile(eventFile, "r");
141
142 if (fgets(line, EVENT_LINE_SIZE, in) == NULL) {
143 printf("Error: no event data in %s\n", eventFile);
144 exit(1);
145 }
146 sscanf(line, "%d %d %d %s %n", &loc, &year, &day, eventTypeStr, &numBytes);
147 eventParamsStr = line + numBytes;
148
149 eventType = getEventType(eventTypeStr);
150 if (eventType == UNKNOWN_EVENT) {
151 printf("Error: unknown event type %s\n", eventTypeStr);
152 exit(1);
153 }
154
155 next = createEventNode(loc, year, day, eventType, eventParamsStr);
156 events[loc] = next;
157 currLoc = loc;
158 currYear = year;
159 currDay = day;
160
161 while (fgets(line, EVENT_LINE_SIZE, in) != NULL) {
162 // We have another event
163 curr = next;
164 sscanf(line, "%d %d %d %s %n", &loc, &year, &day, eventTypeStr, &numBytes);
165 eventParamsStr = line + numBytes;
166
167 eventType = getEventType(eventTypeStr);
168 if (eventType == UNKNOWN_EVENT) {
169 printf("Error: unknown event type %s\n", eventTypeStr);
170 exit(1);
171 }
172
173 // make sure location and time are non-decreasing
174 if (loc < currLoc) {
175 printf("Error reading event file: was reading location %d, trying to "
176 "read location %d\n",
177 currLoc, loc);
178 printf("Event records for a given location should be contiguous, and "
179 "locations should be in ascending order\n");
180 exit(1);
181 }
182 if ((loc == currLoc) && ((year < currYear) || (day < currDay))) {
183 printf("Error reading event file: for location %d, last event was at "
184 "(%d, %d) ",
185 currLoc, currYear, currDay);
186 printf("next event is at (%d, %d)\n", year, day);
187 printf("Event records for a given location should be in time-ascending "
188 "order\n");
189 exit(1);
190 }
191
192 next = createEventNode(loc, year, day, eventType, eventParamsStr);
193 if (currLoc == loc) {
194 // Same location, add the new event to this location's list
195 curr->nextEvent = next;
196 } else {
197 // New location, update location and start a new list
198 currLoc = loc;
199 events[currLoc] = next;
200 }
201 }
202
203 fclose(in);
204 return events;
205}
