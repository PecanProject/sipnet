36// Checks to make sure that all parameters with isRequired=true have been read
37// If not, kills program
38// Writes out names of all parameters that weren't read (even if not required, as a warning message)
39void checkAllRead(SpatialParams *spatialParams) {
40 int i;
41 int okay;
42 OneSpatialParam *param;
43
44 okay = 1; // so far so good
45 for (i = 0; i < spatialParams->numParameters; i++) {
46 param = &(spatialParams->parameters[i]);
47 if (param->value == NULL) {
48 if (param->isRequired) { // should have been read but wasn't!
49 okay = 0;
50 printf("ERROR: Didn't read required parameter %s\n", param->name);
51 }
52 else {
53 printf("WARNING: Didn't read parameter %s (not flagged as required, so continuing)\n", param->name);
54 }
55 }
56 }
57
58 if (!okay)
59 exit(1);
60}

90/* Read array of spatial values from file, for a single parameter
91 Values are returned in the array spatialValues, which must be at least of length numLocs
92 spatialParamFile is the (already-open) file of spatial parameters
93 - the file pointer should be pointing to the start of the line we want to read
94 pName gives the name of the parameter; we'll abort if the next parameter does not have the name given by pName
95 numLocs is the number of values to read
96 */
97void readSpatialValues(double *spatialValues, FILE *spatialParamFile, char pName[PARAM_MAXNAME], int numLocs) {
98 char pNameSpatial[PARAM_MAXNAME];
99 int status;
100 int i;
101
102 // read name from spatialParamFile, make sure it's what we expect:
103 status = fscanf(spatialParamFile, "%s", pNameSpatial);
104 if (status == EOF || status == 0) { // error reading
105 printf("Error trying to read %s from spatialParamFile\n", pName);
106 exit(1);
107 }
108 else if (strcmpIgnoreCase(pName, pNameSpatial) != 0) { // compare, ignore case
109 printf("Error: read %s from spatialParamFile, expected %s\n", pNameSpatial, pName);
110 exit(1);
111 }
112
113 // now read values and assign param->value, guess, best, and knob:
114 for (i = 0; i < numLocs; i++) {
115 status = fscanf(spatialParamFile, "%lf", &(spatialValues[i]));
116 if (status == EOF || status == 0) { // error reading
117 printf("Error: did not find enough values in spatialParamFile for %s\n", pName);
118 exit(1);
119 }
120 }
121
122}

150/* Initialize next spatial parameter:
151 Set name of parameter equal to "name",
152 and set parameter's externalLoc pointer to point to the location given by "externalLoc" parameter
153 Also set parameter's isRequired value: if true, then we'll terminate execution if this parameter is not read from file
154 Note: acceptable to have externalLoc = NULL, but then it won't be assigned by loadSpatialParams
155*/
156void initializeOneSpatialParam(SpatialParams *spatialParams, char *name, double *externalLoc, int isRequired) {
157 int paramIndex; // index of next uninitialized parameter
158 OneSpatialParam *param; // a pointer to the next uninitialized parameter
159
160 if (allParamsInitialized(spatialParams)) { // all parameters have been initialized!
161 printf("Error trying to initialize %s: have already initialized all %d parameters\n", name, spatialParams->maxParameters);
162 printf("Check value of maxParameters passed into newSpatialParams function\n");
163 exit(1);
164 }
165
166 // otherwise, get the index of the next uninitialized parameter
167 paramIndex = spatialParams->numParameters;
168 // and set param to point to it for easier access
169 param = &(spatialParams->parameters[paramIndex]);
170
171 strcpy(param->name, name);
172 param->externalLoc = externalLoc;
173 param->isRequired = isRequired;
174
175 // set value pointer to be NULL (this will mark unread parameter)
176 param->value = NULL;
177
178 spatialParams->numParameters++;
179}

182/* Read all parameters from file, setting all values in spatialParams structure appropriately
183
184 Structure of paramFile:
185 name value changeable min max sigma
186 where value is a number if parameter is non-spatial, or * if parameter is spatial; changeable is 0 or 1 (boolean)
187 If parameter is spatial, look in spatialParamFile for value, which has the following structure:
188 name value [value value...]
189 where the number of values is equal to the number of spatial locations
190
191 The order of the parameters in paramFile does not matter,
192 but parameters must be specified in the same order in paramFile and spatialParamFile
193
194 ! is a comment character in paramFile: anything after a ! on a line is ignored
195 note, though, that comments are not currently allowed in spatialParamFile
196
197 PRE: paramFile is open and file pointer points to start of file
198 spatialParamFile is open and file pointer points to 2nd line (after the numLocs line)
199 */
200void readSpatialParams(SpatialParams *spatialParams, FILE *paramFile, FILE *spatialParamFile) {
201 const char *SEPARATORS = " \t\n\r"; // characters that can separate values in parameter files
202 const char *COMMENT_CHARS = "!"; // comment characters (ignore everything after this on a line)
203
204 char line[256];
205 char pName[PARAM_MAXNAME]; // parameter name
206 int paramIndex;
207 OneSpatialParam *param; // a pointer to a single parameter, for easier access
208 char strValue[32]; // before we know whether value is a number or "*"
209 double value, min, max, sigma;
210 double *spatialValues; // temporary storage for values read from spatial param file
211 int changeable, spatiallyVarying; // is the parameter changeable? is it spatially-varying?
212 char *errc;
213 int isComment;
214 int numLocs, i;
215
216 numLocs = spatialParams->numLocs; // for those parameters that are spatially-varying
217 spatialValues = (double *)malloc(numLocs * sizeof(double));
218
219 while (fgets(line, sizeof(line), paramFile) != NULL) { // while not EOF or error
220 // remove trailing comments:
221 isComment = stripComment(line, COMMENT_CHARS);
222
223 if (!isComment) { // if this isn't just a comment line or blank line
224 // tokenize line:
225 strcpy(pName, strtok(line, SEPARATORS)); // copy first token into pName
226 strcpy(strValue, strtok(NULL, SEPARATORS)); // copy next token into strValue; wait until later to figure out if it's "*" or a number
227 changeable = strtol(strtok(NULL, SEPARATORS), &errc, 0);
228 min = strtod(strtok(NULL, SEPARATORS), &errc);
229 max = strtod(strtok(NULL, SEPARATORS), &errc);
230 sigma = strtod(strtok(NULL, SEPARATORS), &errc);
231
232 // now we need to see if we read in an actual value, or a "*" (if the latter, it's a spatially-varying parameter)
233 // note that we read this before doing the error checking on paramIndex:
234 // that way, even if we ignore this parameter, we've still read (and ignored) the corresponding line in the spatial param file, if it's a spatially-varying parameter
235 if (strcmp(strValue, "*") == 0) { // we read "*": this is a spatially-varying parameter
236 readSpatialValues(spatialValues, spatialParamFile, pName, numLocs);
237 spatiallyVarying = 1;
238 }
239 else {
240 spatiallyVarying = 0;
241 }
242
243 paramIndex = locateParam(spatialParams, pName);
244
245 if (paramIndex == -1) { // not found
246 printf("WARNING: Ignoring parameter %s: this parameter wasn't initialized in the code\n", pName);
247 }
248 else if (valueSet(spatialParams, paramIndex)) {
249 printf("Error reading parameter file: read %s, but this parameter has already been set\n", pName);
250 exit(1);
251 }
252 else { // otherwise, we're good to go
253
254 // set param to point to the appropriate parameter, for easier access
255 param = &(spatialParams->parameters[paramIndex]);
256
257 // mark this as the next parameter read from file:
258 spatialParams->readIndices[spatialParams->numParamsRead] = paramIndex;
259 spatialParams->numParamsRead++;
260
261 // fill the new spatialParam structure with changeable, min, max, and sigma
262 param->isChangeable = changeable;
263 param->min = min;
264 param->max = max;
265 param->sigma = sigma;
266
267 if (changeable) { // we need to update info on changeable params in spatialParams
268 spatialParams->changeableParamIndices[spatialParams->numChangeableParams] = paramIndex;
269 // before we change it, spatialParams->numChangeableParams is the index of the first free spot in the changeableParamIndices vector
270 spatialParams->numChangeableParams++;
271 }
272
273 if (spatiallyVarying) {
274 param->numLocs = numLocs;
275
276 // allocate space:
277 param->value = (double *)malloc(numLocs * sizeof(double));
278 param->guess = (double *)malloc(numLocs * sizeof(double));
279 param->best = (double *)malloc(numLocs * sizeof(double));
280 param->knob = (double *)malloc(numLocs * sizeof(double));
281
282 // now assign param->value, guess, best, and knob (these were read above, in call to readSpatialValues):
283 for (i = 0; i < numLocs; i++) {
284 // assign value, guess and best to all be the guess value initially:
285 param->value[i] = param->guess[i] = param->best[i] = spatialValues[i];
286 param->knob[i] = 0; // assign knob to be 0 initially
287 } // for i
288 } // if spatially-varying
289 else { // non-spatially-varying parameter
290 param->numLocs = 0; // signifies non-spatially-varying
291
292 // allocate space for a single value in each of param->value, guess, best and knob:
293 param->value = (double *)malloc(sizeof(double));
294 param->guess = (double *)malloc(sizeof(double));
295 param->best = (double *)malloc(sizeof(double));
296 param->knob = (double *)malloc(sizeof(double));
297
298 // assign value, guess and best to all be the guess value initially:
299 value = strtod(strValue, &errc); // convert string value to double
300 param->value[0] = param->guess[0] = param->best[0] = value;
301 param->knob[0] = 0; // assign knob to be 0 initially
302 } // else non-spatially-varying
303 } // else (no errors in reading this line from parameter file)
304 } // if !isComment
305 } // while not EOF or error
306
307 // check for error in reading:
308 if (ferror(paramFile)) {
309 printf("Error reading file in readSpatialParams\n");
310 printf("ferror = %d\n", ferror(paramFile));
311 exit(1);
312 }
313
314 checkAllRead(spatialParams); // terminate program if some required parameters weren't read
315
316 free(spatialValues);
317} // readSpatialParams

332// Find parameter with given name in the parameters vector
333// If found, return index in vector, otherwise return -1
334int locateParam(SpatialParams *spatialParams, char *name) {
335 int i;
336 int found;
337
338 i = 0;
339 found = 0;
340 while (i < spatialParams->numParameters && !found) {
341 if (strcmpIgnoreCase(name, spatialParams->parameters[i].name) == 0) {
342 found = 1;
343 }
344 i++;
345 }
346
347 if (found)
348 return (i - 1);
349 else
350 return -1;
351}

479 (Note: ignores parameters that were never read in)
480 If loc = -1, set best value of all parameters at ALL locations to be equal to their current value
481*/
482void setAllSpatialParamBests(SpatialParams *spatialParams, int loc) {
483 int numParams;
484 int firstLoc, lastLoc;
485 int i, currLoc;
486 double value;
487
488 numParams = spatialParams->numParameters;
489 if (loc == -1) { // set best at all locations
490 firstLoc = 0;
491 lastLoc = spatialParams->numLocs - 1;
492 }
493 else // set best at a single location
494 firstLoc = lastLoc = loc;

499 for (currLoc = firstLoc; currLoc <= lastLoc; currLoc++) {
500 value = getSpatialParam(spatialParams, i, currLoc);

503 }
504 else { // non-spatial

506 setSpatialParamBest(spatialParams, i, 0, value);
507 }

526 For spatially-varying parameters, load value at location given by loc
527 (Purpose: load all parameters into a localized structure for a model run, to make the model run more efficient)

533void loadSpatialParams(SpatialParams *spatialParams, int loc) {

548/* If randomReset = 0, set values of all parameters to be equal to guess values
549 If randomReset non-zero, set parameter values to be somewhere (chosen uniform randomly) between min and max
550 - Note that non-changeable parameters will still be set to their guess values, though
551 (Note: ignores parameters that were never read in)
552 (If randomReset non-zero, random number generator must have been seeded)
553 Set best values of all parameters to be equal to current values
554 Set knobs of all parameters to be equal to knob argument
555*/
556void resetSpatialParams(SpatialParams *spatialParams, double knob, int randomReset) {
557 int numParams, numLocs;
558 int i, loc;
559 double value;
560
561 numParams = spatialParams->numParameters;
562 numLocs = spatialParams->numLocs;
563
564 for (i = 0; i < numParams; i++) {
565 if (valueSet(spatialParams, i)) { // if valueSet is false, ignore this parameter (it was never read in)
566 if (isSpatial(spatialParams, i)) {
567 for (loc = 0; loc < numLocs; loc++) {
568 if ((randomReset == 0) || !isChangeable(spatialParams, i)) // non-random: use guess value
569 value = getSpatialParamGuess(spatialParams, i, loc);
570 else // random: set parameter value to be somewhere between min and max
571 value = getSpatialParamMin(spatialParams, i) + (getSpatialParamMax(spatialParams, i) - getSpatialParamMin(spatialParams, i))
572 * ((float)rand()/RAND_MAX);
573 setSpatialParam(spatialParams, i, loc, value);
574 setSpatialParamBest(spatialParams, i, loc, value);
575 setSpatialParamKnob(spatialParams, i, loc, knob);
576 }
577 }
578 else { // non-spatial
579 if ((randomReset == 0) || !isChangeable(spatialParams, i)) // non-random: use guess value
580 value = getSpatialParamGuess(spatialParams, i, 0);
581 else // random: set parameter value to be somewhere between min and max
582 value = getSpatialParamMin(spatialParams, i) + (getSpatialParamMax(spatialParams, i) - getSpatialParamMin(spatialParams, i))
583 * ((float)rand()/RAND_MAX);
584 setSpatialParam(spatialParams, i, 0, value);
585 setSpatialParamBest(spatialParams, i, 0, value);
586 setSpatialParamKnob(spatialParams, i, 0, knob);
587 } // else non-spatial
588 } // if valueSet
589 } // for i
590} // resetSpatialParams

593/* Write best parameter values, and other parameter info, to files
594 (only write parameters that were read in, and in the same order that they were read in)
595 Note that other parameter info won't have changed since the read - we just copy that over
596 Write all parameter info to file with name *paramFile, and spatially-varying parameter values to file with name *spatialParamFile
597 See readSpatialParams for file formats
598*/
599void writeBestSpatialParams(SpatialParams *spatialParams, char *paramFile, char *spatialParamFile) {
600 FILE *paramF, *spatialParamF;
601 int numParamsRead, numLocs;
602 int i, j, index;
603
604 paramF = openFile(paramFile, "w");
605 spatialParamF = openFile(spatialParamFile, "w");
606
607 numParamsRead = getNumParamsRead(spatialParams);
608 numLocs = spatialParams->numLocs;
609
610 fprintf(spatialParamF, "%d\n", numLocs);
611
612 for (i = 0; i < numParamsRead; i++) { // loop through parameters
613 index = spatialParams->readIndices[i]; // allows us to just consider the parameters that were read in, in the proper order
614 fprintf(paramF, "%s\t", spatialParams->parameters[index].name);
615 if (isSpatial(spatialParams, index)) {
616 fprintf(paramF, "*\t"); // write "*" in place of value, then write values to spatialParamFile
617 fprintf(spatialParamF, "%s", spatialParams->parameters[index].name);
618 for (j = 0; j < numLocs; j++) // loop through locations
619 fprintf(spatialParamF, "\t%f", getSpatialParamBest(spatialParams, index, j));
