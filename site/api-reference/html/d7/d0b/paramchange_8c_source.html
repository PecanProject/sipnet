294/* pre: **origArray contains original (unaggregated) data/model output
295 (origArray[i][j] is timestep i, data type j)
296 myNumAggSteps is number of aggregated time steps
297 myAggSteps is a vector [0..myNumAggSteps-1] containing number of time steps in each aggregated steps
298 myStartOpt is index of first time step used in optimization (1-indexed)
299 Note that myNumAggSteps, myAggSteps and myStartOpt are taken from global variables at a single spatial location

302 and return result in theAggedData

304void computeAggedData(double **theAggedData, double **origArray, int myNumAggSteps, int *myAggSteps, int myStartOpt, int numDataTypes) {
305 int i, j, dataType, index;
306
307 index = myStartOpt - 1; // index into data array (myStartOpt is 1-indexed)
308 for (i = 0; i < myNumAggSteps; i++) {
309 for (dataType = 0; dataType < numDataTypes; dataType++)
310 theAggedData[i][dataType] = 0.0; // reset sums
311
312 for (j = 0; j < myAggSteps[i]; j++) { // loop through all steps in this aggregated step
313 for (dataType = 0; dataType < numDataTypes; dataType++) // accumulate each sum
314 theAggedData[i][dataType] += origArray[index][dataType];
315 index++;
316 }

321/* Aggregated difference - ESTIMATES SIGMA, RETURNS AGGREGATE INFO
322 Same as difference function above, but aggregates model output to fewer steps
323 Total difference is a weighted sum of error on aggregated output vs. data
324 and error on unaggregated output vs. data (weight determined by global unaggedWeight)
325
326 Run modelF with given parameters at location loc, compare output with measured data
327 using data types given by dataTypeIndices[0..numDataTypes/2-1]
328 Return a measure of the difference between measured and predicted data
329 (Higher = worse)
330 Assumes error is the same on each aggregated time step,
331 even if aggregation lengths are different
332 Return best sigma value for each data type in sigma[0..numDataTypes-1]
333 (where sigma[i] is sigma for data type given by dataTypeIndices[i%(numDataTypes/2)];
334 if i < numDataTypes/2 then sigma[i] gives sigma for unaggregated data; otherwise for aggregated data)
335 Return mean error, mean daily-aggregated error, and yearly-aggregated output for each year
336 and each data type in *outputInfo array (indexed like sigma array - see above)
337
338 Pre: sigma and outputInfo are already malloced, as are outputInfo[*].years arrays
339 global *numAggSteps, **aggSteps and ***aggedData have all been set appropriately,
340 and global **aggedModel has been malloced appropriately
341 numDataTypes is actually TWICE the number of data types
342 (for each data type, one unaggregated and one aggregated)
343
344 FOR NOW, WE IGNORE THE VALID FRACTION (THIS HASN'T BEEN AGGREGATED UP)
345 Only use points between startOpt and endOpt (set in readData)
346 (NOTE: UNTESTED FOR ANYTHING BUT STARTOPT=1, ENDOPT=-1 (I.E. ALL POINTS))

350// 6/9/11: For now we don't do anything with the cost function or data Type weights - we just want to make
351// it consistent with the difference function above so we don't get an error.
352double aggedDifference(double *sigma, OutputInfo *outputInfo,
353 int loc, SpatialParams *spatialParams, double paramWeight,
354 void (*modelF)(double **, int, int *, SpatialParams *, int),
355 int dataTypeIndices[], int numDataTypes, int costFunction, double dataTypeWeights[])
356{
357 int i, dataNum;
358 double *sumSquares; // one sum of squares value for each data type
359 int *n; // number of data points used in each sumSquares
360 double logLike; // the log likelihood
361
362 sumSquares = makeArray(numDataTypes);
363 n = (int *)malloc(numDataTypes * sizeof(int));
364
365 (*modelF)(model, numDataTypes/2, dataTypeIndices, spatialParams, loc);
366 // run model, put results in model array
367 // divide numDataTypes by 2 so we have actual (unbifurcated) number of data types
368
369 computeAggedData(aggedModel, model, numAggSteps[loc], aggSteps[loc], startOpt[loc], numDataTypes/2); // aggregate model, filling aggedModel array
370 // again, divide numDataTypes by 2 to give actual (unbifurcated) number of data types
371
372 // initialize sumSquares and count arrays
373 for (dataNum = 0; dataNum < numDataTypes; dataNum++) {
374 sumSquares[dataNum] = 0.0;
375 n[dataNum] = 0;
376 }
377
378 // compute sum of squares on unaggregated data:
379 for (i = startOpt[loc] - 1; i < endOpt[loc]; i++) {
380 for (dataNum = 0; dataNum < numDataTypes/2; dataNum++) {
381 if (valid[loc][i][dataNum]) {
382 sumSquares[dataNum] += pow((model[i][dataNum] - data[loc][i][dataNum]), 2);
383 n[dataNum]++;
384 }
385 }
386 }
387
388 // compute sum of squares on aggregated data (note: we don't check validity here - instead use all points):
389 for (i = 0; i < numAggSteps[loc]; i++) {
390 for (dataNum = 0; dataNum < numDataTypes/2; dataNum++) {
391 sumSquares[numDataTypes/2 + dataNum] += pow((aggedModel[i][dataNum] - aggedData[loc][i][dataNum]), 2);
392 n[numDataTypes/2 + dataNum]++;
393 }
394 }
395
396 // calculate aggregate info on each data type
397 for (dataNum = 0; dataNum < numDataTypes/2; dataNum++) {
398 aggregates(outputInfo, model, loc, dataNum);
399
400 // copy aggregate info from position i to position (numDataTypes/2 + i)
401 // (same data type, but aggregated - will have same aggregate info)
402 copyOutputInfo(&(outputInfo[numDataTypes/2 + dataNum]), &(outputInfo[dataNum]));
403 }
404
405 // compute log likelihood (actually, negative log likelihood, discarding constant terms):
406 logLike = 0;
407 for (dataNum = 0; dataNum < numDataTypes; dataNum++) {
408 sigma[dataNum] = sqrt(sumSquares[dataNum]/(double)(n[dataNum]));
409 /* we can estimate sigma[i] using just sumSquares[i] because sigma[i] is calculated by taking the partial derivative
410 of likelihood with respect to sigma[i], and this partial only depends on sumSquares[i]
411 */
412 logLike += n[dataNum] * log(sigma[dataNum]);
413 logLike += sumSquares[dataNum]/(2.0*(sigma[dataNum])*(sigma[dataNum]));
414
415 if (dataNum == numDataTypes/2 - 1) // we just finished all unaggregated types
416 // apply unaggedWeight to current log likelihood
417 logLike *= unaggedWeight;
418 }
419
420 free(sumSquares);
421 free(n);
422
423 // NOTE: this is actually the NEGATIVE log likelihood, discarding constant terms
424
425 return logLike;
426}

430/* Take array of model output, compare output with measured data - for given dataNum
431 (i.e. perform comparisons between model[*][dataNum] and data[loc][*][dataNum]
432 Return (in outputInfo[dataNum]) mean error (per day), mean daily-aggregated error (per day)
433 Also fill years array with annual total NEE for each year
434 and set outputInfo.numYears, the size of the years array
435 Pre: outputInfo is already malloced, as are outputInfo[*].years arrays
436 (Can later use years array to find yearly-aggregated error and interannual variability)
437*/
438void aggregates(OutputInfo *outputInfo, double **model, int loc, int dataNum)
439{
440 const int DAYS_IN_YR[] = {365,366}; /* given leapYr = 0 or 1, DAYS_IN_YR[leapYr]
441 represents the number of days in this year */
442 int i, j, day, julianDay, year;
443 int leapYr; // 1 if this is a leap year
444 int index;
445 double *modelD, *dataD; // daily aggregations
446 int yearIndex; // starting at 0 rather than being a 4-digit year
447 double sum;
448 double netNeeM, netNeeD; // net model and data nee for this aggregated step
449
450 modelD = makeArray(aggInfo[loc].numDays);
451 dataD = makeArray(aggInfo[loc].numDays);
452
453 sum = 0.0;
454 for (i = aggInfo[loc].startPt - 1; i < aggInfo[loc].endPt; i++) {
455 sum += fabs(model[i][dataNum] - data[loc][i][dataNum]);
456 }
457
458 outputInfo[dataNum].meanError = sum/aggInfo[loc].numDays; // mean error per day
459
460 // fill daily aggregation arrays, compute error aggregated over days:
461 sum = 0.0;
462 index = aggInfo[loc].startPt - 1;
463 for (day = 0; day < aggInfo[loc].numDays; day++) {
464 netNeeM = netNeeD = 0.0;
465 for (j = 0; j < aggInfo[loc].spd[day]; j++) {
466 netNeeM += model[index][dataNum];
467 netNeeD += data[loc][index][dataNum];
468 index++;
469 }
470 modelD[day] = netNeeM;
471 dataD[day] = netNeeD;
472 sum += fabs(netNeeM - netNeeD);
473 }
474
475 outputInfo[dataNum].daysError = sum/aggInfo[loc].numDays; // mean daily-aggregated error per day
476
477
478 // now do yearly and overall aggregation:
479
480 // first fill year arrays and find yearly aggregation:
481 day = 0;
482 julianDay = aggInfo[loc].startDay;
483 year = aggInfo[loc].startYear;
484 leapYr = (year % 4 == 0); // holds for 1900 < year < 2100
485 yearIndex = 0;
486 while (day < aggInfo[loc].numDays) {
487 netNeeM = 0.0;
488 while (julianDay <= DAYS_IN_YR[leapYr] && day < aggInfo[loc].numDays) {
489 // loop through current year
490 netNeeM += modelD[day];
491 day++;
492 julianDay++;
493 }
494 // HAPPY NEW YEAR!
495
496 outputInfo[dataNum].years[yearIndex] = netNeeM; // NEE over this whole year
497 // can later use this to find mean yearly-aggregated error per day and interannual variability
498
499 julianDay = 1;
500 year++;
501 yearIndex++;
502 leapYr = (year % 4 == 0); // holds for 1900 < year < 2100
503 }
504 outputInfo[dataNum].numYears = yearIndex; // the actual number of years
505
506 free(modelD);
507 free(dataD);
508}

546/* Parse one string (spdString) from .spd file
547 Three possible options:
548 - If spdString is "-1", return -1, leave spd unchanged, set count to 0
549 - If spd is "#n", where n is a positive integer (e.g. "#100"), return 0, don't change spd (we'll use last spd value), set count to n
550 (this signifies that we should repeat previous value n times)
551 - Otherwise, spd must be a non-negative integer (in string format): return 1, set spd to the number, set count to 1
552*/
553int parseSpdValue(char *spdString, int *spd, int *count) {
554 char *errc;
555
556 if (strcmp(spdString, "-1") == 0) {
557 count = 0;
558 return -1;
559 }
560 else if (spdString[0] == '#') { // spd is "#n" (e.g. #100)
561 *count = strtol(strtok(spdString, "#"), &errc, 0); // get rid of '#', convert to integer
562 return 0;
563 }
564 else { // spd is an integer (in string format)
565 *spd = strtol(spdString, &errc, 0);
566 *count = 1;
567 return 1;
568 }
