18/* return index of next unread parameter
19 if no more unread parameters, returns spatialParams->numParameters
20*/
21int nextUnreadParam(SpatialParams *spatialParams) {
22 int paramIndex;
23
24 paramIndex = 0;
25 while (paramIndex < spatialParams->numParameters && spatialParams->parameters[paramIndex].value != NULL)
26 // while we haven't reached the end of the parameters vector, and we haven't found an unitialized parameter
27 paramIndex++;
28
29 return paramIndex;
30}

53/* If isSpatial = 0, set array[0] = value (ignore loc)
54 Otherwise, if loc = -1, set array[0..length-1] = value
55 Otherwise, set array[loc] = value

57void setPossiblySpatial(double *array, int loc, double value, int isSpatial, int length) {
58 if (isSpatial) {
59 if (loc == -1)
60 setAll(array, length, value);

63 }
64 else // non-spatial
65 array[0] = value;
66}

73// allocate space for a new spatialParams structure, return a pointer to it
74SpatialParams *newSpatialParams(int numParameters, int numLocs) {
75 SpatialParams *spatialParams;
76 int i;
77
78 spatialParams = (SpatialParams *)malloc(sizeof(SpatialParams));
79 spatialParams->numParameters = numParameters;
80 spatialParams->numLocs = numLocs;
81 spatialParams->numChangeableParams = 0; // for now, anyway
82
83 // allocate space for vectors within this structure:
84 spatialParams->parameters = (OneSpatialParam *)malloc(numParameters * sizeof(OneSpatialParam));
85 spatialParams->changeableParamIndices = (int *)malloc(numParameters * sizeof(int)); // the biggest it could have to be
86
87 // set value pointers of all parameters to be NULL (this will mark uninitialized parameter):
88 for (i = 0; i < numParameters; i++)
89 spatialParams->parameters[i].value = NULL;
90

104/* Read next parameter from file, setting all values in spatialParams structure appropriately
105 Ensure that name is the name of the parameter we're reading
106 Set this parameter's externalLoc pointer to point to the location given by the externalLoc parameter
107 Note: acceptable to have externalLoc = NULL, but then (obviously) won't be assigned by loadSpatialParams
108
109 Structure of paramFile:
110 name value changeable min max sigma
111 where value is a number if parameter is non-spatial, or * if parameter is spatial; changeable is 0 or 1 (boolean)
112 If parameter is spatial, look in spatialParamFile for value, which has the following structure:
113 name value [value value...]
114 where the number of values is equal to the number of spatial locations
115
116 PRE: paramFile is open and file pointer points to the next parameter to be read
117 if this is a spatial parameter, spatialParamFile is open and file pointer points to the next parameter to be read
118*/
119void readOneSpatialParam(SpatialParams *spatialParams, FILE *paramFile, FILE *spatialParamFile, char *name, double *externalLoc) {
120 const char *TOKENS = " \t"; // tokens that can separate values in spatialParams file (space & tab)
121
122 char line[256];
123 char pName[64]; // parameter name
124 char strValue[32]; // before we know whether value is a number or "*"
125 double value, min, max, sigma;
126 int changeable;
127 char *errc;
128 int status;
129 int paramIndex; // index of next uninitialized parameter
130 OneSpatialParam *param; // a pointer to the next uninitialized parameter
131 int numLocs, i;
132
133 if (allParamsRead(spatialParams)) { // all parameters have been read
134 printf("Error trying to read %s: no more uninitialized parameters to fill\n", name);
135 exit(1);
136 }
137
138 // otherwise, get the index of the next uninitialized parameter
139 paramIndex = nextUnreadParam(spatialParams);
140 // and set param to point to it for easier access
141 param = &(spatialParams->parameters[paramIndex]);
142
143 // read next line:
144 if (fgets(line, sizeof(line), paramFile) == NULL) {
145 printf("Error trying to read %s: error or unexpected EOF in parameter file\n", name);
146 printf("feof = %d, ferror = %d\n", feof(paramFile), ferror(paramFile));
147 exit(1);
148 }
149
150 // tokenize line:
151 strcpy(pName, strtok(line, TOKENS)); // copy first token into pName
152 if (strcasecmp(pName, name) !=0) { // compare, ignore case
153 printf("Error: read %s, expected %s\n", pName, name);
154 exit(1);
155 }
156
157 // continue to tokenize:
158 strcpy(strValue, strtok(NULL, TOKENS)); // copy next token into strValue; wait until later to figure out if it's "*" or a number
159 changeable = strtol(strtok(NULL, TOKENS), &errc, 0);
160 min = strtod(strtok(NULL, TOKENS), &errc);
161 max = strtod(strtok(NULL, TOKENS), &errc);
162 sigma = strtod(strtok(NULL, TOKENS), &errc);
163
164 // fill the new spatialParam structure with everything but numLocs, value, guess, best, and knob (which we'll do later)
165 strcpy(param->name, pName);
166 param->isChangeable = changeable;
167 param->externalLoc = externalLoc;
168 param->min = min;
169 param->max = max;
170 param->sigma = sigma;
171
172 if (changeable) { // we need to update info on changeable params in spatialParams
173 spatialParams->changeableParamIndices[spatialParams->numChangeableParams] = paramIndex;
174 // before we change it, spatialParams->numChangeableParams is the index of the first free spot in the changeableParamIndices vector
175 spatialParams->numChangeableParams++;
176 }
177
178 // now we need to see if we read in an actual value, or a "*" (if the latter, it's a spatially-varying parameter)
179 if (strcmp(strValue, "*") == 0) { // we read "*": this is a spatially-varying parameter
180 numLocs = param->numLocs = spatialParams->numLocs;
181
182 // read info from spatialParamFile:
183 status = fscanf(spatialParamFile, "%s", pName);
184 if (status == EOF || status == 0) { // error reading
185 printf("Error trying to read %s from spatialParamFile\n", name);
186 exit(1);
187 }
188 else if (strcasecmp(pName, name) != 0) { // compare, ignore case
189 printf("Error: read %s from spatialParamFile, expected %s\n", pName, name);
190 exit(1);
191 }
192
193 // we've read correct name, now read all the values:
194 // first, allocate space:
195 param->value = (double *)malloc(numLocs * sizeof(double));
196 param->guess = (double *)malloc(numLocs * sizeof(double));
197 param->best = (double *)malloc(numLocs * sizeof(double));
198 param->knob = (double *)malloc(numLocs * sizeof(double));
199
200 // now read values and assign param->value, guess, best, and knob:
201 for (i = 0; i < numLocs; i++) {
202 status = fscanf(spatialParamFile, "%lf", &value);
203 if (status == EOF || status == 0) { // error reading
204 printf("Error: did not find enough values in spatialParamFile for %s\n", pName);
205 exit(1);
206 }
207
208 // assign value, guess and best to all be the guess value initially:
209 param->value[i] = param->guess[i] = param->best[i] = value;
210 param->knob[i] = 0; // assign knob to be 0 initially
211 } // for i
212 } // if spatially-varying
213 else { // non-spatially-varying parameter
214 param->numLocs = 0; // signifies non-spatially-varying
215
216 // allocate space for a single value in each of param->value, guess, best and knob:
217 param->value = (double *)malloc(sizeof(double));
218 param->guess = (double *)malloc(sizeof(double));
219 param->best = (double *)malloc(sizeof(double));
220 param->knob = (double *)malloc(sizeof(double));
221
222 // assign value, guess and best to all be the guess value initially:
223 value = strtod(strValue, &errc); // convert string value to double
224 param->value[0] = param->guess[0] = param->best[0] = value;
225 param->knob[0] = 0; // assign knob to be 0 initially
226 } // else non-spatially-varying
227
228} // readOneSpatialParam

349 If loc = -1, set best value of all parameters at ALL locations to be equal to their current value
350*/
351void setAllSpatialParamBests(SpatialParams *spatialParams, int loc) {
352 int numParams;
353 int firstLoc, lastLoc;
354 int i, currLoc;
355 double value;
356
357 numParams = spatialParams->numParameters;
358 if (loc == -1) { // set best at all locations
359 firstLoc = 0;

363 firstLoc = lastLoc = loc;

366 if (isSpatial(spatialParams, i)) {
367 for (currLoc = firstLoc; currLoc <= lastLoc; currLoc++) {
368 value = getSpatialParam(spatialParams, i, currLoc);
369 setSpatialParamBest(spatialParams, i, currLoc, value);
370 }
371 }
372 else { // non-spatial
373 value = getSpatialParam(spatialParams, i, 0);
374 setSpatialParamBest(spatialParams, i, 0, value);

392/* Load spatial parameters into memory locations pointed to by paramPtrs
393 For spatially-varying parameters, load value at location given by loc
394 (Purpose: load all parameters into a localized structure for a model run, to make the model run more efficient)
395
396 PRE: spatialParams->parameters is loaded with param. values (particularly, at given location)
397 externalLoc pointers have been set for each parameter
398 0 <= loc < spatialParams->numLocs
399*/
400void loadSpatialParams(SpatialParams *spatialParams, int loc) {
401 int i, numParams;
402 double value;
403
404 numParams = spatialParams->numParameters;
405 for (i = 0; i < numParams; i++) {
406 // value will equal NULL if it hasn't been set yet; externalLoc will equal NULL if it was assigned to NULL in readOneSpatialParam
407 if (spatialParams->parameters[i].value != NULL && spatialParams->parameters[i].externalLoc != NULL) {
408 value = getSpatialParam(spatialParams, i, loc);
409 *(spatialParams->parameters[i].externalLoc) = value; // copy value into location pointed to by this externalLoc
410 }
411 }
412}

415/* If randomReset = 0, set values of all parameters to be equal to guess values
416 If randomReset non-zero, set parameter values to be somewhere (chosen uniform randomly) between min and max
417 (If randomReset non-zero, random number generator must have been seeded)
418 Set best values of all parameters to be equal to current values
419 Set knobs of all parameters to be equal to knob argument
420*/
421void resetSpatialParams(SpatialParams *spatialParams, double knob, int randomReset) {
422 int numParams, numLocs;
423 int i, loc;
424 double value;
425
426 numParams = spatialParams->numParameters;
427 numLocs = spatialParams->numLocs;
428
429 for (i = 0; i < numParams; i++) {
430 if (isSpatial(spatialParams, i)) {
431 for (loc = 0; loc < numLocs; loc++) {
432 if (randomReset == 0) // non-random: use guess value
433 value = getSpatialParamGuess(spatialParams, i, loc);
434 else // random: set parameter value to be somewhere between min and max
435 value = getSpatialParamMin(spatialParams, i) + (getSpatialParamMax(spatialParams, i) - getSpatialParamMin(spatialParams, i))
436 * ((float)rand()/RAND_MAX);
437 setSpatialParam(spatialParams, i, loc, value);
438 setSpatialParamBest(spatialParams, i, loc, value);
439 setSpatialParamKnob(spatialParams, i, loc, knob);
440 }
441 }
442 else { // non-spatial
443 if (randomReset == 0) // non-random: use guess value
444 value = getSpatialParamGuess(spatialParams, i, 0);
445 else // random: set parameter value to be somewhere between min and max
446 value = getSpatialParamMin(spatialParams, i) + (getSpatialParamMax(spatialParams, i) - getSpatialParamMin(spatialParams, i))
447 * ((float)rand()/RAND_MAX);
448 setSpatialParam(spatialParams, i, 0, value);
449 setSpatialParamBest(spatialParams, i, 0, value);
450 setSpatialParamKnob(spatialParams, i, 0, knob);
451 }
452 }

456/* Write best parameter values, and other parameter info, to files
457 Note that other parameter info won't have changed since the read - we just copy that over
458 Write all parameter info to file with name *paramFile, and spatially-varying parameter values to file with name *spatialParamFile
459 See readOneSpatialParam for file formats
460*/
461void writeBestSpatialParams(SpatialParams *spatialParams, char *paramFile, char *spatialParamFile) {
462 FILE *paramF, *spatialParamF;
463 int numParams, numLocs;
464 int i, j;
465
466 paramF = openFile(paramFile, "w");
467 spatialParamF = openFile(spatialParamFile, "w");
468
469 numParams = spatialParams->numParameters;
470 numLocs = spatialParams->numLocs;
471
472 fprintf(spatialParamF, "%d\n", numLocs);
473
474 for (i = 0; i < numParams; i++) { // loop through parameters
475 fprintf(paramF, "%s\t", spatialParams->parameters[i].name);
476 if (isSpatial(spatialParams, i)) {
477 fprintf(paramF, "*\t"); // write "*" in place of value, then write values to spatialParamFile
478 fprintf(spatialParamF, "%s", spatialParams->parameters[i].name);
479 for (j = 0; j < numLocs; j++) // loop through locations
480 fprintf(spatialParamF, "\t%f", getSpatialParamBest(spatialParams, i, j));
481 fprintf(spatialParamF, "\n");
482 }
483 else // non-spatial: write value
484 fprintf(paramF, "%f\t", getSpatialParamBest(spatialParams, i, 0));
485
486 // now write other info to file (other info is always non-spatial):
487 fprintf(paramF, "%d\t%f\t%f\t%f\n", isChangeable(spatialParams, i), getSpatialParamMin(spatialParams,i),
488 getSpatialParamMax(spatialParams, i), getSpatialParamSigma(spatialParams, i));
489 }
490
491 fclose(paramF);
492 fclose(spatialParamF);
493}

533// Clean up: deallocate spatialParams and any other dynamically-allocated pointers that need deallocating
534void deleteSpatialParams(SpatialParams *spatialParams) {
535 int numParameters, i;
536 OneSpatialParam *param;
537
538 // first loop through individual parameters, de-allocating space they use:
539 numParameters = spatialParams->numParameters;
540 for (i = 0; i < numParameters; i++) {
541 param = &(spatialParams->parameters[i]);
542 if (param->value != NULL) { // this parameter has been used - so space has been allocated for its various values
543 free(param->value);
544 free(param->guess);
545 free(param->best);
546 free(param->knob);
547 }
548 }
549
550 // now free space used by spatialParams structure itself:
551 free(spatialParams->parameters);
552 free(spatialParams->changeableParamIndices);
553}
