40// initialize data indices array to [0,1,...,MAX_DATA_TYPES-1]
41// returns size of array (= MAX_DATA_TYPES)
42int initDataTypeIndices(int dataTypeIndices[]) {
43 int i;
44 for (i = 0; i < MAX_DATA_TYPES; i++)
45 dataTypeIndices[i] = i;
46
47 return MAX_DATA_TYPES;
48}

51/* PRE: dataTypeSwitches is of size MAX_DATA_TYPES
52 dataTypeSwitches[i] is 1 if we are including this data type in optimization, 0 if not
53 dataTypeIndices must be big enough to hold MAX_DATA_TYPES ints
54
55 POST: dataTypeIndices[0..numDataTypes-1] contains the indices of the 1-valued elements in dataTypeSwitches
56 - i.e. all of the data types included in optimization
57 (remaining elements of dataTypeIndices are undefined)
58 returns numDataTypes: number of data types included in optimization
59 */
60int parseDataTypeSwitches(int dataTypeIndices[], int dataTypeSwitches[]) {
61 int numDataTypes;
62 int i;
63
64 numDataTypes = 0;
65 for (i = 0; i < MAX_DATA_TYPES; i++) {
66 if (dataTypeSwitches[i]) {
67 dataTypeIndices[numDataTypes] = i;
68 numDataTypes++;
69 }
70 }
71
72 return numDataTypes;
73}
