66/* Allocate space for a new outputItems structure, return a pointer to it
67 filenameBase is the base name of the files to which we'll output
68 - we'll output to <filenameBase>.<name> for each output item
69 separator is the character separating values in the output files (e.g. space,
70 tab, or comma)
71 */
72OutputItems *newOutputItems(char *filenameBase, char separator) {
73 OutputItems *outputItems;
74
75 outputItems = (OutputItems *)malloc(sizeof(OutputItems));
76
77 outputItems->head = newSingleOutputItem("", NULL); // we'll keep a dummy item
78 // at the head of the list
79 outputItems->tail = outputItems->head;
80 outputItems->count = 0;
81
82 outputItems->filenameBase =
83 (char *)malloc((strlen(filenameBase) + 1) * sizeof(char));
84 strcpy(outputItems->filenameBase, filenameBase);
85
86 outputItems->separator = separator;
87
88 return outputItems;
89}

91/* Add a new singleOutputItem to the end of the list given by outputItems
92 strlen(name) must be < OUTPUT_ITEMS_MAXNAME
93 ptr must be a pointer to the variable holding this item (double)
94
95 After calling this function, the file associated with this output item will

98void addOutputItem(OutputItems *outputItems, char *name, double *ptr) {
99 SingleOutputItem *singleOutputItem;
100
101 singleOutputItem = newSingleOutputItem(name, ptr);
102 openOutputItemFile(singleOutputItem, outputItems->filenameBase);
