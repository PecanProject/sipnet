37// Checks to make sure that all parameters with isRequired=true have been read
38// If not, kills program
39// Writes out names of all parameters that weren't read (even if not required,
40// as a warning message)
41void checkAllRead(SpatialParams *spatialParams) {
42 int i;
43 int okay;
44 OneSpatialParam *param;
45
46 okay = 1; // so far so good
47 for (i = 0; i < spatialParams->numParameters; i++) {
48 param = &(spatialParams->parameters[i]);
49 if (param->value == NULL) {
50 if (param->isRequired) { // should have been read but wasn't!
51 okay = 0;
52 printf("ERROR: Didn't read required parameter %s\n", param->name);
53 } else {
54 printf("WARNING: Didn't read parameter %s (not flagged as required, so "
55 "continuing)\n",
56 param->name);
57 }
58 }
59 }
60
61 if (!okay) {
62 exit(1);
63 }
64}

95/* Read array of spatial values from file, for a single parameter
96 Values are returned in the array spatialValues, which must be at least of
97 length numLocs spatialParamFile is the (already-open) file of spatial
98 parameters
99 - the file pointer should be pointing to the start of the line we want to
100 read pName gives the name of the parameter; we'll abort if the next parameter
101 does not have the name given by pName numLocs is the number of values to read
102 */
103void readSpatialValues(double *spatialValues, FILE *spatialParamFile,
104 char pName[PARAM_MAXNAME], int numLocs) {
105 char pNameSpatial[PARAM_MAXNAME];
106 int status;
107 int i;
108
109 // read name from spatialParamFile, make sure it's what we expect:
110 status = fscanf(spatialParamFile, "%s", pNameSpatial);
111 if (status == EOF || status == 0) { // error reading
112 printf("Error trying to read %s from spatialParamFile\n", pName);
113 exit(1);
114 } else if (strcmpIgnoreCase(pName, pNameSpatial) != 0) { // compare, ignore
115 // case
116 printf("Error: read %s from spatialParamFile, expected %s\n", pNameSpatial,
117 pName);
118 exit(1);
119 }
120
121 // now read values and assign param->value, guess, best, and knob:
122 for (i = 0; i < numLocs; i++) {
123 status = fscanf(spatialParamFile, "%lf", &(spatialValues[i]));
124 if (status == EOF || status == 0) { // error reading
125 printf("Error: did not find enough values in spatialParamFile for %s\n",
126 pName);
127 exit(1);
128 }
129 }
130}

160/* Initialize next spatial parameter:
161 Set name of parameter equal to "name",
162 and set parameter's externalLoc pointer to point to the location given by
163 "externalLoc" parameter Also set parameter's isRequired value: if true, then
164 we'll terminate execution if this parameter is not read from file Note:
165 acceptable to have externalLoc = NULL, but then it won't be assigned by
166 loadSpatialParams
167*/
168void initializeOneSpatialParam(SpatialParams *spatialParams, char *name,
169 double *externalLoc, int isRequired) {
170 int paramIndex; // index of next uninitialized parameter
171 OneSpatialParam *param; // a pointer to the next uninitialized parameter
172
173 if (allParamsInitialized(spatialParams)) { // all parameters have been
174 // initialized!
175 printf("Error trying to initialize %s: have already initialized all %d "
176 "parameters\n",
177 name, spatialParams->maxParameters);
178 printf(
179 "Check value of maxParameters passed into newSpatialParams function\n");
180 exit(1);
181 }
182
183 // otherwise, get the index of the next uninitialized parameter
184 paramIndex = spatialParams->numParameters;
185 // and set param to point to it for easier access
186 param = &(spatialParams->parameters[paramIndex]);
187
188 strcpy(param->name, name);
189 param->externalLoc = externalLoc;
190 param->isRequired = isRequired;
191
192 // set value pointer to be NULL (this will mark unread parameter)
193 param->value = NULL;
194
195 spatialParams->numParameters++;
196}

198/* Read all parameters from file, setting all values in spatialParams structure
199 appropriately
200
201 Structure of paramFile:
202 name value changeable min max sigma
203 where value is a number if parameter is non-spatial, or * if parameter is
204 spatial; changeable is 0 or 1 (boolean) If parameter is spatial, look in
205 spatialParamFile for value, which has the following structure: name value
206 [value value...] where the number of values is equal to the number of
207 spatial locations
208
209 The order of the parameters in paramFile does not matter,
210 but parameters must be specified in the same order in paramFile and
211 spatialParamFile
212
213 ! is a comment character in paramFile: anything after a ! on a line is
214 ignored note, though, that comments are not currently allowed in
215 spatialParamFile
216
217 PRE: paramFile is open and file pointer points to start of file
218 spatialParamFile is open and file pointer points to 2nd line (after the
219 numLocs line)
220 */
221void readSpatialParams(SpatialParams *spatialParams, FILE *paramFile,
222 FILE *spatialParamFile) {
223 const char *SEPARATORS = " \t\n\r"; // characters that can separate values in
224 // parameter files
225 const char *COMMENT_CHARS = "!"; // comment characters (ignore everything
226 // after this on a line)
227
228 char line[256];
229 char pName[PARAM_MAXNAME]; // parameter name
230 int paramIndex;
231 OneSpatialParam *param; // a pointer to a single parameter, for easier access
232 char strValue[32]; // before we know whether value is a number or "*"
233 double value, min, max, sigma;
234 double *spatialValues; // temporary storage for values read from spatial
235 // param file
236 int changeable, spatiallyVarying; // is the parameter changeable? is it
237 // spatially-varying?
238 char *errc;
239 int isComment;
240 int numLocs, i;
241
242 numLocs = spatialParams->numLocs; // for those parameters that are
243 // spatially-varying
244 spatialValues = (double *)malloc(numLocs * sizeof(double));
245
246 while (fgets(line, sizeof(line), paramFile) != NULL) { // while not EOF or
247 // error
248 // remove trailing comments:
249 isComment = stripComment(line, COMMENT_CHARS);
250
251 if (!isComment) { // if this isn't just a comment line or blank line
252 // tokenize line:
253 strcpy(pName, strtok(line, SEPARATORS)); // copy first token into pName
254 strcpy(strValue, strtok(NULL, SEPARATORS)); // copy next token into
255 // strValue; wait until later
256 // to figure out if it's "*"
257 // or a number
258 changeable = strtol(strtok(NULL, SEPARATORS), &errc, 0);
259 min = strtod(strtok(NULL, SEPARATORS), &errc);
260 max = strtod(strtok(NULL, SEPARATORS), &errc);
261 sigma = strtod(strtok(NULL, SEPARATORS), &errc);
262
263 // now we need to see if we read in an actual value, or a "*" (if the
264 // latter, it's a spatially-varying parameter) note that we read this
265 // before doing the error checking on paramIndex:
266 // that way, even if we ignore this parameter, we've still read (and
267 // ignored) the corresponding line in the spatial param file, if it's a
268 // spatially-varying parameter
269 if (strcmp(strValue, "*") == 0) { // we read "*": this is a
270 // spatially-varying parameter
271 readSpatialValues(spatialValues, spatialParamFile, pName, numLocs);
272 spatiallyVarying = 1;
273 } else {
274 spatiallyVarying = 0;
275 }
276
277 paramIndex = locateParam(spatialParams, pName);
278
279 if (paramIndex == -1) { // not found
280 printf("WARNING: Ignoring parameter %s: this parameter wasn't "
281 "initialized in the code\n",
282 pName);
283 } else if (valueSet(spatialParams, paramIndex)) {
284 printf("Error reading parameter file: read %s, but this parameter has "
285 "already been set\n",
286 pName);
287 exit(1);
288 } else { // otherwise, we're good to go
289
290 // set param to point to the appropriate parameter, for easier access
291 param = &(spatialParams->parameters[paramIndex]);
292
293 // mark this as the next parameter read from file:
294 spatialParams->readIndices[spatialParams->numParamsRead] = paramIndex;
295 spatialParams->numParamsRead++;
296
297 // fill the new spatialParam structure with changeable, min, max, and
298 // sigma
299 param->isChangeable = changeable;
300 param->min = min;
301 param->max = max;
302 param->sigma = sigma;
303
304 if (changeable) { // we need to update info on changeable params in
305 // spatialParams
306 spatialParams
307 ->changeableParamIndices[spatialParams->numChangeableParams] =
308 paramIndex;
309 // before we change it, spatialParams->numChangeableParams is the
310 // index of the first free spot in the changeableParamIndices vector
311 spatialParams->numChangeableParams++;
312 }
313
314 if (spatiallyVarying) {
315 param->numLocs = numLocs;
316
317 // allocate space:
318 param->value = (double *)malloc(numLocs * sizeof(double));
319 param->guess = (double *)malloc(numLocs * sizeof(double));
320 param->best = (double *)malloc(numLocs * sizeof(double));
321 param->knob = (double *)malloc(numLocs * sizeof(double));
322
323 // now assign param->value, guess, best, and knob (these were read
324 // above, in call to readSpatialValues):
325 for (i = 0; i < numLocs; i++) {
326 // assign value, guess and best to all be the guess value initially:
327 param->value[i] = param->guess[i] = param->best[i] =
328 spatialValues[i];
329 param->knob[i] = 0; // assign knob to be 0 initially
330 } // for i
331 } // if spatially-varying
332 else { // non-spatially-varying parameter
333 param->numLocs = 0; // signifies non-spatially-varying
334
335 // allocate space for a single value in each of param->value, guess,
336 // best and knob:
337 param->value = (double *)malloc(sizeof(double));
338 param->guess = (double *)malloc(sizeof(double));
339 param->best = (double *)malloc(sizeof(double));
340 param->knob = (double *)malloc(sizeof(double));
341
342 // assign value, guess and best to all be the guess value initially:
343 value = strtod(strValue, &errc); // convert string value to double
344 param->value[0] = param->guess[0] = param->best[0] = value;
345 param->knob[0] = 0; // assign knob to be 0 initially
346 } // else non-spatially-varying
347 } // else (no errors in reading this line from parameter file)
348 } // if !isComment
349 } // while not EOF or error
350
351 // check for error in reading:
352 if (ferror(paramFile)) {
353 printf("Error reading file in readSpatialParams\n");
354 printf("ferror = %d\n", ferror(paramFile));
355 exit(1);
356 }
357
358 checkAllRead(spatialParams); // terminate program if some required parameters
359 // weren't read
360
361 free(spatialValues);
362} // readSpatialParams

376// If found, return index in vector, otherwise return -1
377int locateParam(SpatialParams *spatialParams, char *name) {
378 int i;
379 int found;
380
381 i = 0;
382 found = 0;
383 while (i < spatialParams->numParameters && !found) {
384 if (strcmpIgnoreCase(name, spatialParams->parameters[i].name) == 0) {
385 found = 1;
386 }
387 i++;
388 }
389
390 if (found) {
391 return (i - 1);
392 }
393
394 return -1;

522void setAllSpatialParamBests(SpatialParams *spatialParams, int loc) {

541 for (currLoc = firstLoc; currLoc <= lastLoc; currLoc++) {

568/* Load spatial parameters into memory locations pointed to by paramPtrs
569 For spatially-varying parameters, load value at location given by loc
570 (Purpose: load all parameters into a localized structure for a model run, to
571 make the model run more efficient)

573 PRE: spatialParams->parameters is loaded with param. values (particularly, at
574 given location) externalLoc pointers have been set for each parameter 0 <= loc
575 < spatialParams->numLocs

577void loadSpatialParams(SpatialParams *spatialParams, int loc) {
578 int i, numParams;

580

594/* If randomReset = 0, set values of all parameters to be equal to guess values
595 If randomReset non-zero, set parameter values to be somewhere (chosen uniform
596 randomly) between min and max
597 - Note that non-changeable parameters will still be set to their guess
598 values, though (Note: ignores parameters that were never read in) (If
599 randomReset non-zero, random number generator must have been seeded)
600 Set best values of all parameters to be equal to current values Set knobs of
601 all parameters to be equal to knob argument
602*/
603void resetSpatialParams(SpatialParams *spatialParams, double knob,
604 int randomReset) {
605 int numParams, numLocs;
606 int i, loc;
607 double value;
608
609 numParams = spatialParams->numParameters;
610 numLocs = spatialParams->numLocs;
611
612 for (i = 0; i < numParams; i++) {
613 if (valueSet(spatialParams, i)) { // if valueSet is false, ignore this
614 // parameter (it was never read in)
615 if (isSpatial(spatialParams, i)) {
616 for (loc = 0; loc < numLocs; loc++) {
617 if ((randomReset == 0) || !isChangeable(spatialParams, i)) {
618 // non-random: use guess value
619 value = getSpatialParamGuess(spatialParams, i, loc);
620 } else {
621 // random: set parameter value to be somewhere between min and max
622 value = getSpatialParamMin(spatialParams, i) +
623 (getSpatialParamMax(spatialParams, i) -
624 getSpatialParamMin(spatialParams, i)) *
625 ((float)rand() / (float)RAND_MAX);
626 }
627 setSpatialParam(spatialParams, i, loc, value);
628 setSpatialParamBest(spatialParams, i, loc, value);
629 setSpatialParamKnob(spatialParams, i, loc, knob);
630 }
631 } else { // non-spatial
632 if ((randomReset == 0) || !isChangeable(spatialParams, i)) {
633 // non-random: use guess value
634 value = getSpatialParamGuess(spatialParams, i, 0);
635 } else {
636 // random: set parameter value to be somewhere between min and max
637 value = getSpatialParamMin(spatialParams, i) +
638 (getSpatialParamMax(spatialParams, i) -
639 getSpatialParamMin(spatialParams, i)) *
640 ((float)rand() / (float)RAND_MAX);
641 }
642 setSpatialParam(spatialParams, i, 0, value);
643 setSpatialParamBest(spatialParams, i, 0, value);
644 setSpatialParamKnob(spatialParams, i, 0, knob);
645 } // else non-spatial
646 } // if valueSet
647 } // for i
648} // resetSpatialParams

650/* Write best parameter values, and other parameter info, to files
651 (only write parameters that were read in, and in the same order that they
652 were read in) Note that other parameter info won't have changed since the
653 read - we just copy that over Write all parameter info to file with name
654 *paramFile, and spatially-varying parameter values to file with name
655 *spatialParamFile See readSpatialParams for file formats
656*/
657void writeBestSpatialParams(SpatialParams *spatialParams, char *paramFile,
658 char *spatialParamFile) {
659 FILE *paramF, *spatialParamF;
660 int numParamsRead, numLocs;
661 int i, j, index;
662
663 paramF = openFile(paramFile, "w");
