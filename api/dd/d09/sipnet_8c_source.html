514void readClimData(const char *climFile) {
515 FILE *in;
516 ClimateNode *curr, *next;
517 int year, day;
518 int lastYear = -1;
519 double time, length; // time in hours, length in days (or fraction of day)
520 double tair, tsoil, par, precip, vpd, vpdSoil, vPress, wspd, soilWetness;
521
522#if GDD
523 double thisGdd; // growing degree days of this time step
524 double gdd = 0.0; // growing degree days since the last Jan. 1
525#endif
526
527 int status; // status of the read
528
529 // for format check
530 int firstLoc, dummyLoc, numFields;
531 int expectedNumCols = NUM_CLIM_FILE_COLS;
532 char *firstLine = NULL;
533 size_t lineCap = 0;
534 int hasLoc = 0;
535 const char *SEPARATORS = " \t\n\r"; // characters that can separate values in
536 // parameter files
537
538 in = openFile(climFile, "r");
539
540 // Check format of first line to see if location is still specified (we will
541 // ignore it if so)
542 if (getline(&firstLine, &lineCap, in) == -1) { // EOF
543 printf("Error: no climate data in %s\n", climFile);
544 exit(EXIT_CODE_INPUT_FILE_ERROR);
545 }
546
547 numFields = countFields(firstLine, SEPARATORS);
548 if (numFields == expectedNumCols + 1) {
549 logWarning("ignoring location column in %s (found %d cols)\n", climFile,
550 numFields);
551 ++expectedNumCols;
552 hasLoc = 1;
553 }
554
555 if (hasLoc) {
556 status = sscanf(firstLine, // NOLINT
557 "%d %d %d %lf %lf %lf %lf %lf %lf %lf %lf %lf %lf %lf",
558 &firstLoc, &year, &day, &time, &length, &tair, &tsoil, &par,
559 &precip, &vpd, &vpdSoil, &vPress, &wspd, &soilWetness);
560 } else {
561 status = sscanf(firstLine, // NOLINT
562 "%d %d %lf %lf %lf %lf %lf %lf %lf %lf %lf %lf %lf", &year,
563 &day, &time, &length, &tair, &tsoil, &par, &precip, &vpd,
564 &vpdSoil, &vPress, &wspd, &soilWetness);
565 }
566 free(firstLine);
567
568 if (status != expectedNumCols) {
569 printf("Error reading climate file: bad data on first line\n");
570 exit(EXIT_CODE_INPUT_FILE_ERROR);
571 }
572
573 firstClimate = (ClimateNode *)malloc(sizeof(ClimateNode));
574 next = firstClimate;
575
576 while (status != EOF) {
577 // we have another day's climate
578 curr = next;
579
580 curr->year = year;
581 curr->day = day;
582 curr->time = time;
583
584 if (length < 0) { // parse as seconds
585 length = length / -86400.; // convert to days
586 }
587 curr->length = length;
588
589 curr->tair = tair;
590 curr->tsoil = tsoil;
591 curr->par = par * (1.0 / length);
592 // convert par from Einsteins * m^-2 to Einsteins * m^-2 * day^-1
593 curr->precip = precip * 0.1; // convert from mm to cm
594 curr->vpd = vpd * 0.001; // convert from Pa to kPa
595 if (curr->vpd < TINY) {
596 curr->vpd = TINY; // avoid divide by zero
597 }
598 curr->vpdSoil = vpdSoil * 0.001; // convert from Pa to kPa
599 curr->vPress = vPress * 0.001; // convert from Pa to kPa
600 curr->wspd = wspd;
601 if (curr->wspd < TINY) {
602 curr->wspd = TINY; // avoid divide by zero
603 }
604 curr->soilWetness = soilWetness;
605
606#if GDD
607 if (year != lastYear) { // HAPPY NEW YEAR!
608 gdd = 0; // reset growing degree days
609 }
610 thisGdd = tair * length;
611 if (thisGdd < 0) { // can't have negative growing degree days
612 thisGdd = 0;
613 }
614 gdd += thisGdd;
615 curr->gdd = gdd;
616#endif
617
618 lastYear = year;
619
620 if (hasLoc) {

906void calcLightEff(double *lightEff, double lai, double par) {
907
908 // Information on the distribution of LAI with height is available
909 // as of March 2007 ... contact Dr. Maggie Prater Maggie.Prater@colorado.edu
910
911 static const int NUM_LAYERS = 6;
912 // believe it or not, 6 layers gives approximately the same result as 100
913 // layers
914
915 int layer; // counter
916 double cumLai; // lai from this layer up
917 double lightIntensity;
918 double currLightEff = 0.0, cumLightEff;
919 double currfAPAR = 0.0, cumfAPAR;
920 int coeff; // current coefficient in Simpson's rule
921
922 if (lai > 0 && par > 0) { // must have at least some leaves and some light
923 cumLightEff = 0.0; // the running sum
924 layer = 0;
925 coeff = 1;
926 int err;
927 double fAPAR; // Calculation of fAPAR according to 1 - exp(attenuation*LAI)
928 // double APAR; // Absorbed PAR by the canopy
929 // double lightIntensityTop, lightIntensityBottom; // PAR absorbed by the
930 // canopy
931 cumfAPAR = 0.0;
932
933 while (layer <= NUM_LAYERS) {
934 cumLai = lai * ((double)layer / NUM_LAYERS); // lai from this layer up
935 // (starting at top)
936
937 // between 0 and par
938 lightIntensity = par * exp(-1.0 * params.attenuation * cumLai);
939
940 // between 0 and 1
941 currLightEff = (1 - pow(2, (-1.0 * lightIntensity / params.halfSatPar)));
942
943 // when lightIntensity = halfSatPar, currLightEff = 1/2
944 cumLightEff += coeff * currLightEff;
945
946 currfAPAR = 1 - (lightIntensity / par);
947 cumfAPAR += coeff * currfAPAR;
948
949 // now move to the next layer:
950 layer++;
951 coeff = 2 * (1 + layer % 2); // coeff. goes 1, 4, 2, 4, ..., 2, 4, 2
952 }
953 // last value should have had a coefficient of 1, but actually had a
954 // coefficient of 2, so subtract 1:
955 cumLightEff -= currLightEff;
956 *lightEff = cumLightEff / (3.0 * NUM_LAYERS); // multiplying by (h/3) in
957 // Simpson's rule
958
959 // now calculate fAPAR
960 cumfAPAR -= currfAPAR;
961 // the average value, multiplying by h/3 in Simpson's rule, and dividing by
962 // the number of steps
963 fAPAR = cumfAPAR / (3.0 * NUM_LAYERS);
964
965 err = addValueToMeanTracker(meanFPAR, fAPAR, 1); // update running mean of
966 // FPAR (we don't care
967 // about climate length)
968 if (err != 0) {
969 printf("******* Error type %d while trying to add value to FPAR mean "
970 "tracker in sipnet:potPSN() *******\n",
971 err);
972 printf("FPAR = %f, climate->length = %f\n", fAPAR, climate->length);
973 printf("Suggestion: try changing MEAN_FPAR_MAX_ENTRIES in sipnet.c\n");
974 exit(1);
975 }
976
977 } else { // no leaves or no light!
978 *lightEff = 0;
979 }
980}

982// calculate gross photosynthesis without water effect (g C * m^-2 ground area *
983// day^-1) and base foliar respiration without temp, water, etc. (g C * m^-2
984// ground area * day^-1)
985void potPsn(double *potGrossPsn, double *baseFolResp, double lai, double tair,
986 double vpd, double par, int /*day*/) {
987 double grossAMax; // maximum possible gross respiration (nmol CO2 * g^-1 leaf
988 // * sec^-1)
989 double dTemp, dVpd, lightEff; // decrease in photosynth. due to temp, vpd and
990 // amt. of light absorbed
991 double respPerGram; // base foliar respiration in nmol CO2 * g^-1 leaf *
992 // sec^-1
993 double conversion; /* convert from (nmol CO2 * g^-1 leaf * sec^-1)
994 to (g C * m^-2 ground area * day^-1) */
995
996 respPerGram = params.baseFolRespFrac * params.aMax;
997 // foliar respiration, unmodified by temp, etc.
998 grossAMax = params.aMax * params.aMaxFrac + respPerGram;
999
1000 dTemp = (params.psnTMax - tair) * (tair - params.psnTMin) /
1001 pow((params.psnTMax - params.psnTMin) / 2.0, 2);
1002 if (dTemp < 0) {
1003 dTemp = 0.0;
1004 }
1005 dVpd = 1.0 - params.dVpdSlope * pow(vpd, params.dVpdExp);
1006 if (dVpd < 0) {
1007 dVpd = 0.0;
1008 }
1009 calcLightEff(&lightEff, lai, par);
1010
1011 conversion = C_WEIGHT * (1.0 / TEN_9) *
1012 (params.leafCSpWt / params.cFracLeaf) * lai *
1013 SEC_PER_DAY; // to convert units
1014 *potGrossPsn = grossAMax * dTemp * dVpd * lightEff * conversion;
1015 *baseFolResp = respPerGram * conversion; // do foliar resp. even if no
1016 // photosynthesis in this time step
1017}

1019// calculate transpiration (cm H20 * day^-1)
1020// and dWater (factor between 0 and 1)
1021void moisture(double *trans, double *dWater, double potGrossPsn, double vpd,
1022 double soilWater) {
1023 double potTrans; // potential transpiration in the absense of plant water
1024 // stress (cm H20 * day^-1)
1025 double removableWater;
1026 double wue; // water use efficiency, in mg CO2 fixed * g^-1 H20 transpired
1027
1028 if (potGrossPsn < TINY) { // avoid divide by 0
1029 *trans = 0.0; // no photosynthesis -> no transpiration
1030 *dWater = 1; // dWater doesn't matter, since we don't have any
1031 // photosynthesis
1032 }
1033
1034 else {
1035 wue = params.wueConst / vpd;
1036 potTrans = potGrossPsn / wue * 1000.0 * (44.0 / 12.0) * (1.0 / 10000.0);
1037 // 1000 converts g to mg; 44/12 converts g C to g CO2, 1/10000 converts m^2
1038 // to cm^2
1039
1040 removableWater = soilWater * params.waterRemoveFrac;
1041 if (climate->tsoil < params.frozenSoilThreshold) {
1042 // frozen soil - less or no water available
1043 /* frozen soil effect: fraction of water available if soil is frozen
1044 (assume amt. of water avail. w/
1045 frozen soil scales linearly with amt. of water avail. in thawed soil)
1046 */
1047 removableWater *= params.frozenSoilEff;
1048 }
1049 if (removableWater >= potTrans) {
1050 *trans = potTrans;
1051 } else {
1052 *trans = removableWater;
1053 }
1054
1055 // from PnET: equivalent to setting DWATER_MAX = 1
1056 *dWater = *trans / potTrans;
1057 }
1058}

1095// calculate leafCreation and leafLitter fluxes (g C/m^2 ground/day)
1096// leafCreation is a fraction of recent mean npp, plus some constant amount at
1097// start of growing season leafLitter is a constant rate, plus some additional
1098// fraction of leaves at end of growing season
1099void leafFluxes(double *leafCreation, double *leafLitter, double plantLeafC) {
1100 double npp; // temporal mean of recent npp (g C * m^-2 ground * day^-1)
1101
1102 npp = getMeanTrackerMean(meanNPP);
1103 // first determine the fluxes that happen at every time step, not just start &
1104 // end of growing season:
1105 if (npp > 0) {
1106 *leafCreation = npp * params.leafAllocation; // a fraction of NPP is
1107 // allocated to leaf growth
1108 } else { // net loss of C in this time step - no C left for growth
1109 *leafCreation = 0;
1110 }
1111 // a constant fraction of leaves fall in each time step
1112 *leafLitter = plantLeafC * params.leafTurnoverRate;
1113
1114 // now add add'l fluxes at start/end of growing season:
1115
1116 // first check for new year; if new year, reset trackers (since we haven't
1117 // done leaf growth or fall yet in this new year):
1118 if (climate->year > phenologyTrackers.lastYear) { // HAPPY NEW YEAR!
1119 phenologyTrackers.didLeafGrowth = 0;
1120 phenologyTrackers.didLeafFall = 0;
1121 phenologyTrackers.lastYear = climate->year;
1122 }
1123
1124 // check for start of growing season:
1125 if (!phenologyTrackers.didLeafGrowth && pastLeafGrowth()) {
1126 // we just reached the start of the growing season
1127 *leafCreation += (params.leafGrowth / climate->length);
1128 phenologyTrackers.didLeafGrowth = 1;
1129 }
1130
1131 // check for end of growing season:
1132 if (!phenologyTrackers.didLeafFall && pastLeafFall()) {
1133 // we just reached the end of the growing season
1134 *leafLitter += (plantLeafC * params.fracLeafFall) / climate->length;
1135 phenologyTrackers.didLeafFall = 1;
1136 }
1137}

1139// calculate rain and snowfall (cm water equiv./day)
1140// calculate snow melt (cm water equiv./day) and drainage(cm/day)
1141// drainage here is any water that exceeds water holding capacity
1142
1143// this is the simplified water flow function, which has only one soil moisture
1144// layer and does not do evaporation of any kind, or fast flow (sets these all
1145// to 0)
1146void simpleWaterFlow(double *rain, double *snowFall, double *immedEvap,
1147 double *snowMelt, double *sublimation, double *fastFlow,
1148 double *evaporation, double *topDrainage,
1149 double *bottomDrainage, double water, double snow,
1150 double precip, double temp, double length, double trans) {
1151 double netIn; // net water into soil, in cm
1152
1153#if SNOW // we're modeling snow
1154 if (temp <= 0) { // below freezing
1155 *snowFall = precip / length;
1156 *rain = 0;
1157 *snowMelt = 0;
1158 } else { // above freezing
1159 *snowFall = 0;
1160 *rain = precip / length;
1161 if (snow > 0) {
1162 *snowMelt = params.snowMelt * temp; // snow melt proportional to temp.
1163 if ((*snowMelt * length) > snow) { // can only melt what's there!
1164 *snowMelt = snow / length;
1165 }
1166 } else {
1167 *snowMelt = 0;
1168 }
1169 }
1170
1171#else // not modeling snow
1172 *rain = precip / length;
1173 *snowFall = 0;
1174 *snowMelt = 0;
1175#endif // #if snow
1176
1177 netIn = (*rain + *snowMelt - trans) * length;
1178 *bottomDrainage = ((water + netIn) - params.soilWHC) / length;
1179 if (*bottomDrainage < 0) {
1180 *bottomDrainage = 0;
1181 }

1206 Higher LAI will mean more canopy evap. but less evap. from pools on the

1208 evap. is a constant fraction Note that we don't evaporate snow here (we'll

1212#if LEAF_WATER

1214

1228// snowpack dynamics:
1229// calculate snow melt (cm water equiv./day) & sublimation (cm water equiv./day)
1230// ensure we don't overdrain the snowpack (so that it becomes negative)
1231// snowFall in cm/day
1232void snowPack(double *snowMelt, double *sublimation, double snowFall) {
1233 // conversion factor for sublimation
1234 static const double CONVERSION = (RHO * CP) / GAMMA * (1. / LAMBDA_S) *
1235 1000. * 1000. * (1. / 10000) * SEC_PER_DAY;
1236 // 1000 converts kg to g, 1000 converts kPa to Pa, 1/10000 converts m^2 to
1237 // cm^2
1238
1239 double rd; // aerodynamic resistance between ground and canopy air space
1240 // (sec/m)
1241 double snowRemaining; // to make sure we don't get rid of more than there is
1242
1243 // if no snow, set fluxes to 0
1244 if (envi.snow <= 0) {
1245 *snowMelt = 0;
1246 *sublimation = 0;
1247 }
1248
1249 // else: there is snow
1250 else {
1251 // first calculate sublimation, then snow melt
1252 // (if there's not enough snow to do both, priority given to sublimation)
1253 rd = (params.rdConst) / (climate->wspd); // aerodynamic resistance (sec/m)
1254 *sublimation = CONVERSION * (E_STAR_SNOW - climate->vPress) / rd;
1255
1256 snowRemaining = envi.snow + (snowFall * climate->length);
1257
1258 // remove to allow sublimation of a negative amount of snow
1259 // right now we can't sublime a negative amount of snow
1260 if (*sublimation < 0) {
1261 *sublimation = 0;
1262 }
1263
1264 // make sure we don't sublime more than there is to sublime:
1265 if (snowRemaining - (*sublimation * climate->length) < 0) {
1266 *sublimation = snowRemaining / climate->length;
1267 snowRemaining = 0;
1268 }
1269
1270 else {
1271 snowRemaining -= (*sublimation * climate->length);
1272 }
1273
1274 // below freezing: no snow melt
1275 if (climate->tair <= 0) {
1276 *snowMelt = 0;
1277 }
1278
1279 // above freezing: melt snow
1280 else {
1281 *snowMelt = params.snowMelt * climate->tair; // snow melt proportional to
1282 // temp.
1283
1284 // make sure we don't melt more than there is to melt:
1285 if (snowRemaining - (*snowMelt * climate->length) < 0) {
1286 *snowMelt = snowRemaining / climate->length;
1287 }
1288 } // end else above freezing
1289 } // end else there is snow
1290} // end snowPack

1292// water calculations relating to the top (litter/evaporative) layer of soil
1293// (if only modeling one soil water layer, these relate to that layer)
1294// calculates fastFlow (cm/day), evaporation (cm/day) and drainage to lower
1295// layer (cm/day)
1296// water is amount of water in evaporative layer (cm)
1297// whc is water holding capacity of evaporative layer (cm)
1298// (water and whc are used rather than envi variables to allow use of either
1299// litterWater or soilWater)
1300// net rain (cm/day) is (rain - immedEvap) - i.e. the amount available to enter
1301// the soil
1302// snowMelt in cm water equiv./day
1303// fluxesOut is the sum of any fluxes out of this layer that have already been
1304// calculated
1305// (e.g. transpiration if we're just using one layer) (for calculating remaining
1306// water/drainage) (cm/day)
1307void evapSoilFluxes(double *fastFlow, double *evaporation, double *drainage,
1308 double water, double whc, double netRain, double snowMelt,
1309 double fluxesOut) {
1310 // conversion factor for evaporation
1311 static const double CONVERSION = (RHO * CP) / GAMMA * (1. / LAMBDA) * 1000. *
1312 1000. * (1. / 10000) * SEC_PER_DAY;
1313 // 1000 converts kg to g, 1000 converts kPa to Pa, 1/10000 converts m^2 to
1314 // cm^2
1315
1316 double waterRemaining; /* keep running total of water remaining, in cm
1317 (to make sure we don't evap. or drain too much, and so we can drain
1318 any overflow) */
1319 double netIn; // keep track of net water into soil, in cm/day
1320 double rd; // aerodynamic resistance between ground and canopy air space
1321 // (sec/m)
1322 double rsoil; // bare soil surface resistance (sec/m)
1323
1324 netIn = netRain + snowMelt;
1325
1326 // fast flow: fraction that goes directly to drainage
1327 *fastFlow = netIn * params.fastFlowFrac;
1328 netIn -= *fastFlow;
1329
1330 // calculate evaporation:
1331 // first calculate how much water is left to evaporate (used later)
1332 waterRemaining =
1333 water + netIn * climate->length - fluxesOut * climate->length;
1334
1335 // if there's a snow pack, don't evaporate from soil:
1336 if (envi.snow > 0) {
1337 *evaporation = 0;
1338 }
1339
1340 // else no snow pack:
1341 else {
1342 rd = (params.rdConst) / (climate->wspd); // aerodynamic resistance (sec/m)
1343 rsoil = exp(params.rSoilConst1 - params.rSoilConst2 * (water / whc));
1344 *evaporation = CONVERSION * climate->vpdSoil / (rd + rsoil);
1345 // by using vpd we assume that relative humidity of soil pore space is 1
1346 // (when this isn't true, there won't be much water evaporated anyway)
1347
1348 // remove to allow negative evaporation (i.e. condensation):
1349 if (*evaporation < 0) {
1350 *evaporation = 0;
1351 }
1352
1353 // make sure we don't evaporate more than we have:
1354 if (waterRemaining - (*evaporation * climate->length) < TINY) {
1355 // leave a tiny little bit, to avoid negative water due to round-off
1356 // errors
1357 *evaporation = (waterRemaining - TINY) / climate->length;
1358 waterRemaining = 0;
1359 } else {
1360 waterRemaining -= (*evaporation * climate->length);
1361 }
1362 }
1363
1364 *drainage = 0;
1365
1366 // drain any water that remains beyond water holding capacity:
1367 if (waterRemaining > whc) {
1368 *drainage += (waterRemaining - whc) / (climate->length);
1369 }
1370}

1372// calculate drainage from bottom (soil/transpiration) layer (cm/day)
1373// based on current soil water store (cm), whc, drainage from top (cm/day) and
1374// transpiration (cm/day)
1375void transSoilDrainage(double *bottomDrainage, double topDrainage, double trans,
1376 double soilWater) {

1378
1379 waterRemaining = soilWater + (topDrainage - trans) * climate->length;
1380 *bottomDrainage = (waterRemaining - params.soilWHC) / (climate->length);
1381 if (*bottomDrainage < 0) {
1382 *bottomDrainage = 0;

1386// calculates fastFlow (cm/day), evaporation (cm/day) and drainage to lower
1387// layer (cm/day)
1388// net rain (cm/day) is (rain - immedEvap) - i.e. the amount available to enter
1389// the soil
1390// snowMelt in cm water equiv./day
1391// litterWater and soilWater in cm
1392// Also calculates drainage from bottom (soil/transpiration) layer (cm/day)
1393// Note that there may only be one layer, in which case we have only the
1394// bottomDrainage term, and evap. and trans. come from same layer.
1395void soilWaterFluxes(double *fastFlow, double *evaporation, double *topDrainage,
1396 double *bottomDrainage, double netRain, double snowMelt,
1397 double trans, double litterWater, double soilWater) {

1400 *topDrainage = 0; // no top layer, only a bottom layer
