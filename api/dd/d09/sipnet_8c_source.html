151void readClimData(const char *climFile) {
152 FILE *in;
153 ClimateNode *curr, *next;
154 int year, day;
155 int lastYear = -1;
156 double time, length; // time in hours, length in days (or fraction of day)
157 double tair, tsoil, par, precip, vpd, vpdSoil, vPress, wspd, soilWetness;
158
159 double thisGdd; // growing degree days of this time step
160 double gdd = 0.0; // growing degree days since the last Jan. 1
161
162 int status; // status of the read
163
164 // for format check
165 int firstLoc, dummyLoc, numFields;
166 int expectedNumCols;
167 int legacyFormat;
168 char *firstLine = NULL;
169 size_t lineCap = 0;
170 const char *SEPARATORS = " \t\n\r"; // characters that can separate values in
171 // parameter files
172
173 in = openFile(climFile, "r");
174
175 // Check format of first line to see if location is still specified (we will
176 // ignore it if so)
177 if (getline(&firstLine, &lineCap, in) == -1) { // EOF
178 logError("no climate data in %s\n", climFile);
179 exit(EXIT_CODE_INPUT_FILE_ERROR);
180 }
181
182 numFields = countFields(firstLine, SEPARATORS);
183 switch (numFields) {
184 case NUM_CLIM_FILE_COLS:
185 // Standard format
186 expectedNumCols = NUM_CLIM_FILE_COLS;
187 legacyFormat = 0;
188 break;
189 case NUM_CLIM_FILE_COLS_LEGACY:
190 expectedNumCols = NUM_CLIM_FILE_COLS_LEGACY;
191 legacyFormat = 1;
192 logWarning("old climate file format detected (found %d cols); ignoring "
193 "location and soilWetness columns in %s\n",
194 numFields, climFile);
195 break;
196 default:
197 // Unrecognized format
198 logError("format unrecognized in climate file %s; %d columns found, "
199 "expected %d or %d (legacy format)\n",
200 climFile, numFields, NUM_CLIM_FILE_COLS,
201 NUM_CLIM_FILE_COLS_LEGACY);
202 exit(EXIT_CODE_INPUT_FILE_ERROR);
203 }
204
205 if (legacyFormat) {
206 status = sscanf(firstLine, // NOLINT
207 "%d %d %d %lf %lf %lf %lf %lf %lf %lf %lf %lf %lf %lf",
208 &firstLoc, &year, &day, &time, &length, &tair, &tsoil, &par,
209 &precip, &vpd, &vpdSoil, &vPress, &wspd, &soilWetness);
210 } else {
211 status = sscanf(firstLine, // NOLINT
212 "%d %d %lf %lf %lf %lf %lf %lf %lf %lf %lf %lf", &year,
213 &day, &time, &length, &tair, &tsoil, &par, &precip, &vpd,
214 &vpdSoil, &vPress, &wspd);
215 }
216 free(firstLine);
217
218 if (status != expectedNumCols) {
219 logError("while reading climate file: bad data on first line\n");
220 exit(EXIT_CODE_INPUT_FILE_ERROR);
221 }
222
223 firstClimate = (ClimateNode *)malloc(sizeof(ClimateNode));
224 next = firstClimate;
225
226 while (status != EOF) {
227 // we have another day's climate
228 curr = next;
229
230 curr->year = year;
231 curr->day = day;
232 curr->time = time;
233
234 if (length < 0) { // parse as seconds
235 length = length / -86400.; // convert to days
236 }
237 curr->length = length;
238
239 curr->tair = tair;
240 curr->tsoil = tsoil;
241 curr->par = par * (1.0 / length);
242 // convert par from Einsteins * m^-2 to Einsteins * m^-2 * day^-1
243 curr->precip = precip * 0.1; // convert from mm to cm
244 curr->vpd = vpd * 0.001; // convert from Pa to kPa
245 if (curr->vpd < TINY) {
246 curr->vpd = TINY; // avoid divide by zero
247 }
248 curr->vpdSoil = vpdSoil * 0.001; // convert from Pa to kPa
249 curr->vPress = vPress * 0.001; // convert from Pa to kPa
250 curr->wspd = wspd;
251 if (curr->wspd < TINY) {
252 curr->wspd = TINY; // avoid divide by zero
253 }
254
255 if (ctx.gdd) {
256 // :: from [1], growing degree day calculations to support described
257 // modification to leaf phenology on pg 350
258 if (year != lastYear) { // HAPPY NEW YEAR!
259 gdd = 0; // reset growing degree days
260 }
261 thisGdd = tair * length;
262 if (thisGdd < 0) { // can't have negative growing degree days
263 thisGdd = 0;
264 }
265 gdd += thisGdd;
266 curr->gdd = gdd;
267 }
268
269 lastYear = year;
270
271 if (legacyFormat) {
272 status =
273 fscanf(in, // NOLINT
274 "%d %d %d %lf %lf %lf %lf %lf %lf %lf %lf %lf %lf %lf",
275 &dummyLoc, &year, &day, &time, &length, &tair, &tsoil, &par,
276 &precip, &vpd, &vpdSoil, &vPress, &wspd, &soilWetness);
277 } else {
278 status = fscanf(in, // NOLINT
279 "%d %d %lf %lf %lf %lf %lf %lf %lf %lf %lf %lf", &year,
280 &day, &time, &length, &tair, &tsoil, &par, &precip, &vpd,
281 &vpdSoil, &vPress, &wspd);
282 }
283 if (status != EOF) {
284 if (status != expectedNumCols) {
285 logError("while reading climate file: bad data near year %d day %d\n",
286 year, day);
287 exit(EXIT_CODE_INPUT_FILE_ERROR);
288 }
289 // Check for older file with multiple locations - that's an error now
290 if (legacyFormat && (dummyLoc != firstLoc)) {
291 logError("while reading legacy climate file %s: multiple locations "
292 "not supported (locations found: %d and %d)\n",
293 climFile, firstLoc, dummyLoc);
294 exit(EXIT_CODE_INPUT_FILE_ERROR);
295 }
296
297 // add a new climate node at end of linked list
298 next = (ClimateNode *)malloc(sizeof(ClimateNode));
299 curr->nextClim = next;
300 // set this down here rather than at top of loop so head treated the
301 // same as rest of list
302 } else { // status == EOF - no more records
303 curr->nextClim = NULL; // terminate this last linked list
304 }
305 } // end while
306
307 fclose(in);
308}

523void calcLightEff(double *lightEff, double lai, double par) {
524
525 // Information on the distribution of LAI with height is available
526 // as of March 2007 ... contact Dr. Maggie Prater Maggie.Prater@colorado.edu
527
528 // All of this modeling comes from [1]
529
530 static const int NUM_LAYERS = 6;
531 // believe it or not, 6 layers gives approximately the same result as 100
532 // layers
533
534 int layer; // counter
535 double cumLai; // lai from this layer up
536 double lightIntensity;
537 double currLightEff = 0.0, cumLightEff;
538 double currfAPAR = 0.0, cumfAPAR;
539 int coeff; // current coefficient in Simpson's rule
540
541 if (lai > 0 && par > 0) { // must have at least some leaves and some light
542 cumLightEff = 0.0; // the running sum
543 layer = 0;
544 coeff = 1;
545 int err;
546 double fAPAR; // Calculation of fAPAR according to 1 - exp(attenuation*LAI)
547 cumfAPAR = 0.0;
548
549 while (layer <= NUM_LAYERS) { // layer 0 means top layer
550 // lai from this layer up, starting at top
551 // :: from [1], description above eq (A11)
552 cumLai = lai * ((double)layer / NUM_LAYERS);
553
554 // par attenuated down to current layer
555 // :: from [1], eq (A11)
556 lightIntensity = par * exp(-1.0 * params.attenuation * cumLai);
557
558 // between 0 and 1; when lightIntensity = halfSatPar, currLightEff = 1/2
559 // :: from [1], eq (A12), modified as power of 2 instead of e (but
560 // equivalent mathematically)
561 currLightEff = (1 - pow(2, (-1.0 * lightIntensity / params.halfSatPar)));
562
563 // CHANGE FROM [1]: instead of taking mean of the light effects at each
564 // layer, we approximate the integral via simpson's rule, and then divide
565 // by the number of layers
566 cumLightEff += coeff * currLightEff;
567
568 // FPAR calc for the tracker below (likely cruft, not part of [1])
569 currfAPAR = 1 - (lightIntensity / par);
570 cumfAPAR += coeff * currfAPAR;
571
572 // now move to the next layer:
573 layer++;
574 coeff = 2 * (1 + layer % 2); // coeff. goes 1, 4, 2, 4, ..., 2, 4, 2
575 }

605// calculate gross photosynthesis without water effect (g C * m^-2 ground area *
606// day^-1) and base foliar respiration without temp, water, etc. (g C * m^-2
607// ground area * day^-1)
608
627void potPsn(double *potGrossPsn, double *baseFolResp, double lai, double tair,
628 double vpd, double par) {
629 // Calculation of potGrossPsn proceeds as described in [1], with minor
630 // modifications as noted below.
631
632 // Calculation of baseFolResp is used as part of [1], eq (A18), calculated
633 // here for convenience (see vegResp() fur use of baseFolResp).
634
635 // maximum possible gross respiration (nmol CO2 * g^-1 leaf * sec^-1)
636 double grossAMax;
637 // effect of temperature on photosynthesis (range: [0:1])
638 double dTemp;
639 // decrease in leaf gas exchange due to vapor pressure deficit
640 double dVpd;
641 // decrease in photosynthesis due to amt. of light absorbed
642 double dLight;
643 // base foliar respiration in nmol CO2 * g^-1 leaf * sec^-1
644 double respPerGram;
645 // convert from (nmol CO2 * g^-1 leaf * sec^-1) to
646 // (g C * m^-2 ground area * day^-1)
647 double conversion;
648
649 // foliar respiration, unmodified by temp, etc.
650 // :: from [1], eq (A5)
651 respPerGram = params.baseFolRespFrac * params.aMax;
652 // daily maximum gross photosynthetic rate
653 // :: from [1], eq (A6)
654 grossAMax = params.aMax * params.aMaxFrac + respPerGram;
655
656 // Now to calculate reductions to the daily maximum - dTemp, dVpd, dLight
657 // :: from [1], eq (A9)
658 dTemp = (params.psnTMax - tair) * (tair - params.psnTMin) /
659 pow((params.psnTMax - params.psnTMin) / 2.0, 2);
660 dTemp = fmax(dTemp, 0.0);
661 // :: from [1], eq (A10); modified to accept a variable exponent, [1] uses
662 // dVpdExp = 2 [TAG:UNKNOWN_PROVENANCE] vpd exponent
663 dVpd = 1.0 - params.dVpdSlope * pow(vpd, params.dVpdExp);
664 dVpd = fmax(dVpd, 0.0);
665 // dLight calculated as described in [1], see calcLightEff()
666 calcLightEff(&dLight, lai, par);
667
668 // :: from [1], unit conversion taking into account eq (A8)
669 conversion = C_WEIGHT * (1.0 / TEN_9) *
670 (params.leafCSpWt / params.cFracLeaf) * lai *
671 SEC_PER_DAY; // to convert units
672 // :: from [1], eq (A7)
673 *potGrossPsn = grossAMax * dTemp * dVpd * dLight * conversion;
674
675 // do foliar resp. even if no photosynthesis in this time step
676 // :: from [1] in part, used in eq (A18) later
677 *baseFolResp = respPerGram * conversion;
678}

680// calculate transpiration (cm H20 * day^-1)
681// and dWater (factor between 0 and 1)
693void moisture(double *trans, double *dWater, double potGrossPsn, double vpd,
694 double soilWater) {
695 // potential transpiration in the absense of plant water stress
696 // (cm H20 * day^-1)
697 double potTrans;
698 // amount of water available for photosynthesis
699 double removableWater;
700 // water use efficiency, in mg CO2 fixed * g^-1 H20 transpired
701 double wue;
702
703 if (potGrossPsn < TINY) { // avoid divide by 0
704 *trans = 0.0; // no photosynthesis -> no transpiration
705 *dWater = 1; // dWater doesn't matter, since we don't have any
706 // photosynthesis
707 }
708
709 else {
710 // :: from [1], eq (A13)
711 wue = params.wueConst / vpd;
712
713 // 1000 converts g to mg; 44/12 converts g C to g CO2, 1/10000 converts m^2
714 // to cm^2
715 // :: from [1], eq (A14) plus unit conversion as described there
716 potTrans = potGrossPsn / wue * 1000.0 * (44.0 / 12.0) * (1.0 / 10000.0);
717
718 // :: from [1], discussion below eq (A14)
719 removableWater = soilWater * params.waterRemoveFrac;
720
721 // :: from [2], snowpack modification
722 if (climate->tsoil < params.frozenSoilThreshold) {
723 // frozen soil - less or no water available
724 // frozen soil effect: fraction of water available if soil is frozen
725 // (assume amount of water available in frozen soil scales linearly
726 // with amount of water available in thawed soil)
727 removableWater *= params.frozenSoilEff;
728 }
729
730 // :: from [1], eq (A15)
731 *trans = fmin(removableWater, potTrans);
732
733 // :: from [1], eq (A16)
734 *dWater = *trans / potTrans;
735 }
736}

870// snowpack dynamics:
871// calculate snow melt (cm water equiv./day) & sublimation (cm water equiv./day)
872// ensure we don't overdrain the snowpack (so that it becomes negative)
873// snowFall in cm/day
874void snowPack(double *snowMelt, double *sublimation, double snowFall) {
875 // conversion factor for sublimation
876 static const double CONVERSION = (RHO * CP) / GAMMA * (1. / LAMBDA_S) *
877 1000. * 1000. * (1. / 10000) * SEC_PER_DAY;
878 // 1000 converts kg to g, 1000 converts kPa to Pa, 1/10000 converts m^2 to
879 // cm^2
880
881 double rd; // aerodynamic resistance between ground and canopy air space
882 // (sec/m)
883 double snowRemaining; // to make sure we don't get rid of more than there is
884
885 // if no snow, set fluxes to 0
886 if (envi.snow <= 0) {
887 *snowMelt = 0;
888 *sublimation = 0;
889 }
890
891 // else: there is snow
892 else {
893 // first calculate sublimation, then snow melt
894 // (if there's not enough snow to do both, priority given to sublimation)
895 rd = (params.rdConst) / (climate->wspd); // aerodynamic resistance (sec/m)
896 *sublimation = CONVERSION * (E_STAR_SNOW - climate->vPress) / rd;
897
898 snowRemaining = envi.snow + (snowFall * climate->length);
899
900 // remove to allow sublimation of a negative amount of snow
901 // right now we can't sublime a negative amount of snow
902 if (*sublimation < 0) {
903 *sublimation = 0;
904 }
905
906 // make sure we don't sublime more than there is to sublime:
907 if (snowRemaining - (*sublimation * climate->length) < 0) {
908 *sublimation = snowRemaining / climate->length;
909 snowRemaining = 0;
910 }
911
912 else {
913 snowRemaining -= (*sublimation * climate->length);
914 }
915
916 // below freezing: no snow melt
917 if (climate->tair <= 0) {
918 *snowMelt = 0;
919 }
920
921 // above freezing: melt snow
922 else {
923 *snowMelt = params.snowMelt * climate->tair; // snow melt proportional to
924 // temp.
925
926 // make sure we don't melt more than there is to melt:
927 if (snowRemaining - (*snowMelt * climate->length) < 0) {
928 *snowMelt = snowRemaining / climate->length;
929 }
930 } // end else above freezing
931 } // end else there is snow
932} // end snowPack

934// water calculations relating to the top (litter/evaporative) layer of soil
935// (if only modeling one soil water layer, these relate to that layer)
936// calculates fastFlow (cm/day), evaporation (cm/day) and drainage to lower
937// layer (cm/day)
938// water is amount of water in evaporative layer (cm)
939// whc is water holding capacity of evaporative layer (cm)
940// (water and whc are used rather than envi variables to allow use of either
941// litterWater or soilWater)
942// net rain (cm/day) is (rain - immedEvap) - i.e. the amount available to enter
943// the soil
944// snowMelt in cm water equiv./day
945// fluxesOut is the sum of any fluxes out of this layer that have already been
946// calculated
947// (e.g. transpiration if we're just using one layer) (for calculating remaining
948// water/drainage) (cm/day)
949void evapSoilFluxes(double *fastFlow, double *evaporation, double *drainage,
950 double water, double whc, double netRain, double snowMelt,
951 double fluxesOut) {
952 // conversion factor for evaporation
953 static const double CONVERSION = (RHO * CP) / GAMMA * (1. / LAMBDA) * 1000. *
954 1000. * (1. / 10000) * SEC_PER_DAY;
955 // 1000 converts kg to g, 1000 converts kPa to Pa, 1/10000 converts m^2 to
956 // cm^2
957
958 double waterRemaining; /* keep running total of water remaining, in cm
959 (to make sure we don't evap. or drain too much, and so we can drain
960 any overflow) */
961 double netIn; // keep track of net water into soil, in cm/day
962 double rd; // aerodynamic resistance between ground and canopy air space
963 // (sec/m)
964 double rsoil; // bare soil surface resistance (sec/m)
965
966 netIn = netRain + snowMelt;
967
968 // fast flow: fraction that goes directly to drainage
969 *fastFlow = netIn * params.fastFlowFrac;
970 netIn -= *fastFlow;
971
972 // calculate evaporation:
973 // first calculate how much water is left to evaporate (used later)
974 waterRemaining =
975 water + netIn * climate->length - fluxesOut * climate->length;
976
977 // if there's a snow pack, don't evaporate from soil:
978 if (envi.snow > 0) {
979 *evaporation = 0;
980 }
981
982 // else no snow pack:
983 else {
984 rd = (params.rdConst) / (climate->wspd); // aerodynamic resistance (sec/m)
985 rsoil = exp(params.rSoilConst1 - params.rSoilConst2 * (water / whc));
986 *evaporation = CONVERSION * climate->vpdSoil / (rd + rsoil);
987 // by using vpd we assume that relative humidity of soil pore space is 1
988 // (when this isn't true, there won't be much water evaporated anyway)
989
990 // remove to allow negative evaporation (i.e. condensation):
991 if (*evaporation < 0) {
992 *evaporation = 0;
993 }
994
995 // make sure we don't evaporate more than we have:
996 if (waterRemaining - (*evaporation * climate->length) < TINY) {
997 // leave a tiny little bit, to avoid negative water due to round-off
998 // errors
999 *evaporation = (waterRemaining - TINY) / climate->length;
1000 waterRemaining = 0;
1001 } else {
1002 waterRemaining -= (*evaporation * climate->length);
1003 }
1004 }
1005
1006 *drainage = 0;
1007
1008 // drain any water that remains beyond water holding capacity:
1009 if (waterRemaining > whc) {
1010 *drainage += (waterRemaining - whc) / (climate->length);
1011 }
1012}

1015// layer (cm/day)
1016// net rain (cm/day) is (rain - immedEvap) - i.e. the amount available to enter
1017// the soil
1018// snowMelt in cm water equiv./day
1019// soilWater in cm
1020// Also calculates drainage from bottom (soil/transpiration) layer (cm/day)
1021// Note that there may only be one layer, in which case we have only the
1022// bottomDrainage term, and evap. and trans. come from same layer.
1023void soilWaterFluxes(double *fastFlow, double *evaporation, double *topDrainage,
1024 double *bottomDrainage, double netRain, double snowMelt,
1025 double trans, double soilWater) {
1026
1027 // only one soil moisture pool: evap. and trans. both happen from this pool
1028 *topDrainage = 0; // no top layer, only a bottom layer
1029
1030 // last param = fluxes out that have already been calculated: transpiration
1031 evapSoilFluxes(fastFlow, evaporation, bottomDrainage, soilWater,
1032 params.soilWHC, netRain, snowMelt, trans);
1033}

1235 } else {
1236 // With no microbes, rSoil flux is just the maintenance respiration
1237 // (at least, that's how maintResp was calculated; we still have some
1238 // naming weirdness here)
1239 fluxes.rSoil = fluxes.maintRespiration;
1240 if (ctx.litterPool) {
1241 // :: from [2], litter model description
1242 envi.litter += (fluxes.woodLitter + fluxes.leafLitter -
1243 fluxes.litterToSoil - fluxes.rLitter) *
1244 climate->length;
1245
1246 // from [2] and [3], litter and root terms respectively
1247 envi.soil += (fluxes.coarseRootLoss + fluxes.fineRootLoss +
1248 fluxes.litterToSoil - fluxes.rSoil) *
1249 climate->length;
1250 } else {
1251 // Normal pool (single pool, no microbes)
1252 // :: from [1] (and others, TBD), eq (A3), where:
1253 // L_w = fluxes.woodLitter
1254 // L_l = fluxes.leafLitter
1255 // R_h = fluxes.rSoil
1256 // :: from [3], root terms
1257 envi.soil += (fluxes.coarseRootLoss + fluxes.fineRootLoss +
1258 fluxes.woodLitter + fluxes.leafLitter - fluxes.rSoil) *
1259 climate->length;
1260 }
1261 }
1262 // :: from [3], root model description
1263 envi.coarseRootC +=
1264 (fluxes.coarseRootCreation - fluxes.coarseRootLoss - fluxes.rCoarseRoot) *
1265 climate->length;
1266 envi.fineRootC +=
1267 (fluxes.fineRootCreation - fluxes.fineRootLoss - fluxes.rFineRoot) *
1268 climate->length;
1269}

1306double soilBreakdown(double poolC, double baseRate, double water, double whc,
1307 double tsoil, double Q10) {
1308 double tempEffect = pow(Q10, tsoil / 10.0);
1309 double moistEffect;
1310
1311 if (ctx.waterHResp) {
1312 // :: from [2], snowpack addition
1313 if (climate->tsoil < 0) {
1314 moistEffect = 1; // Ignore moisture effects in frozen soils
1315 } else {
1316 // As in calcMaintenanceRespiration, provenance of soilRespMoistEffect

1319 // moistEffect = water / whc
1320 // which matches here with params.soilRespMoistEffect=1 (the default

1326 // :: that is, moisture-independent R_h
1327 moistEffect = 1;
1328 }

1338void calculateFluxes(void) {
1339 // base foliar respiration, calc'd as part of potential photosynthesis
1340 double baseFolResp;
1341 // potential photosynthesis, without water stress
1342 double potGrossPsn;
1343 // reduction in photosynthesis due to soil dryness
1344 double dWater;
1345 // m^2 leaf/m^2 ground (calculated from plantLeafC)
1346 double lai;
1347 // total litter breakdown (i.e. litterToSoil + rLitter) (g C/m^2 ground/day)
1348 double litterBreakdown;
1349 // maintenance respiration terms (g C * m^-2 ground area * day^-1)
1350 double folResp, woodResp;
1351 // amount of water in litter and soil (cm) taken from environment
1352 double litterWater, soilWater;
1353 // Growth respiration, if splitting growth and maintenance
1354 // (g C * m^-2 ground area * day^-1)
1355 double growthResp;
1356 // net rain, equal to (rain - immedEvap) (cm/day)
1357 double netRain;
1358
1359 litterWater = envi.litterWater;
1360 soilWater = envi.soilWater;
1361
1362 lai = envi.plantLeafC / params.leafCSpWt; // current lai
1363
1364 potPsn(&potGrossPsn, &baseFolResp, lai, climate->tair, climate->vpd,
1365 climate->par);
1366 moisture(&(fluxes.transpiration), &dWater, potGrossPsn, climate->vpd,
1367 soilWater);
1368
1369 calcPrecip(&(fluxes.rain), &(fluxes.snowFall), &(fluxes.immedEvap), lai);
1370 netRain = fluxes.rain - fluxes.immedEvap;
1371 snowPack(&(fluxes.snowMelt), &(fluxes.sublimation), fluxes.snowFall);
1372 soilWaterFluxes(&(fluxes.fastFlow), &(fluxes.evaporation),
1373 &(fluxes.topDrainage), &(fluxes.bottomDrainage), netRain,
1374 fluxes.snowMelt, fluxes.transpiration, soilWater);
1375
1376 getGpp(&(fluxes.photosynthesis), potGrossPsn, dWater);
1377
1378 if (ctx.growthResp) {
1379 vegResp2(&folResp, &woodResp, &growthResp, baseFolResp,
1380 fluxes.photosynthesis);
1381 fluxes.rVeg = folResp + woodResp + growthResp;
1382 } else {
1383 vegResp(&folResp, &woodResp, baseFolResp);
1384 fluxes.rVeg = folResp + woodResp;
1385 }
1386
1387 leafFluxes(&(fluxes.leafCreation), &(fluxes.leafLitter), envi.plantLeafC);
1388
1389 if (ctx.litterPool) {
1390 litterBreakdown =
1391 soilBreakdown(envi.litter, params.litterBreakdownRate, litterWater,
1392 params.litterWHC, climate->tsoil, params.soilRespQ10);
1393 fluxes.rLitter = litterBreakdown * params.fracLitterRespired;
1394 fluxes.litterToSoil = litterBreakdown * (1.0 - params.fracLitterRespired);
1395 } else {
1396 // litterBreakdown = 0;
1397 fluxes.rLitter = 0;
1398 fluxes.litterToSoil = 0;
1399 }
1400
1401 // finally, calculate fluxes that we haven't already calculated:
1402
1403 fluxes.woodLitter = woodLitterF(envi.plantWoodC);
1404
1405 double coarseExudate, fineExudate; // exudates in and out of soil
1406 double npp, gppSoil; // running means of our tracker variables
1407
1408 npp = getMeanTrackerMean(meanNPP);
1409 gppSoil = getMeanTrackerMean(meanGPP);
1410 if (npp > 0) {
1411 // :: from [3], root model description and eq (3)
1412 fluxes.coarseRootCreation = params.coarseRootAllocation * npp;
1413 fluxes.fineRootCreation = params.fineRootAllocation * npp;
1414 fluxes.woodCreation = params.woodAllocation * npp;
1415 } else {
1416 fluxes.coarseRootCreation = 0;
1417 fluxes.fineRootCreation = 0;
1418 fluxes.woodCreation = 0;
1419 }
1420
1421 if ((gppSoil > 0) & (envi.fineRootC > 0)) {
1422 // :: from [3], eq (5)
1423 coarseExudate = params.coarseRootExudation * gppSoil;
1424 fineExudate = params.fineRootExudation * gppSoil;
1425 } else {
1426 fineExudate = 0;
1427 coarseExudate = 0;
1428 }
1429
1430 // :: from [3], roots model descriptions and [4] eq (5.11)
1431 fluxes.coarseRootLoss = (1 - params.microbePulseEff) * coarseExudate +
1432 params.coarseRootTurnoverRate * envi.coarseRootC;
1433 fluxes.fineRootLoss = (1 - params.microbePulseEff) * fineExudate +
1434 params.fineRootTurnoverRate * envi.fineRootC;
1435
1436 // fluxes that get added to microbe pool
1437 // :: from [4], eq (5.11) first line
1438 fluxes.soilPulse = params.microbePulseEff * (coarseExudate + fineExudate);
1439
1440 // :: from [3], root model description
1441 calcRootResp(&fluxes.rCoarseRoot, params.coarseRootQ10,
1442 params.baseCoarseRootResp, envi.coarseRootC);
1443 calcRootResp(&fluxes.rFineRoot, params.fineRootQ10, params.baseFineRootResp,
1444 envi.fineRootC);
1445}

1740// calculate all fluxes and update state for this time step
1741// we calculate all fluxes before updating state in case flux calculations
1742// depend on the old state
1743void updateState(void) {
1744 double npp; // net primary productivity, g C * m^-2 ground area * day^-1
1745 double oldSoilWater; // how much soil water was there before we updated it?
1746 // Used in trackers
1747 int err;
1748 oldSoilWater = envi.soilWater;
1749
1750 // 1. Calculate Fluxes
1751
1752 calculateFluxes();
1753 soilDegradation(); // This updates all the soil functions
1754
1755 // 2. Update Pools
1756
1757 // Update the stocks, with fluxes adjusted for length of time step.
1758 // Note: the soil C pool(s) (envi.soil, envi.fineRootC, envi.CoarseRootC)
1759 // were updated in soilDegradation(); also, envi.litter when that is in use.
1760 // And yes, I would love to break that out to maintain more rigor in this
1761 // flow process.
1762
1763 // :: from [1], eq (A1), where:
1764 // GPP = fluxes.photosynthesis
1765 // R_a = fluxes.rVeg
1766 // L_w = fluxes.woodLitter
1767 // L = fluxes.leafCreation
1768 // :: from [3], additional root terms (woodCreation, fineRootC, coarseRootC)
1769 // from NPP allocation
1770 envi.plantWoodC += (fluxes.photosynthesis + fluxes.woodCreation -
1771 fluxes.leafCreation - fluxes.woodLitter - fluxes.rVeg -
1772 fluxes.coarseRootCreation - fluxes.fineRootCreation) *
1773 climate->length;
1774
1775 // :: from [1], eq (A2), where:
1776 // L = fluxes.leafCreation
1777 // L_L = fluxes.leafLitter
1778 envi.plantLeafC +=
1779 (fluxes.leafCreation - fluxes.leafLitter) * climate->length;
1780
1781 // :: from [1], eq (A4), where:
1782 // P = (fluxes.rain - fluxes.immedEvap)
1783 // T = (fluxes.transpiration + fluxes.evaporation)
1784 // D = (fluxes.bottomDrainage + fluxes.fastFlow)
1785 // from [2], addition of fluxes.snowMelt, as well as the modifications in
1786 // the terms above: immedEvap in P, evaporation in T, and fastFlow in D
1787 envi.soilWater +=
1788 (fluxes.rain + fluxes.snowMelt - fluxes.immedEvap - fluxes.fastFlow -
1789 fluxes.evaporation - fluxes.transpiration - fluxes.bottomDrainage) *
1790 climate->length;
1791
1792 // if ctx.snow = 0, some or all of these fluxes will always be 0
1793 // :: from [2], addition of snowpack description
1794 envi.snow += (fluxes.snowFall - fluxes.snowMelt - fluxes.sublimation) *
1795 climate->length;
1796
1797 ensureNonNegativeStocks();
1798
1799 // 3. Process events for this location/year/day, AFTER updates are made to
1800 // fluxes and state variables above. Events are (currently, Jan 25, 2025)
1801 // handled as instantaneous deltas to relevant state (envi and fluxes fields),
1802 processEvents();
1803
1804 // 4. Update trackers
1805 npp = fluxes.photosynthesis - fluxes.rVeg - fluxes.rCoarseRoot -
1806 fluxes.rFineRoot;
1807
1808 err = addValueToMeanTracker(meanNPP, npp, climate->length); // update running
1809 // mean of NPP
1810 if (err != 0) {
1811 printf("******* Error type %d while trying to add value to NPP mean "
1812 "tracker in sipnet:updateState() *******\n",
1813 err);
1814 printf("npp = %f, climate->length = %f\n", npp, climate->length);
1815 printf("Suggestion: try changing MEAN_NPP_MAX_ENTRIES in sipnet.c\n");
1816 exit(1);
1817 }
1818
1819 err = addValueToMeanTracker(meanGPP, fluxes.photosynthesis,
1820 climate->length); // update running mean of GPP
1821 if (err != 0) {
1822 printf("******* Error type %d while trying to add value to GPP mean "
1823 "tracker in sipnet:updateState() *******\n",
1824 err);
1825 printf("GPP = %f, climate->length = %f\n", fluxes.photosynthesis,
1826 climate->length);
1827 printf("Suggestion: try changing MEAN_GPP_SOIL_MAX_ENTRIES in sipnet.c\n");
1828 exit(1);
1829 }
1830
1831 updateTrackers(oldSoilWater);
1832}
