133void readClimData(const char *climFile) {
134 FILE *in;
135 ClimateNode *curr, *next;
136 int year, day;
137 int lastYear = -1;
138 double time, length; // time in hours, length in days (or fraction of day)
139 double tair, tsoil, par, precip, vpd, vpdSoil, vPress, wspd, soilWetness;
140
141 double thisGdd; // growing degree days of this time step
142 double gdd = 0.0; // growing degree days since the last Jan. 1
143
144 int status; // status of the read
145
146 // for format check
147 int firstLoc, dummyLoc, numFields;
148 int expectedNumCols;
149 int legacyFormat;
150 char *firstLine = NULL;
151 size_t lineCap = 0;
152 const char *SEPARATORS = " \t\n\r"; // characters that can separate values in
153 // parameter files
154
155 in = openFile(climFile, "r");
156
157 // Check format of first line to see if location is still specified (we will
158 // ignore it if so)
159 if (getline(&firstLine, &lineCap, in) == -1) { // EOF
160 logError("no climate data in %s\n", climFile);
161 exit(EXIT_CODE_INPUT_FILE_ERROR);
162 }
163
164 numFields = countFields(firstLine, SEPARATORS);
165 switch (numFields) {
166 case NUM_CLIM_FILE_COLS:
167 // Standard format
168 expectedNumCols = NUM_CLIM_FILE_COLS;
169 legacyFormat = 0;
170 break;
171 case NUM_CLIM_FILE_COLS_LEGACY:
172 expectedNumCols = NUM_CLIM_FILE_COLS_LEGACY;
173 legacyFormat = 1;
174 logWarning("old climate file format detected (found %d cols); ignoring "
175 "location and soilWetness columns in %s\n",
176 numFields, climFile);
177 break;
178 default:
179 // Unrecognized format
180 logError("format unrecognized in climate file %s; %d columns found, "
181 "expected %d or %d (legacy format)\n",
182 climFile, numFields, NUM_CLIM_FILE_COLS,
183 NUM_CLIM_FILE_COLS_LEGACY);
184 exit(EXIT_CODE_INPUT_FILE_ERROR);
185 }
186
187 if (legacyFormat) {
188 status = sscanf(firstLine, // NOLINT
189 "%d %d %d %lf %lf %lf %lf %lf %lf %lf %lf %lf %lf %lf",
190 &firstLoc, &year, &day, &time, &length, &tair, &tsoil, &par,
191 &precip, &vpd, &vpdSoil, &vPress, &wspd, &soilWetness);
192 } else {
193 status = sscanf(firstLine, // NOLINT
194 "%d %d %lf %lf %lf %lf %lf %lf %lf %lf %lf %lf", &year,
195 &day, &time, &length, &tair, &tsoil, &par, &precip, &vpd,
196 &vpdSoil, &vPress, &wspd);
197 }
198 free(firstLine);
199
200 if (status != expectedNumCols) {
201 logError("while reading climate file: bad data on first line\n");
202 exit(EXIT_CODE_INPUT_FILE_ERROR);
203 }
204
205 firstClimate = (ClimateNode *)malloc(sizeof(ClimateNode));
206 next = firstClimate;
207
208 while (status != EOF) {
209 // we have another day's climate
210 curr = next;
211
212 curr->year = year;
213 curr->day = day;
214 curr->time = time;
215
216 if (length < 0) { // parse as seconds
217 length = length / -86400.; // convert to days
218 }
219 curr->length = length;
220
221 curr->tair = tair;
222 curr->tsoil = tsoil;
223 curr->par = par * (1.0 / length);
224 // convert par from Einsteins * m^-2 to Einsteins * m^-2 * day^-1
225 curr->precip = precip * 0.1; // convert from mm to cm
226 curr->vpd = vpd * 0.001; // convert from Pa to kPa
227 if (curr->vpd < TINY) {
228 curr->vpd = TINY; // avoid divide by zero
229 }
230 curr->vpdSoil = vpdSoil * 0.001; // convert from Pa to kPa
231 curr->vPress = vPress * 0.001; // convert from Pa to kPa
232 curr->wspd = wspd;
233 if (curr->wspd < TINY) {
234 curr->wspd = TINY; // avoid divide by zero
235 }
236
237 if (ctx.gdd) {
238 // :: from [1], growing degree day calculations to support described
239 // modification to leaf phenology on pg 350
240 if (year != lastYear) { // HAPPY NEW YEAR!
241 gdd = 0; // reset growing degree days
242 }
243 thisGdd = tair * length;
244 if (thisGdd < 0) { // can't have negative growing degree days
245 thisGdd = 0;
246 }
247 gdd += thisGdd;
248 curr->gdd = gdd;
249 }
250
251 lastYear = year;
252
253 if (legacyFormat) {
254 status =
255 fscanf(in, // NOLINT
256 "%d %d %d %lf %lf %lf %lf %lf %lf %lf %lf %lf %lf %lf",
257 &dummyLoc, &year, &day, &time, &length, &tair, &tsoil, &par,
258 &precip, &vpd, &vpdSoil, &vPress, &wspd, &soilWetness);
259 } else {
260 status = fscanf(in, // NOLINT
261 "%d %d %lf %lf %lf %lf %lf %lf %lf %lf %lf %lf", &year,
262 &day, &time, &length, &tair, &tsoil, &par, &precip, &vpd,
263 &vpdSoil, &vPress, &wspd);
264 }
265 if (status != EOF) {
266 if (status != expectedNumCols) {
267 logError("while reading climate file: bad data near year %d day %d\n",
268 year, day);
269 exit(EXIT_CODE_INPUT_FILE_ERROR);
270 }
271 // Check for older file with multiple locations - that's an error now
272 if (legacyFormat && (dummyLoc != firstLoc)) {
273 logError("while reading legacy climate file %s: multiple locations "
274 "not supported (locations found: %d and %d)\n",
275 climFile, firstLoc, dummyLoc);
276 exit(EXIT_CODE_INPUT_FILE_ERROR);
277 }
278
279 // add a new climate node at end of linked list
280 next = (ClimateNode *)malloc(sizeof(ClimateNode));
281 curr->nextClim = next;
282 // set this down here rather than at top of loop so head treated the
283 // same as rest of list
284 } else { // status == EOF - no more records
285 curr->nextClim = NULL; // terminate this last linked list
286 }
287 } // end while
288
289 fclose(in);
290}

482void calcLightEff(double *lightEff, double lai, double par) {
483
484 // Information on the distribution of LAI with height is available
485 // as of March 2007 ... contact Dr. Maggie Prater Maggie.Prater@colorado.edu
486
487 // All of this modeling comes from [1]
488
489 static const int NUM_LAYERS = 6;
490 // believe it or not, 6 layers gives approximately the same result as 100
491 // layers
492
493 int layer; // counter
494 double cumLai; // lai from this layer up
495 double lightIntensity;
496 double currLightEff = 0.0, cumLightEff;
497 int coeff; // current coefficient in Simpson's rule
498
499 if (lai > 0 && par > 0) { // must have at least some leaves and some light
500 cumLightEff = 0.0; // the running sum
501 layer = 0;
502 coeff = 1;
503
504 while (layer <= NUM_LAYERS) { // layer 0 means top layer
505 // lai from this layer up, starting at top
506 // :: from [1], description above eq (A11)
507 cumLai = lai * ((double)layer / NUM_LAYERS);
508
509 // par attenuated down to current layer
510 // :: from [1], eq (A11)
511 lightIntensity = par * exp(-1.0 * params.attenuation * cumLai);
512
513 // between 0 and 1; when lightIntensity = halfSatPar, currLightEff = 1/2
514 // :: from [1], eq (A12), modified as power of 2 instead of e (but
515 // equivalent mathematically)
516 currLightEff = (1 - pow(2, (-1.0 * lightIntensity / params.halfSatPar)));
517
518 // CHANGE FROM [1]: instead of taking mean of the light effects at each
519 // layer, we approximate the integral via simpson's rule, and then divide
520 // by the number of layers
521 cumLightEff += coeff * currLightEff;
522
523 // now move to the next layer:
524 layer++;
525 coeff = 2 * (1 + layer % 2); // coeff. goes 1, 4, 2, 4, ..., 2, 4, 2
526 }
527 // last value should have had a coefficient of 1, but actually had a
528 // coefficient of 2, so subtract 1:
529 cumLightEff -= currLightEff;
530 *lightEff = cumLightEff / (3.0 * NUM_LAYERS); // multiplying by (h/3) in
531 // Simpson's rule
532 } else { // no leaves or no light!
533 *lightEff = 0;
534 }
535}

555void potPsn(double *potGrossPsn, double *baseFolResp, double lai, double tair,
556 double vpd, double par) {
557 // Calculation of potGrossPsn proceeds as described in [1], with minor
558 // modifications as noted below.
559
560 // Calculation of baseFolResp is used as part of [1], eq (A18), calculated
561 // here for convenience (see vegResp() fur use of baseFolResp).
562
563 // maximum possible gross respiration (nmol CO2 * g^-1 leaf * sec^-1)
564 double grossAMax;
565 // effect of temperature on photosynthesis (range: [0:1])
566 double dTemp;
567 // decrease in leaf gas exchange due to vapor pressure deficit
568 double dVpd;
569 // decrease in photosynthesis due to amt. of light absorbed
570 double dLight;
571 // base foliar respiration in nmol CO2 * g^-1 leaf * sec^-1
572 double respPerGram;
573 // convert from (nmol CO2 * g^-1 leaf * sec^-1) to
574 // (g C * m^-2 ground area * day^-1)
575 double conversion;
576
577 // foliar respiration, unmodified by temp, etc.
578 // :: from [1], eq (A5)
579 respPerGram = params.baseFolRespFrac * params.aMax;
580 // daily maximum gross photosynthetic rate
581 // :: from [1], eq (A6)
582 grossAMax = params.aMax * params.aMaxFrac + respPerGram;
583
584 // Now to calculate reductions to the daily maximum - dTemp, dVpd, dLight
585 // :: from [1], eq (A9)
586 dTemp = (params.psnTMax - tair) * (tair - params.psnTMin) /
587 pow((params.psnTMax - params.psnTMin) / 2.0, 2);
588 dTemp = fmax(dTemp, 0.0);
589 // :: from [1], eq (A10); modified to accept a variable exponent, [1] uses
590 // dVpdExp = 2 [TAG:UNKNOWN_PROVENANCE] vpd exponent
591 dVpd = 1.0 - params.dVpdSlope * pow(vpd, params.dVpdExp);
592 dVpd = fmax(dVpd, 0.0);
593 // dLight calculated as described in [1], see calcLightEff()
594 calcLightEff(&dLight, lai, par);
595
596 // :: from [1], unit conversion taking into account eq (A8)
597 conversion = C_WEIGHT * (1.0 / TEN_9) *
598 (params.leafCSpWt / params.cFracLeaf) * lai *
599 SEC_PER_DAY; // to convert units
600 // :: from [1], eq (A7)
601 *potGrossPsn = grossAMax * dTemp * dVpd * dLight * conversion;
602
603 // do foliar resp. even if no photosynthesis in this time step
604 // :: from [1] in part, used in eq (A18) later
605 *baseFolResp = respPerGram * conversion;
606}

608// calculate transpiration (cm H20 * day^-1)
609// and dWater (factor between 0 and 1)
621void moisture(double *trans, double *dWater, double potGrossPsn, double vpd,
622 double soilWater) {
623 // potential transpiration in the absense of plant water stress
624 // (cm H20 * day^-1)
625 double potTrans;
626 // amount of water available for photosynthesis
627 double removableWater;
628 // water use efficiency, in mg CO2 fixed * g^-1 H20 transpired
629 double wue;
630
631 if (potGrossPsn < TINY) { // avoid divide by 0
632 *trans = 0.0; // no photosynthesis -> no transpiration
633 *dWater = 1; // dWater doesn't matter, since we don't have any
634 // photosynthesis
635 }
636
637 else {
638 // :: from [1], eq (A13)
639 wue = params.wueConst / vpd;
640
641 // 1000 converts g to mg; 44/12 converts g C to g CO2, 1/10000 converts m^2
642 // to cm^2
643 // :: from [1], eq (A14) plus unit conversion as described there
644 potTrans = potGrossPsn / wue * 1000.0 * (44.0 / 12.0) * (1.0 / 10000.0);
645
646 // :: from [1], discussion below eq (A14)
647 removableWater = soilWater * params.waterRemoveFrac;
648
649 // :: from [2], snowpack modification
650 if (climate->tsoil < params.frozenSoilThreshold) {
651 // frozen soil - less or no water available
652 // frozen soil effect: fraction of water available if soil is frozen
653 // (assume amount of water available in frozen soil scales linearly
654 // with amount of water available in thawed soil)
655 removableWater *= params.frozenSoilEff;
656 }
657
658 // :: from [1], eq (A15)
659 *trans = fmin(removableWater, potTrans);
660
661 // :: from [1], eq (A16)
662 *dWater = *trans / potTrans;
663 }
664}

713 // [TAG:UNKNOWN_PROVENANCE] leaf phenology combination

766 *snowFall = climate->precip / climate->length;

777 and evaporation from pools on the ground

780 evap. is a constant fraction Note that we don't evaporate snow here (we'll

799// snowpack dynamics:
800// calculate snow melt (cm water equiv./day) & sublimation (cm water equiv./day)
801// ensure we don't overdrain the snowpack (so that it becomes negative)
802// snowFall in cm/day
803void snowPack(double *snowMelt, double *sublimation, double snowFall) {
804 // conversion factor for sublimation
805 static const double CONVERSION = (RHO * CP) / GAMMA * (1. / LAMBDA_S) *
806 1000. * 1000. * (1. / 10000) * SEC_PER_DAY;
807 // 1000 converts kg to g, 1000 converts kPa to Pa, 1/10000 converts m^2 to
808 // cm^2
809
810 double rd; // aerodynamic resistance between ground and canopy air space
811 // (sec/m)
812 double snowRemaining; // to make sure we don't get rid of more than there is
813
814 // if no snow, set fluxes to 0
815 if (envi.snow <= 0) {
816 *snowMelt = 0;
817 *sublimation = 0;
818 }
819
820 // else: there is snow
821 else {
822 // first calculate sublimation, then snow melt
823 // (if there's not enough snow to do both, priority given to sublimation)
824 rd = (params.rdConst) / (climate->wspd); // aerodynamic resistance (sec/m)
825 *sublimation = CONVERSION * (E_STAR_SNOW - climate->vPress) / rd;
826
827 snowRemaining = envi.snow + (snowFall * climate->length);
828
829 // remove to allow sublimation of a negative amount of snow
830 // right now we can't sublime a negative amount of snow
831 if (*sublimation < 0) {
832 *sublimation = 0;
833 }
834
835 // make sure we don't sublime more than there is to sublime:
836 if (snowRemaining - (*sublimation * climate->length) < 0) {
837 *sublimation = snowRemaining / climate->length;
838 snowRemaining = 0;
839 }
840
841 else {
842 snowRemaining -= (*sublimation * climate->length);
843 }
844
845 // below freezing: no snow melt
846 if (climate->tair <= 0) {
847 *snowMelt = 0;
848 }
849
850 // above freezing: melt snow
851 else {
852 *snowMelt = params.snowMelt * climate->tair; // snow melt proportional to
853 // temp.
854
855 // make sure we don't melt more than there is to melt:
856 if (snowRemaining - (*snowMelt * climate->length) < 0) {
857 *snowMelt = snowRemaining / climate->length;
858 }
859 } // end else above freezing
860 } // end else there is snow
861} // end snowPack

878void calcSoilWaterFluxes(double *fastFlow, double *evaporation,
879 double *drainage, double water, double netRain,
880 double snowMelt, double trans) {
881 // conversion factor for evaporation
882 // 1000 converts kg to g, 1000 converts kPa to Pa, 1/10000 converts m^2 to
883 // cm^2
884 static const double CONVERSION = (RHO * CP) / GAMMA * (1. / LAMBDA) * 1000. *
885 1000. * (1. / 10000) * SEC_PER_DAY;
886 // keep running total of water remaining, in cm to make sure we don't
887 // evaporate or drain too much, and so we can drain any overflow
888 double waterRemaining;
889 // keep track of net water into soil, in cm/day
890 double netIn;
891 // aerodynamic resistance between ground and canopy air space (sec/m)
892 double rd;
893 // bare soil surface resistance (sec/m)
894 double rsoil;
895
896 netIn = netRain + snowMelt;
897
898 // fast flow: fraction that goes directly to drainage
899 *fastFlow = netIn * params.fastFlowFrac;
900 netIn -= *fastFlow;
901
902 // calculate evaporation:
903 // first calculate how much water is left to evaporate (used later)
904 waterRemaining = water + netIn * climate->length - trans * climate->length;
905
906 // if there's a snow pack, don't evaporate from soil:
907 if (envi.snow > 0) {
908 *evaporation = 0;
909 }
910
911 // else no snow pack:
912 else {
913 double waterFrac = water / params.soilWHC;
914 rsoil = exp(params.rSoilConst1 - params.rSoilConst2 * (waterFrac));
915 rd = (params.rdConst) / (climate->wspd); // aerodynamic resistance (sec/m)
916 *evaporation = CONVERSION * climate->vpdSoil / (rd + rsoil);
917 // by using vpd we assume that relative humidity of soil pore space is 1
918 // (when this isn't true, there won't be much water evaporated anyway)
919
920 // remove to allow negative evaporation (i.e. condensation):
921 if (*evaporation < 0) {
922 *evaporation = 0;
923 }
924
925 // make sure we don't evaporate more than we have:
926 if (waterRemaining - (*evaporation * climate->length) < TINY) {
927 // leave a tiny little bit, to avoid negative water due to round-off
928 // errors
929 *evaporation = (waterRemaining - TINY) / climate->length;
930 waterRemaining = 0;
931 } else {
932 waterRemaining -= (*evaporation * climate->length);
933 }
934 }
935
936 // drain any water that remains beyond water holding capacity:
937 if (waterRemaining > params.soilWHC) {
938 *drainage = (waterRemaining - params.soilWHC) / (climate->length);
939 } else {
940 *drainage = 0;
941 }
942}

1211void calculateFluxes(void) {

1219 double lai;

1221 double folResp, woodResp;

1223 // (g C * m^-2 ground area * day^-1)
1224 double growthResp;
1225 // net rain, equal to (rain - immedEvap) (cm/day)
1226 double netRain;
1227
1228 // Psn, moisture and water fluxes

1442void updateMainPools() {
1443 // Update the stocks, with fluxes adjusted for length of time step.
1444 // Notes:
1445 // - GPP shows up twice (direct + via NPP --> woodCreation), but
1446 // the math works out to:
1447 // envi.plantWoodC += NPP_allocation_to_wood − woodLitter.
1448 // - The soil C pool(s) (envi.soil, envi.fineRootC, envi.CoarseRootC)
1449 // are updated in updatePoolsForSoil().
1450
1451 // :: from [1], eq (A1), where:
1452 // GPP = fluxes.photosynthesis
1453 // R_a = fluxes.rVeg
1454 // L_w = fluxes.woodLitter
1455 // L = fluxes.leafCreation
1456 // :: from [3], additional root terms (woodCreation, fineRootC, coarseRootC)
1457 // from NPP allocation
1458 envi.plantWoodC += (fluxes.photosynthesis + fluxes.woodCreation -
1459 fluxes.leafCreation - fluxes.woodLitter - fluxes.rVeg -
1460 fluxes.coarseRootCreation - fluxes.fineRootCreation) *
1461 climate->length;
1462
1463 // :: from [1], eq (A2), where:
1464 // L = fluxes.leafCreation
1465 // L_L = fluxes.leafLitter
1466 envi.plantLeafC +=
1467 (fluxes.leafCreation - fluxes.leafLitter) * climate->length;
1468
1469 // :: from [1], eq (A4), where:
1470 // P = (fluxes.rain - fluxes.immedEvap)
1471 // T = (fluxes.transpiration + fluxes.evaporation)
1472 // D = (fluxes.bottomDrainage + fluxes.fastFlow)
1473 // from [2], addition of fluxes.snowMelt, as well as the modifications in
1474 // the terms above: immedEvap in P, evaporation in T, and fastFlow in D
1475 envi.soilWater +=
1476 (fluxes.rain + fluxes.snowMelt - fluxes.immedEvap - fluxes.fastFlow -
1477 fluxes.evaporation - fluxes.transpiration - fluxes.drainage) *
1478 climate->length;
1479
1480 // if ctx.snow = 0, some or all of these fluxes will always be 0
1481 // :: from [2], addition of snowpack description
1482 envi.snow += (fluxes.snowFall - fluxes.snowMelt - fluxes.sublimation) *
1483 climate->length;
1484}

1496void updatePoolsForSoil(void) {
1497
1498 // UPDATE POOLS
1499 if (ctx.microbes) {
1500 double microbeEff = params.efficiency;
1501
1502 // :: from [1] for litter terms
1503 // :: from [3] for root terms
1504 // :: from [4] for microbeIngestion term
1505 // Note: no rSoil term here, as soil resp is handled by microbeIngestion
1506 envi.soil +=
1507 (fluxes.coarseRootLoss + fluxes.fineRootLoss + fluxes.woodLitter +
1508 fluxes.leafLitter - fluxes.microbeIngestion) *
1509 climate->length;
1510 // microbeC additions due to ingestion and incorporation of root exudates,
1511 // with reduction from microbe respiration
1512 // :: from [4], eq (5.9) for first (ingestion) term,
1513 // :: eq (5.11) used for soilPulse, and
1514 // :: eq (5.12) used for maintRespiration
1515 envi.microbeC += (microbeEff * fluxes.microbeIngestion + fluxes.soilPulse -
1516 fluxes.maintRespiration) *
1517 climate->length;
1518
1519 // rSoil is maintenance resp + growth (microbe) resp
1520 // :: from [4], eq (5.10) for microbe term
1521 fluxes.rSoil =
1522 fluxes.maintRespiration + (1 - microbeEff) * fluxes.microbeIngestion;
1523 } else {
1524 if (ctx.litterPool) {
1525 // :: from [2], litter model description
1526 envi.litter += (fluxes.woodLitter + fluxes.leafLitter -
1527 fluxes.litterToSoil - fluxes.rLitter) *
1528 climate->length;
1529
1530 // from [2] and [3], litter and root terms respectively
1531 envi.soil += (fluxes.coarseRootLoss + fluxes.fineRootLoss +
1532 fluxes.litterToSoil - fluxes.rSoil) *
1533 climate->length;
1534 } else {
1535 // Normal pool (single pool, no microbes)
1536 // :: from [1] (and others, TBD), eq (A3), where:
1537 // L_w = fluxes.woodLitter
1538 // L_l = fluxes.leafLitter
1539 // R_h = fluxes.rSoil
1540 // :: from [3], root terms
1541 envi.soil += (fluxes.coarseRootLoss + fluxes.fineRootLoss +
1542 fluxes.woodLitter + fluxes.leafLitter - fluxes.rSoil) *
1543 climate->length;
1544 }
1545 }
1546 // :: from [3], root model description
1547 envi.coarseRootC +=
1548 (fluxes.coarseRootCreation - fluxes.coarseRootLoss - fluxes.rCoarseRoot) *
1549 climate->length;
1550 envi.fineRootC +=
1551 (fluxes.fineRootCreation - fluxes.fineRootLoss - fluxes.rFineRoot) *
1552 climate->length;
1553}
