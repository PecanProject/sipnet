569/* Read climate file into linked lists,
570 make firstClimates be a vector where each element is a pointer to the head of
571 a list corresponding to one spatial location
572
573 return an array containing the number of time steps in each location
574 (dynamically allocated with malloc)
575
576 numLocs = number of spatial locations: there should be one set of entries in
577 climFile for each location, or a location can be skipped, and we'll use
578 climate at location 0 for that location, too
579
580 format of climFile:
581 each line represents one time step, with the following format:
582
583 location year day time intervalLength tair tsoil par precip vpd vpdSoil
584 vPress wspd soilWetness
585
586 NOTE: there should be NO blank lines in file, even between different spatial
587 locations; all entries for a given location should be contiguous, and
588 locations should be in ascending order. There may be locations for which
589 there is no climate data, in which case we use climate from location 0
590*/
591int *readClimData(char *climFile, int numLocs) {
592 FILE *in;
593 ClimateNode *curr, *next;
594 int loc, year, day;
595 int lastYear = -1;
596 double time, length; // time in hours, length in days (or fraction of day)
597 double tair, tsoil, par, precip, vpd, vpdSoil, vPress, wspd, soilWetness;
598 int currLoc;
599 int count;
600 int i;
601 int *steps; // # of time steps in each location

604 double thisGdd; // growing degree days of this time step
605 double gdd = 0.0; // growing degree days since the last Jan. 1
606#endif

609

611 for (i = 0; i < numLocs; i++) {

613 }

616
617 status = fscanf(in, "%d %d %d %lf %lf %lf %lf %lf %lf %lf %lf %lf %lf %lf",

624

628 }

630 if (loc != 0) {
631 printf("Error reading climate file: first location must be loc. 0\n");
632 exit(EXIT_CODE_INPUT_FILE_ERROR);
633 }
634
635 firstClimates =
636 (ClimateNode **)malloc(numLocs * sizeof(ClimateNode *)); // vector of
637 // heads
638 for (currLoc = 0; currLoc < numLocs; currLoc++) {
639 firstClimates[currLoc] = NULL; // default is null

641 // if firstClimates[i] stays null for some i, that means we didn't read any
642 // climate data for location i, so use climate from location 0
643
644 currLoc = loc;
645 // allocate space for head
646 firstClimates[currLoc] = (ClimateNode *)malloc(sizeof(ClimateNode));
647 next = firstClimates[currLoc];
648 count = 0;
649 while (status != EOF) {

651 curr = next;
652 count++; // # of time steps in this location
653
654 curr->year = year;
655 curr->day = day;

674 curr->wspd = wspd;
675 if (curr->wspd < TINY) {
676 curr->wspd = TINY; // avoid divide by zero
677 }
678 curr->soilWetness = soilWetness;
679
680#if GDD
681 if (year != lastYear) { // HAPPY NEW YEAR!
682 gdd = 0; // reset growing degree days
683 }
684 thisGdd = tair * length;

686 thisGdd = 0;

690#endif

696 &precip, &vpd, &vpdSoil, &vPress, &wspd, &soilWetness);

700 // location (currLoc), make sure new location is valid, and act
701 // accordingly

703 if (status != NUM_CLIM_FILE_COLS) {
704 printf("Error reading climate file: bad data near loc %d year %d"
705 " day %d\n",
706 loc, year, day);
707 exit(EXIT_CODE_INPUT_FILE_ERROR);
708 }

710 if (loc == currLoc) {
711 // still reading climate records from the same place: add a node at end
712 // of linked list
713 next = (ClimateNode *)malloc(sizeof(ClimateNode));
714 curr->nextClim = next;
715 // set this down here rather than at top of loop so head treated the
716 // same as rest of list
717 } else if (loc >= numLocs) { // (loc == numLocs is an error since we use
718 // 0-indexing)
719 printf("Error reading climate file: trying to read location %d, but "
720 "numLocs = %d\n",
721 loc, numLocs);
722 exit(1);
723 } else if (loc > currLoc) {
724 // we've advanced to the next location (note: possible that we skipped
725 // some locations: this is OK)
726 curr->nextClim = NULL; // terminate last linked list
727 steps[currLoc] = count; // record the number of time steps for last
728 // location
729 lastYear = -1;
730 count = 0; // reset count of # of time steps
731#if GDD
732 gdd = 0; // reset growing degree days
733#endif
734 currLoc = loc;
735 firstClimates[currLoc] =
736 (ClimateNode *)malloc(sizeof(ClimateNode)); // allocate space for
737 // head
738 next = firstClimates[currLoc]; // we'll start writing to next
739 // linked list
740 } else { // loc < currLoc
741 printf("Error reading climate file: was reading location %d, trying to "
742 "read location %d\n",
743 currLoc, loc);
744 printf("Climate records for a given location should be contiguous, and "
745 "locations should be in ascending order\n");
746 exit(1);
747 }
748 } else { // status == EOF - no more records
749 curr->nextClim = NULL; // terminate this last linked list
750 steps[currLoc] = count; // record the number of time steps for last
751 // location
752 }
753
754 } // end while
755
756 fclose(in);
757
758 for (i = 0; i < numLocs; i++) {
759 if (steps[i] == 0) { // nothing read for this location
760 steps[i] = steps[0]; // this location will duplicate location 0
761 }
762 }
763 return steps;
764}

1044void calcLightEff(double *lightEff, double lai, double par) {
1045

1047 // as of March 2007 ... contact Dr. Maggie Prater Maggie.Prater@colorado.edu

1051 // layers

1054 double cumLai; // lai from this layer up
1055 double lightIntensity;
1056 double currLightEff = 0.0, cumLightEff;
1057 double currfAPAR = 0.0, cumfAPAR;
1058 int coeff; // current coefficient in Simpson's rule
1059
1060 if (lai > 0 && par > 0) { // must have at least some leaves and some light
1061 cumLightEff = 0.0; // the running sum
1062 layer = 0;
1063 coeff = 1;
1064 int err;
1065 double fAPAR; // Calculation of fAPAR according to 1 - exp(attenuation*LAI)
1066 // double APAR; // Absorbed PAR by the canopy
1067 // double lightIntensityTop, lightIntensityBottom; // PAR absorbed by the
1068 // canopy
1069 cumfAPAR = 0.0;
1070
1071 while (layer <= NUM_LAYERS) {
1072 cumLai = lai * ((double)layer / NUM_LAYERS); // lai from this layer up
1073 // (starting at top)
1074
1075 // between 0 and par
1076 lightIntensity = par * exp(-1.0 * params.attenuation * cumLai);
1077
1078 // between 0 and 1
1079 currLightEff = (1 - pow(2, (-1.0 * lightIntensity / params.halfSatPar)));
1080
1081 // when lightIntensity = halfSatPar, currLightEff = 1/2
1082 cumLightEff += coeff * currLightEff;
1083
1084 currfAPAR = 1 - (lightIntensity / par);
1085 cumfAPAR += coeff * currfAPAR;
1086
1087 // now move to the next layer:
1088 layer++;
1089 coeff = 2 * (1 + layer % 2); // coeff. goes 1, 4, 2, 4, ..., 2, 4, 2
1090 }
1091 // last value should have had a coefficient of 1, but actually had a
1092 // coefficient of 2, so subtract 1:
1093 cumLightEff -= currLightEff;
1094 *lightEff = cumLightEff / (3.0 * NUM_LAYERS); // multiplying by (h/3) in
1095 // Simpson's rule
1096
1097 // now calculate fAPAR
1098 cumfAPAR -= currfAPAR;
1099 // the average value, multiplying by h/3 in Simpson's rule, and dividing by
1100 // the number of steps
1101 fAPAR = cumfAPAR / (3.0 * NUM_LAYERS);
1102
1103 // lightIntensityTop = par; // Energy at the top of the canopy
1104 // lightIntensityBottom = par * exp(-1.0 * params.attenuation * lai); // LAI
1105 // at the bottom of the canopy
1106 // between 0 and par

1581// snowpack dynamics:

1584// snowFall in cm/day
1585void snowPack(double *snowMelt, double *sublimation, double snowFall) {
1586 // conversion factor for sublimation
1587 static const double CONVERSION = (RHO * CP) / GAMMA * (1. / LAMBDA_S) *
1588 1000. * 1000. * (1. / 10000) * SEC_PER_DAY;

1593 // (sec/m)

1599 *sublimation = 0;
1600 }

1603 else {

1605 // (if there's not enough snow to do both, priority given to sublimation)
1606 rd = (params.rdConst) / (climate->wspd); // aerodynamic resistance (sec/m)
1607 *sublimation = CONVERSION * (E_STAR_SNOW - climate->vPress) / rd;
1608

1610
1611 // remove to allow sublimation of a negative amount of snow
1612 // right now we can't sublime a negative amount of snow
1613 if (*sublimation < 0) {
1614 *sublimation = 0;
1615 }
1616
1617 // make sure we don't sublime more than there is to sublime:
1618 if (snowRemaining - (*sublimation * climate->length) < 0) {
1619 *sublimation = snowRemaining / climate->length;
1620 snowRemaining = 0;
1621 }
1622
1623 else {
1624 snowRemaining -= (*sublimation * climate->length);
1625 }
1626

1628 if (climate->tair <= 0) {
1629 *snowMelt = 0;
1630 }
1631
1632 // above freezing: melt snow
1633 else {
1634 *snowMelt = params.snowMelt * climate->tair; // snow melt proportional to
1635 // temp.
1636
1637 // make sure we don't melt more than there is to melt:
1638 if (snowRemaining - (*snowMelt * climate->length) < 0) {
1639 *snowMelt = snowRemaining / climate->length;
1640 }
1641 } // end else above freezing
1642 } // end else there is snow
1643} // end snowPack

1666 // 1000 converts kg to g, 1000 converts kPa to Pa, 1/10000 converts m^2 to

1668
1669 double waterRemaining; /* keep running total of water remaining, in cm
1670 (to make sure we don't evap. or drain too much, and so we can drain
1671 any overflow) */
1672 double netIn; // keep track of net water into soil, in cm/day
1673 double rd; // aerodynamic resistance between ground and canopy air space
1674 // (sec/m)
1675 double rsoil; // bare soil surface resistance (sec/m)
1676
1677 netIn = netRain + snowMelt;
1678
1679 // fast flow: fraction that goes directly to drainage
1680 *fastFlow = netIn * params.fastFlowFrac;
1681 netIn -= *fastFlow;
1682
1683 // calculate evaporation:
1684 // first calculate how much water is left to evaporate (used later)
1685 waterRemaining =
1686 water + netIn * climate->length - fluxesOut * climate->length;
1687
1688 // if there's a snow pack, don't evaporate from soil:
1689 if (envi.snow > 0) {
1690 *evaporation = 0;
1691 }
1692
1693 // else no snow pack:
1694 else {
1695 rd = (params.rdConst) / (climate->wspd); // aerodynamic resistance (sec/m)
1696 rsoil = exp(params.rSoilConst1 - params.rSoilConst2 * (water / whc));
1697 *evaporation = CONVERSION * climate->vpdSoil / (rd + rsoil);
1698 // by using vpd we assume that relative humidity of soil pore space is 1
1699 // (when this isn't true, there won't be much water evaporated anyway)
1700
1701 // remove to allow negative evaporation (i.e. condensation):
1702 if (*evaporation < 0) {
1703 *evaporation = 0;
1704 }
1705
1706 // make sure we don't evaporate more than we have:
1707 if (waterRemaining - (*evaporation * climate->length) < TINY) {
1708 // leave a tiny little bit, to avoid negative water due to round-off
1709 // errors
1710 *evaporation = (waterRemaining - TINY) / climate->length;
1711 waterRemaining = 0;
1712 } else {
1713 waterRemaining -= (*evaporation * climate->length);
1714 }
1715 }
1716
1717#if LITTER_WATER_DRAINAGE // we're calculating drainage even when evap. layer
1718 // is not overflowing
1719 // drainage rate is proportional to fractional soil moisture
1720 *drainage = params.litWaterDrainRate * (water / whc);
1721 // make sure we don't drain more than we have:
1722 if (waterRemaining - (*drainage * climate->length) < TINY) {
1723 // leave a tiny little bit, to avoid negative water due to round-off errors
1724 *drainage = (waterRemaining - TINY) / climate->length;
1725 waterRemaining = 0;
1726 } else
1727 waterRemaining -= (*drainage * climate->length);
1728#else // LITTER_WATER_DRAINAGE = 0
1729 *drainage = 0;
1730#endif
1731
1732 // drain any water that remains beyond water holding capacity:
1733 if (waterRemaining > whc) {
1734 *drainage += (waterRemaining - whc) / (climate->length);
1735 }
1736}
