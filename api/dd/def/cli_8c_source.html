89void parseCommandLineArgs(int argc, char *argv[]) {
90 /* getopt_long stores the option index here. */
91 int longIndex = 0;
92 int shortIndex;
93 // get command-line arguments:
94 while ((shortIndex = getopt_long(argc, argv, "hi:v", long_options,
95 &longIndex)) != -1) {
96
97 switch (shortIndex) {
98 case 0:
99 // long form option, flag == 0
100 updateIntContext(argNameMap[longIndex], ctx.tmpFlag, CTX_COMMAND_LINE);
101 break;
102 case 'h':
103 usage(argv[0]);
104 exit(EXIT_CODE_SUCCESS);
105 case 'i':
106 if (strlen(optarg) >= FILENAME_MAXLEN) {
107 printf("ERROR: input filename %s exceeds maximum length of %d\n",
108 optarg, FILENAME_MAXLEN);
109 printf("Either change the name or increase INPUT_MAXNAME in "
110 "frontend.c\n");
111 exit(1);
112 }
113 updateCharContext("inputFile", optarg, CTX_COMMAND_LINE);
114 break;
115 case 'v':
116 version();
117 exit(EXIT_CODE_SUCCESS);
118 default:
119 usage(argv[0]);
120 exit(EXIT_CODE_BAD_CLI_ARGUMENT);
121 }
122 }

125// Safety check: make sure the elements of the cli argNameMap are actually
126// valid members - that is, that we can successfully find a metadata for each
127// one. This protects against changing a field name here and forgetting to
128// update that map.
129void checkCLINameMap(void) {
130 int numFlags = 0;
131 struct context_metadata *s;
132
133 // Make sure everything in argNameMap maps to a valid metadata
134 for (int ind = 0; ind < 2 * NUM_FLAG_OPTIONS; ++ind) {
135 s = getContextMetadata(argNameMap[ind]);
136 if (s == NULL) {
137 logError("Internal error: mismatched argNameMap and Context contents; "
138 "missing Context metadata\n");
139 exit(EXIT_CODE_INTERNAL_ERROR);
140 }
141 }
142 // Make sure the number of flag metadata structs equals NUM_FLAG_OPTIONS
143 for (s = ctx.metaMap; s != NULL;
144 s = (struct context_metadata *)(s->hh.next)) {
145 if (s->type == CTX_INT) {
146 ++numFlags;
147 }
148 }
149 if (numFlags != NUM_FLAG_OPTIONS) {
150 logError("Internal error: mismatched argNameMap and Context contents; "
151 "missing argNameMap entry\n");
152 exit(EXIT_CODE_INTERNAL_ERROR);
153 }
