112void parseCommandLineArgs(int argc, char *argv[]) {
113 /* getopt_long stores the option index here. */
114 int longIndex = 0;
115 int shortIndex;
116 // get command-line arguments:
117 while ((shortIndex = getopt_long(argc, argv, "hi:v", long_options,
118 &longIndex)) != -1) {
119
120 switch (shortIndex) {
121 case 0:
122 // long form option, flag == 0
123 updateIntContext(argNameMap[longIndex], ctx.tmpFlag, CTX_COMMAND_LINE);
124 break;
125 case 'f':
126 if (strlen(optarg) >= FILENAME_MAXLEN) {
127 printf("ERROR: filename %s exceeds maximum length of %d\n", optarg,
128 FILENAME_MAXLEN);
129 printf("Either change the name or increase INPUT_MAXNAME in "
130 "frontend.c\n");
131 exit(1);
132 }
133 updateCharContext("fileName", optarg, CTX_COMMAND_LINE);
134 break;
135 case 'h':
136 usage(argv[0]);
137 exit(EXIT_CODE_SUCCESS);
138 case 'i':
139 if (strlen(optarg) >= FILENAME_MAXLEN) {
140 printf("ERROR: input filename %s exceeds maximum length of %d\n",
141 optarg, FILENAME_MAXLEN);
142 printf("Either change the name or increase INPUT_MAXNAME in "
143 "frontend.c\n");
144 exit(1);
145 }
146 updateCharContext("inputFile", optarg, CTX_COMMAND_LINE);
147 break;
148 case 'v':
149 version();
150 exit(EXIT_CODE_SUCCESS);
151 default:
152 usage(argv[0]);
153 exit(EXIT_CODE_BAD_CLI_ARGUMENT);
154 }
155 }

158// Safety check: make sure the elements of the cli argNameMap are actually
159// valid members - that is, that we can successfully find a metadata for each
160// one. This protects against changing a field name here and forgetting to
161// update that map.
162void checkCLINameMap(void) {
163 int numFlags = 0;
164 struct context_metadata *s;
165
166 // Make sure everything in argNameMap maps to a valid metadata
167 for (int ind = 0; ind < 2 * NUM_FLAG_OPTIONS; ++ind) {
168 s = getContextMetadata(argNameMap[ind]);
169 if (s == NULL) {
170 logInternalError("mismatched argNameMap and Context contents; "
171 "no metadata for %s\n",
172 argNameMap[ind]);
173 exit(EXIT_CODE_INTERNAL_ERROR);
174 }
175 }
176 // Make sure the number of flag metadata structs equals NUM_FLAG_OPTIONS
177 for (s = ctx.metaMap; s != NULL;
178 s = (struct context_metadata *)(s->hh.next)) {
179 if (s->isFlag) {
180 ++numFlags;
181 }
182 }
183 if (numFlags != NUM_FLAG_OPTIONS) {
184 logInternalError("mismatched argNameMap and Context contents; "
185 "%d metaMap entries, NUM_FLAG_OPTIONS=%d\n",
186 numFlags, NUM_FLAG_OPTIONS);
187 exit(EXIT_CODE_INTERNAL_ERROR);
188 }
