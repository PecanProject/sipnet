99/* PRE: dataTypeSwitches is of size MAX_DATA_TYPES
100 dataTypeSwitches[i] is 1 if we are including this data type in optimization,
101 0 if not dataTypeIndices must be big enough to hold MAX_DATA_TYPES ints
102
103 POST: dataTypeIndices[0..numDataTypes-1] contains the indices of the 1-valued
104 elements in dataTypeSwitches
105 - i.e. all of the data types included in optimization
106 (remaining elements of dataTypeIndices are undefined)
107 returns numDataTypes: number of data types included in optimization
108 */
109int parseDataTypeSwitches(int dataTypeIndices[], int dataTypeSwitches[]) {
110 int numDataTypes;
111 int i;
112
113 numDataTypes = 0;
114 for (i = 0; i < MAX_DATA_TYPES; i++) {
115 if (dataTypeSwitches[i]) {
116 dataTypeIndices[numDataTypes] = i;
117 numDataTypes++;
118 }
119 }
120
121 return numDataTypes;
122}
