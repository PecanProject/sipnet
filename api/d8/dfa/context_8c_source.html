51// With all the different permutations of spellings for config params, lets
52// strip names down by removing chars like dashes and underscores, and
53// converting to lowercase. This will allow different versions to still find
54// the relevant metadata, while retaining uniqueness.
55void nameToKey(const char *name) {
56 int keyInd = 0;
57 // Drop all non-alphanumeric chars (eg '-', '_'), and convert to lowercase
58 for (int i = 0; name[i]; i++) {
59 if (isalnum(name[i])) {
60 keyName[keyInd] = tolower(name[i]); // NOLINT
61 ++keyInd;
62 }
63 }
64 keyName[keyInd] = '\0';
65}
