64// With all the different permutations of spellings for config params, lets
65// strip names down by removing chars like dashes and underscores, and
66// converting to lowercase. This will allow different versions to still find
67// the relevant metadata, while retaining uniqueness.
68void nameToKey(const char *name) {
69 int keyInd = 0;
70 // Drop all non-alphanumeric chars (eg '-', '_'), and convert to lowercase
71 for (int i = 0; name[i]; i++) {
72 if (isalnum(name[i])) {
73 keyName[keyInd] = tolower(name[i]); // NOLINT
74 ++keyInd;
75 }
76 }
77 keyName[keyInd] = '\0';
78}
