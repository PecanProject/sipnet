45// With all the different permutations of spellings for config params, lets
46// strip names down by removing chars like dashes and underscores, and
47// converting to lowercase. This will allow different versions to still find
48// the relevant metadata, while retaining uniqueness.
49void nameToKey(const char *name) {
50 int keyInd = 0;
51 // Drop all non-alphanumeric chars (eg '-', '_'), and convert to lowercase
52 for (int i = 0; name[i]; i++) {
53 if (isalnum(name[i])) {
54 keyName[keyInd] = tolower(name[i]); // NOLINT
55 ++keyInd;
56 }
57 }
58 keyName[keyInd] = '\0';
59}
