110// reset parameter, log likelihood variables, etc. for start of MCMC chain
111// parameters from spatialParams through outputInfo are outputs, rest are inputs
112// (spatialParams also provides some inputs) all output parameters are arrays
113// except ltotnew, ltotold and ltotmax; loglikely, outputInfo and sigma are
114// spatial (1st dimension of sigma and outputInfo is spatial dimension;
115// loglikely is only one-dimensional)
116//
117// if loc = -1, we're running everywhere (number of locations, which is also the
118// size of loglikely and sigma, specified in spatialParams)
119//
120// if loc >= 0, we're running at a single location; in this case, 1st dimension
121// of loglikely and sigma only has length 1
122//
123// randomStart is boolean: do we start each chain with a random param. set (as
124// opposed to guess values)?
125void reset(
126 SpatialParams *spatialParams, double *loglikely, double *ltotnew,
127 double *ltotold, double *ltotmax, double **sigma, OutputInfo **outputInfo,
128 int loc, int randomStart, double addFraction, FILE *userOut,
129 double (*likely)(double *, OutputInfo *, int, SpatialParams *, double,
130 void (*)(double **, int, int *, SpatialParams *, int),
131 int[], int, int, double[]),
132 double paramWeight,
133 void (*model)(double **, int, int *, SpatialParams *, int),
134 int dataTypeIndices[], int numDataTypes, int costFunction,
135 double dataTypeWeights[]) {
136 int currLoc;
137 int firstLoc, lastLoc, locIndex;
138
139 if (loc == -1) { // running at all locations
140 firstLoc = 0;
141 lastLoc = spatialParams->numLocs - 1;
142 } else {
143 // only running at one location
144 firstLoc = lastLoc = loc;
145 }
146 // reset value, best & knob (set knob to addFraction)
147 resetSpatialParams(spatialParams, addFraction, randomStart);
148
149 /*compute log-likelihood of parameter set*/
150 for (currLoc = firstLoc; currLoc <= lastLoc; currLoc++) {
151 locIndex = currLoc - firstLoc; // index into arrays
152 loglikely[locIndex] = // NOLINTNEXTLINE (outputInfo false warning)
153 -1.0 * (*likely)(sigma[locIndex], outputInfo[locIndex], currLoc,
154 spatialParams, paramWeight, model, dataTypeIndices,
155 numDataTypes, costFunction, dataTypeWeights);
156 }
157 *ltotold = *ltotmax = *ltotnew = sumArray(loglikely, lastLoc - firstLoc + 1);
158
159 /*screen output*/
160 fprintf(userOut, "\n\t\t\t**ESTIMATOR**\n");
161 writeChangeableParamInfo(spatialParams, loc, userOut);
162 fprintf(userOut, "\n\t\tlTOT\tnew= %9.6f\tmax= %9.6f\n", *ltotnew, *ltotmax);
163}

167/* write (to chainInfo file) all information needed to restart a chain from a
168 given point format of file: binary file as follows: ltotold ltotmax <np> *
169 (value best knob) where np is the number of changeable parameters
170
171 if loc = -1, then for each spatial parameter we write all values (i.e. value
172 at each point) followed by all bests and all knobs
173
174 if loc >= 0, then we only write value, best and knob at the given location
175 where all values are doubles
176*/
177void writeChainInfo(char *chainInfo, double ltotold, double ltotmax,
178 SpatialParams *spatialParams, int loc) {
179 FILE *out;
180 int np; // number of changeable parameters
181 int i, index;
182 int firstLoc, lastLoc, thisFirstLoc, thisLastLoc, currLoc;
183 double value[1]; // next value to be written
184
185 out = openFile(chainInfo, "w");
186
187 np = spatialParams->numChangeableParams;
188 if (loc == -1) { // running at all locations
189 firstLoc = 0;
190 lastLoc = spatialParams->numLocs - 1;
191 } else {
192 // only running at one location
193 firstLoc = lastLoc = loc;
194 }
195
196 // write last and maximum log likelihoods:
197 fwrite(&ltotold, sizeof(double), 1, out);
198 fwrite(&ltotmax, sizeof(double), 1, out);
199
200 // write parameter info:
201 for (i = 0; i < np; i++) {
202 index = spatialParams->changeableParamIndices[i]; // get index of ith
203 // changeable parameter
204 if (isSpatial(spatialParams, index)) {
205 thisFirstLoc = firstLoc;
206 thisLastLoc = lastLoc;
207 } else {
208 // non-spatial - we'll just use one location (doesn't matter which
209 // one)
210 thisFirstLoc = thisLastLoc = 0;
211 }
212
213 for (currLoc = thisFirstLoc; currLoc <= thisLastLoc; currLoc++) {
214 // if non-spatial, or loc != -1, we'll only execute this loop once
215 value[0] = getSpatialParam(spatialParams, index, currLoc);
216 fwrite(value, sizeof(double), 1, out);
217 }
218 for (currLoc = thisFirstLoc; currLoc <= thisLastLoc; currLoc++) {
219 // if non-spatial, or loc != -1, we'll only execute this loop once
220 value[0] = getSpatialParamBest(spatialParams, index, currLoc);
221 fwrite(value, sizeof(double), 1, out);
222 }
223 for (currLoc = thisFirstLoc; currLoc <= thisLastLoc; currLoc++) {
224 // if non-spatial, or loc != -1, we'll only execute this loop once
225 value[0] = getSpatialParamKnob(spatialParams, index, currLoc);
226 fwrite(value, sizeof(double), 1, out);
227 }
228 }
229
230 fclose(out);
231}

295/* Puts best parameters found in spatialParams
296 If loc = -1, run at all locations; if loc >= 0, run only at that single
297 location. randomStart is boolean: do we start each chain with a random param.
298 set (as opposed to guess values)? NOTE: anything but a scale factor of 1 goes
299 against theory */
300void metropolis(
301 char *outNameBase, SpatialParams *spatialParams, int loc,
302 double (*likely)(double *, OutputInfo *, int, SpatialParams *, double,
303 void (*)(double **, int, int *, SpatialParams *, int),
304 int[], int, int, double[]),
305 void (*model)(double **, int, int *, SpatialParams *, int),
306 double addFraction, long estSteps, int numAtOnce, int numChains,
307 int randomStart, long numSpinUps, double paramWeight, double scaleFactor,
308 int dataTypeIndices[], int numDataTypes, int costFunction,
309 double dataTypeWeights[], FILE *userOut) {
310 const double INC = pow(DEC, ((A_STAR - 1) / A_STAR));
311 // want INC^A_STAR * DEC^(1 - A_STAR) = 1
312
313 char histFileBase[256], histFileName[270], chainInfo[256];
314 FILE **histFiles; // vector of FILE ptrs (one for each location)
315 int accept, yes = 0;
316 long k;
317 int chainNum; // which starting chain are we on (for running multiple chains
318 // to convergence then choosing best)
319 int firstLoc, lastLoc, numLocs; // first, last and number of locations that
320 // we're actually running at (based on both loc and spatialParams->numLocs)
321 int thisFirstLoc, thisLastLoc; // for spatial params, same as firstLoc and
322 // lastLoc; for non-spatial params, both are 0
323 int currLoc, locIndex;
324 int ichg; // parameter to change
325 double range; // (max - min) of current parameter
326 double *loglikely; // spatial
327 double ltotnew, ltotold, ltotmax;
328 double bestChainLtotmax; // ltotmax of the best starting chain so far
329 double oldVal;
330 double *pold; // spatial
331 double pdelta, padd;
332 double **sigma; // sigma of this point, estimated by likely function (2-d
333 // array: sigma[i][j] is sigma at loc. i, data type j)
334 OutputInfo **outputInfo; // store error info and annual totals for each run
335 // & each data type 2-d array: outputInfo[i][j] is outputInfo at location i,
336 // data type j
337 int converged = 0; /* have we converged on correct param ranges yet?
338 we've converged when we're near A_STAR acceptance
339 If numChains > 1, converged will only be set to 1 after we've converged
340 (numChains) times */
341 long totalIters = numSpinUps + estSteps; // how many total steps to take once
342 // temperatures have converged
343
344 if (loc == -1) { // running at all locations
345 numLocs = spatialParams->numLocs;
346 firstLoc = 0;
347 lastLoc = numLocs - 1;
348 } else { // only running at one location
349 firstLoc = lastLoc = loc;
350 numLocs = 1;
351 }
352
353 loglikely = makeArray(numLocs);
354 pold = makeArray(numLocs);
355 sigma = make2DArray(numLocs, numDataTypes);
356 outputInfo = (OutputInfo **)malloc(numLocs * sizeof(OutputInfo *));
357 for (currLoc = firstLoc; currLoc <= lastLoc; currLoc++) {
358 outputInfo[currLoc - firstLoc] = newOutputInfo(numDataTypes, currLoc);
359 }
360 histFiles = (FILE **)malloc(numLocs * sizeof(FILE *)); // one file for each
361 // location
362
363 strcpy(histFileBase, outNameBase);
364 strcpy(chainInfo, outNameBase);
365 strcat(histFileBase, ".hist");
366 strcat(chainInfo, ".chain_info");
367
368 chainNum = 1;
369 fprintf(userOut, "\n\nRESETTING FOR START CHAIN %d of %d\n\n", chainNum,
370 numChains);
371 reset(spatialParams, loglikely, &ltotnew, &ltotold, &ltotmax, sigma,
372 outputInfo, loc, randomStart, addFraction, userOut, likely, paramWeight,
373 model, dataTypeIndices, numDataTypes, costFunction, dataTypeWeights);
374
375 writeChangeableParamInfo(spatialParams, loc, userOut);
376
377 bestChainLtotmax = -1.0 * DBL_MAX; // really friggin' small (i.e. really
378 // friggin' big in the negative direction)
379 k = 1;
380
381 // open all hist files (one for each location), assign file pointers
382 // (histFiles), write header to each
383 for (currLoc = firstLoc; currLoc <= lastLoc; currLoc++) {
384 locIndex = currLoc - firstLoc; // index into array
385 // append currLoc to end of histFileBase to get name of current file
386 sprintf(histFileName, "%s%d", histFileBase, currLoc);
387 histFiles[locIndex] = openFile(histFileName, "w");
388
389 // writeHistFileHeader(histFiles[locIndex],
390 // outputInfo[locIndex][0].numYears, spatialParams->numChangeableParams,
391 // numDataTypes);
392 writeHistFileBinHeader(histFiles[locIndex],
393 outputInfo[locIndex][0].numYears,
394 spatialParams->numChangeableParams, numDataTypes);
395 /* NOTE: 1) this has to be done AFTER call to reset, since
396 outputInfo.numYears is set in call to likely function
397
398 2) it doesn't matter which data type we use for outputInfo (here we use
399 0), since all will have the same numYears */
400 }
401
402 /*****metropolis loop***********************************************/
403
404 while ((converged == 0) || (k <= totalIters)) {
405 accept = 1; // so far, we're in accept mode - we haven't rejected the new
406 // point yet
407
408 /*select parameter to change at random*/
409 ichg = randomChangeableSpatialParam(spatialParams);
410
411 /* determine first and last location for THIS parameter (depends on whether
412 parameter is spatial) note that we'll still use firstLoc and lastLoc for
413 things like actually running model; thisFirstLoc and thisLastLoc just
414 refer to locations we care about for parameter-related things like
415 choosing a new parameter value */
416 if (isSpatial(spatialParams, ichg)) {
417 thisFirstLoc = firstLoc;
418 thisLastLoc = lastLoc;
419 } else {
420 // non-spatial - we just use one location (doesn't matter which one)
421 thisFirstLoc = thisLastLoc = 0;
422 }
423
424 /*change parameter*/
425 range = (getSpatialParamMax(spatialParams, ichg) -
426 getSpatialParamMin(spatialParams, ichg)); // range is the same for
427 // all locations
428 for (currLoc = thisFirstLoc; currLoc <= thisLastLoc; currLoc++) {
429 oldVal = getSpatialParam(spatialParams, ichg, currLoc);
430 pold[currLoc - thisFirstLoc] = oldVal; // remember old value
431 if (accept == 1) {
432 // we haven't set accept to 0 yet (if accept = 0 already, don't bother
433 // setting new param. values)
434
435 // pdelta is expressed as fraction of parameter's range
436 pdelta = getSpatialParamKnob(spatialParams, ichg, currLoc);
437 padd = (rand() * 1.0 / RAND_MAX - 0.5) * pdelta * range;
438 // new value will be oldVal + padd
439 if (checkSpatialParam(spatialParams, ichg, oldVal + padd) ==
440 0) { // outside allowable range
441 accept = 0; /* if new value is outside allowable range at any
442 location, reject point (equivalent to making likelihood tiny)
443 however, we'll continue the loop, because we need to fill pold
444 vector in order to reset param. values later */
445 } else {
446 // set new value equal to oldVal + padd
447 setSpatialParam(spatialParams, ichg, currLoc, oldVal + padd);
448 }
449 } // if accept == 1
450 } // for currLoc
451
452 if (accept == 1) {
453 // we're within allowable range at all locations; run
454 // model at all locations and check new total likelihood
455
456 /*compute log-likelihood of new parameter set*/
457 for (currLoc = firstLoc; currLoc <= lastLoc; currLoc++) {
458 locIndex = currLoc - firstLoc; // index into arrays
459 loglikely[locIndex] =
460 -1.0 * (*likely)(sigma[locIndex], outputInfo[locIndex], currLoc,
461 spatialParams, paramWeight, model, dataTypeIndices,
462 numDataTypes, costFunction, dataTypeWeights);
463 }
464 ltotnew = sumArray(loglikely, numLocs);
465
466 // compare "new" likelihood to "max" likelihood and act
467 if (ltotnew > ltotmax) {
468 ltotmax = ltotnew;
469 setAllSpatialParamBests(spatialParams, loc); /* set best equal to
470 current value for all parameters if loc == -1, this will set
471 bests at all locations */
472 }
473
474 /* compare new to old and accept or reject */
475 if (ltotnew > ltotold // standard check
476 ||
477 // note: anything but a scaleFactor of 1 goes against theory
478 randm() < scaleFactor * (ltotnew - ltotold)) {
479 accept = 1;
480 } else {
481 accept = 0;
482 }
483 } // end if (accept == 1)
484
485 /* act on acceptance */
486 if (accept == 1) {
487 /* update likelihoods */
488 ltotold = ltotnew;
489
490 yes++; // chalk up one more acceptance
491
492 if (converged == 0) { // we haven't converged yet - twist knob
493 // (equivalent to old pdelta)
494 for (currLoc = thisFirstLoc; currLoc <= thisLastLoc; currLoc++) {
495 pdelta = getSpatialParamKnob(spatialParams, ichg, currLoc);
496 pdelta = pdelta * INC; // increase temperature
497 // we used to prevent temperature from going above 1, but we no longer
498 // care about how big pdelta gets
499 setSpatialParamKnob(spatialParams, ichg, currLoc, pdelta);
500 }
501 } else if (k > numSpinUps) { // only write to history files if we have
502 // been converged for > numSpinUps steps
503 /* NOTE: I THINK WE SHOULD TECHNICALLY BE WRITING TO HIST FILE WHETHER
504 WE ACCEPT OR REJECT IF WE REJECT, SHOULD RE-WRITE OLD POINT TO HIST
505 FILE (WILL HAVE TO SAVE OLD LOGLIKELY, SIGMA, AND OUTPUTINFO) TO DO
506 THIS, REMOVE THIS ELSE IF BLOCK, AND PUT IT IN AN IF (CONVERGED && K
507 > NUMSPINUPS) BLOCK AFTER END OF ELSE (REJECTION) BLOCK */
508 for (currLoc = firstLoc; currLoc <= lastLoc; currLoc++) {
509 locIndex = currLoc - firstLoc;
510 // writeHistFile(histFiles[locIndex], loglikely[locIndex],
511 // sigma[locIndex], outputInfo[locIndex], numDataTypes, spatialParams,
512 // currLoc);
513 writeHistFileBin(histFiles[locIndex], loglikely[locIndex],
514 sigma[locIndex], outputInfo[locIndex], numDataTypes,
515 spatialParams, currLoc);
516 }
517 }
518 }
519
520 /* act on rejection */
521 else {
522 /* return to "old" parameters */
523 for (currLoc = thisFirstLoc; currLoc <= thisLastLoc; currLoc++) {
524 oldVal = pold[currLoc - thisFirstLoc];
525 setSpatialParam(spatialParams, ichg, currLoc, oldVal); // restore old
526 // value
527 }
528
529 if (converged == 0) { // we haven't converged yet - twist knob
530 // (equivalent to old pdelta)
531 for (currLoc = thisFirstLoc; currLoc <= thisLastLoc; currLoc++) {
532 pdelta = getSpatialParamKnob(spatialParams, ichg, currLoc);
533 pdelta = pdelta * DEC; // decrease temperature
534 if (pdelta < DBL_EPSILON) {
535 // don't let temperature get too small
536 pdelta = DBL_EPSILON;
537 }
538 setSpatialParamKnob(spatialParams, ichg, currLoc, pdelta);
539 } // for (currLoc)
540 } // if (converged == 0)
541 } // else (rejection)
542
543 if (k % numAtOnce == 0) {
544 // we've run for numAtOnce iterations - time to output
545
546 /**************screen output**********************/
547 fprintf(userOut, "\n\t\t\tITERATION %6ld\n", k);
548 fprintf(userOut, "\t\t\tFRACTION ACCEPTED %3.2f\n\n",
549 yes * 1.0 / numAtOnce);
550 writeChangeableParamInfo(spatialParams, loc, userOut);
551 fprintf(userOut, "\n\t\tlTOT\tnew= %9.6f\tmax= %9.6f\n", ltotnew,
552 ltotmax);
553
554 /* END SCREEN OUTPUT */
555
556 if (converged == 0) { // we haven't converged yet - check for
557 // convergence:
558 if (fabs(yes * 1.0 / numAtOnce - A_STAR) < THRESH) { // we've
559 // newly-converged
560 if (ltotmax >= bestChainLtotmax) { // this is the best chain we've
561 // found so far
562 bestChainLtotmax = ltotmax;
563 // write current info to file so we can restore it later:
564 writeChainInfo(chainInfo, ltotold, ltotmax, spatialParams, loc);
565 fprintf(userOut, "\n\nBEST START CHAIN FOUND SO FAR: WRITING CHAIN "
566 "INFO TO FILE\n\n");
567 }
568
569 if (chainNum < numChains) { // reset for the next chain
570 chainNum++; // move on to the next convergence chain
571 fprintf(userOut, "\n\nRESETTING FOR START CHAIN %d of %d\n\n",
572 chainNum, numChains);
573 reset(spatialParams, loglikely, &ltotnew, &ltotold, &ltotmax, sigma,
574 outputInfo, loc, randomStart, addFraction, userOut, likely,
575 paramWeight, model, dataTypeIndices, numDataTypes,
576 costFunction, dataTypeWeights);
577 }
578
579 else { // chainNum >= numChains: we're done running start chains,
580 // time to read best from file
581 converged = 1;
582 fprintf(userOut, "\n\nCONVERGED\n\n");
583
584 if (ltotmax < bestChainLtotmax) { // the most recent chain wasn't
585 // the best: read best from file
586 readChainInfo(chainInfo, &ltotold, &ltotmax, spatialParams, loc);
587 fprintf(userOut, "\n\nREADING BEST CHAIN INFO FROM FILE:\n\n");
588 writeChangeableParamInfo(spatialParams, loc, userOut);
589 fprintf(userOut, "\n\t\tlTOT\told= %9.6f\tmax= %9.6f\n", ltotold,
590 ltotmax);
591 }
592
593 else {
594 // the most recent chain was the best: no need to read from file
595 fprintf(userOut, "\n\nKEEPING MOST RECENT START CHAIN INFO\n\n");
596 }
