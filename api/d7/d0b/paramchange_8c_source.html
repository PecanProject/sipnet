335/* pre: **origArray contains original (unaggregated) data/model output
336 (origArray[i][j] is timestep i, data type j)
337 myNumAggSteps is number of aggregated time steps
338 myAggSteps is a vector [0..myNumAggSteps-1] containing number of time steps

340 optimization (1-indexed) Note that myNumAggSteps, myAggSteps and myStartOpt

344 and return result in theAggedData

346void computeAggedData(double **theAggedData, double **origArray,
347 int myNumAggSteps, int *myAggSteps, int myStartOpt,
348 int numDataTypes) {
349 int i, j, dataType, index;
350
351 index = myStartOpt - 1; // index into data array (myStartOpt is 1-indexed)
352 for (i = 0; i < myNumAggSteps; i++) {
353 for (dataType = 0; dataType < numDataTypes; dataType++) {
354 theAggedData[i][dataType] = 0.0; // reset sums

356
357 for (j = 0; j < myAggSteps[i]; j++) { // loop through all steps in this

360 // accumulate each sum
361 theAggedData[i][dataType] += origArray[index][dataType];
362 }
363 index++;
364 }
365 }

401// 6/9/11: For now we don't do anything with the cost function or data Type
402// weights - we just want to make it consistent with the difference function

404double aggedDifference(double *sigma, OutputInfo *outputInfo, int loc,
405 SpatialParams *spatialParams, double paramWeight,
406 void (*modelF)(double **, int, int *, SpatialParams *,
407 int),
408 int dataTypeIndices[], int numDataTypes,
409 int costFunction, double dataTypeWeights[]) {
410 int i, dataNum;
411 double *sumSquares; // one sum of squares value for each data type
412 int *n; // number of data points used in each sumSquares
413 double logLike; // the log likelihood
414
415 sumSquares = makeArray(numDataTypes);
416 n = (int *)malloc(numDataTypes * sizeof(int));
417
418 (*modelF)(model, numDataTypes / 2, dataTypeIndices, spatialParams, loc);
419 // run model, put results in model array
420 // divide numDataTypes by 2 so we have actual (unbifurcated) number of data
421 // types
422
423 // aggregate model, filling aggedModel array
424 computeAggedData(aggedModel, model, numAggSteps[loc], aggSteps[loc],
425 startOpt[loc], numDataTypes / 2);
426
427 // again, divide numDataTypes by 2 to give actual (unbifurcated) number of
428 // data types
429
430 // initialize sumSquares and count arrays
431 for (dataNum = 0; dataNum < numDataTypes; dataNum++) {
432 sumSquares[dataNum] = 0.0;
433 n[dataNum] = 0;
434 }
435
436 // compute sum of squares on unaggregated data:
437 for (i = startOpt[loc] - 1; i < endOpt[loc]; i++) {
438 for (dataNum = 0; dataNum < numDataTypes / 2; dataNum++) {
439 if (valid[loc][i][dataNum]) {
440 sumSquares[dataNum] +=
441 pow((model[i][dataNum] - data[loc][i][dataNum]), 2);
442 n[dataNum]++;
443 }
444 }
445 }
446
447 // compute sum of squares on aggregated data (note: we don't check validity
448 // here - instead use all points):
449 for (i = 0; i < numAggSteps[loc]; i++) {
450 for (dataNum = 0; dataNum < numDataTypes / 2; dataNum++) {
451 sumSquares[numDataTypes / 2 + dataNum] +=
452 pow((aggedModel[i][dataNum] - aggedData[loc][i][dataNum]), 2);
453 n[numDataTypes / 2 + dataNum]++;
454 }
455 }
456
457 // calculate aggregate info on each data type
458 for (dataNum = 0; dataNum < numDataTypes / 2; dataNum++) {
459 aggregates(outputInfo, model, loc, dataNum);
460
461 // copy aggregate info from position i to position (numDataTypes/2 + i)
462 // (same data type, but aggregated - will have same aggregate info)
463 copyOutputInfo(&(outputInfo[numDataTypes / 2 + dataNum]),
464 &(outputInfo[dataNum]));
465 }
466
467 // compute log likelihood (actually, negative log likelihood, discarding
468 // constant terms):
469 logLike = 0;
470 for (dataNum = 0; dataNum < numDataTypes; dataNum++) {
471 sigma[dataNum] = sqrt(sumSquares[dataNum] / (double)(n[dataNum]));
472 /* we can estimate sigma[i] using just sumSquares[i] because sigma[i] is
473 calculated by taking the partial derivative of likelihood with respect to
474 sigma[i], and this partial only depends on sumSquares[i]
475 */
476 logLike += n[dataNum] * log(sigma[dataNum]);
477 logLike +=
478 sumSquares[dataNum] / (2.0 * (sigma[dataNum]) * (sigma[dataNum]));
479
480 if (dataNum == numDataTypes / 2 - 1) {
481 // we just finished all unaggregated types, so apply unaggedWeight to
482 // current log likelihood
483 logLike *= unaggedWeight;
484 }
485 }
486
487 free(sumSquares);
488 free(n);
489
490 // NOTE: this is actually the NEGATIVE log likelihood, discarding constant
491 // terms
492
493 return logLike;
494}

496/* Take array of model output, compare output with measured data - for given
497 dataNum (i.e. perform comparisons between model[*][dataNum] and
498 data[loc][*][dataNum] Return (in outputInfo[dataNum]) mean error (per day),
499 mean daily-aggregated error (per day) Also fill years array with annual total
500 NEE for each year and set outputInfo.numYears, the size of the years array
501 Pre: outputInfo is already malloced, as are outputInfo[*].years arrays
502 (Can later use years array to find yearly-aggregated error and interannual
503 variability)
504*/
505void aggregates(OutputInfo *outputInfo, double **localModel, int loc,
506 int dataNum) {
507 const int DAYS_IN_YR[] = {365, 366}; /* given leapYr = 0 or 1,
508 DAYS_IN_YR[leapYr] represents the number of days in this year */
509 int i, j, day, julianDay, year;
510 int leapYr; // 1 if this is a leap year
511 int index;
512 double *modelD, *dataD; // daily aggregations
513 int yearIndex; // starting at 0 rather than being a 4-digit year
514 double sum;
515 double netNeeM, netNeeD; // net model and data nee for this aggregated step
516
517 modelD = makeArray(aggInfo[loc].numDays);
518 dataD = makeArray(aggInfo[loc].numDays);
519
520 sum = 0.0;
521 for (i = aggInfo[loc].startPt - 1; i < aggInfo[loc].endPt; i++) {
522 sum += fabs(localModel[i][dataNum] - data[loc][i][dataNum]);
523 }
524
525 outputInfo[dataNum].meanError = sum / aggInfo[loc].numDays; // mean error per
526 // day
527
528 // fill daily aggregation arrays, compute error aggregated over days:
529 sum = 0.0;
530 index = aggInfo[loc].startPt - 1;
531 for (day = 0; day < aggInfo[loc].numDays; day++) {
532 netNeeM = netNeeD = 0.0;
533 for (j = 0; j < aggInfo[loc].spd[day]; j++) {
534 netNeeM += localModel[index][dataNum];
535 netNeeD += data[loc][index][dataNum];
536 index++;
537 }
538 modelD[day] = netNeeM;
539 dataD[day] = netNeeD;
540 sum += fabs(netNeeM - netNeeD);
541 }
542
543 // set mean daily-aggregated error per day
544 outputInfo[dataNum].daysError = sum / aggInfo[loc].numDays;
545
546 // now do yearly and overall aggregation:
547
548 // first fill year arrays and find yearly aggregation:
549 day = 0;
550 julianDay = aggInfo[loc].startDay;
551 year = aggInfo[loc].startYear;
552 leapYr = (year % 4 == 0); // holds for 1900 < year < 2100
553 yearIndex = 0;
554 while (day < aggInfo[loc].numDays) {
555 netNeeM = 0.0;
556 while (julianDay <= DAYS_IN_YR[leapYr] && day < aggInfo[loc].numDays) {
557 // loop through current year
558 netNeeM += modelD[day];
559 day++;
560 julianDay++;
561 }
562 // HAPPY NEW YEAR!
563
564 outputInfo[dataNum].years[yearIndex] = netNeeM; // NEE over this whole year
565 // can later use this to find mean yearly-aggregated error per day and
566 // interannual variability
567
568 julianDay = 1;
569 year++;
570 yearIndex++;
571 leapYr = (year % 4 == 0); // holds for 1900 < year < 2100
572 }
573 outputInfo[dataNum].numYears = yearIndex; // the actual number of years
574
575 free(modelD);
576 free(dataD);
577}

619 - If spd is "#n", where n is a positive integer (e.g. "#100"), return 0,
620 don't change spd (we'll use last spd value), set count to n (this signifies
621 that we should repeat previous value n times)
622 - Otherwise, spd must be a non-negative integer (in string format): return 1,
623 set spd to the number, set count to 1
624*/
625int parseSpdValue(char *spdString, int *spd, int *count) {
626 char *errc;
627
628 if (strcmp(spdString, "-1") == 0) {
629 return -1;
630 }
631
632 if (spdString[0] == '#') { // spd is "#n" (e.g. #100)
633 // get rid of '#', convert to integer
634 *count = strtol(strtok(spdString, "#"), &errc, 0);
635 return 0;
636 }
637
638 // spd is an integer (in string format)
639 *spd = strtol(spdString, &errc, 0);
640 *count = 1;
641 return 1;
642}
