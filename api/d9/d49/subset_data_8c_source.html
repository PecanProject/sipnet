1/* subsetData: A stand-alone program
2
3 Given a set of SIPNET input files,
4 along with a start year, start day, and number of days,
5 create subsetted files (.clim, .dat, .valid, .sigma, .spd)
6 that start at the given date and extend for the given # of days.
7
8 To start at the beginning of the record, specify START_YEAR = START_DAY = -1
9 To go until the end of the record, specify NUM_DAYS = -1
10
11 Also create a file for aggregation (*.agg), using the "AGGREGATION" input
12 - Right now, set up to do no aggregation ("none") or daily aggregation
13 ("daily")
14
15 Note that you could use this program to just create desired .spd and .agg
16 files without doing any subsetting (START_YEAR = START_DAY = NUM_DAYS = -1)
17
18 Note that this only works for input files with a single location (loc=0)
19
20 Author: Bill Sacks
21 Creation date: 6/6/07
22 */
23

48/* Read next line from climate file f, which must be open for reading
49 Return the line in *line (up to maxLength characters)
50 Also return the location, year and day in their respective variables
51
52 Return value: a pointer to line if read was successful, otherwise NULL

54char *readClimLine(FILE *f, char *line, int maxLength, int *loc, int *year,
55 int *day) {
56 char *status; // return value
57
58 status = fgets(line, maxLength, f);
59 if (status != NULL) {
60 sscanf(line, "%d %d %d", loc, year, day);
61 }
62

96/* Read lines from clim, data, valid and sigma files,
97 until we reach a point on or after the date given by startYear, startDay
98 (startDay is Julian day of year)
99
100 startYear and startDay may be modified to hold the ACTUAL starting year and
101 day (which may be different from the values passed in - for example if these
102 values are -1, indicating that we should just start at the beginning)
103
104 The three files should be open for reading and rewound
105
106 At the completion of this function, the file pointers for the three files
107 will point to the start of the first line on or after the given date
108
109 This is only implemented for files containing a single location
110*/
111void gotoStart(FILE *climFin, FILE *dataFin, FILE *validFin, FILE *sigmaFin,
112 int *startYear, int *startDay) {
113 char line[MAX_LINE];
114 fpos_t fpos; // file position in climFin, so we can "unread" the last line
115 char *status;
116 int loc, year, day;
117 int err;
118
119 // Read first line:
120 err = fgetpos(climFin, &fpos);
121 if (err) {
122 printf("Error in fgetpos\n");
123 exit(1);
124 }
125 status = readClimLine(climFin, line, MAX_LINE, &loc, &year, &day);
126 checkStatus(status, "ERROR reading climate file: no valid data\n");
127
128 // Keep reading lines until we find the start day:
129 while (!pastStart(year, day, *startYear, *startDay)) {
130 status = fgets(line, MAX_LINE, dataFin); // read & ignore line from data
131 // file
132 checkStatus(status, "ERROR reading data file: Unexpected EOF or error\n");
133 status = fgets(line, MAX_LINE, validFin); // read & ignore line from valid
134 // file
135 checkStatus(status, "ERROR reading valid file: Unexpected EOF or error\n");
136 status = fgets(line, MAX_LINE, sigmaFin); // read & ignore line from valid
137 // file
138 checkStatus(status, "ERROR reading sigma file: Unexpected EOF or error\n");
139
140 // read next line from climate file:
141 err = fgetpos(climFin, &fpos);
142 if (err) {
143 printf("Error in fgetpos\n");
144 exit(1);
145 }
146 status = readClimLine(climFin, line, MAX_LINE, &loc, &year, &day);
147 checkStatus(status, "ERROR reading climate file: reached EOF before "
148 "finding desired start date\n");
149 } // while !pastStart
150
151 // set actual starting year and day:
152 *startYear = year;
153 *startDay = day;
154
155 // now we've read one extra line from climate file: have to unread it:
156 err = fsetpos(climFin, &fpos);
157 if (err) {
158 printf("Error in fsetpos\n");
159 exit(1);
160 }
161}

163/* Starting at current positions in climate, data, and valid input files,
164 copy from input files to output files until we've read numDays days
165 (numDays <= 0 indicates that we should read until end of file).
166 Also create spdFout (steps per day; including leading startYear, startDay)
167 and aggFout (# steps per aggregation time step)
168 - Aggregation time step determined by aggType (must be one of the
169 AGGREGATION_TYPES)
170 - Right now, only implemented for NO_AGGREGATION (1 step per aggregation
171 step) and DAILY_AGGREGATION
172
173 All output files should be open for writing
174
175 Assumes that there is at least one time step per day in the input files
176 (i.e. the interval between two time steps is not greater than a day)
177 */
178void copySubset(FILE *climFin, FILE *dataFin, FILE *validFin, FILE *sigmaFin,
179 FILE *climFout, FILE *dataFout, FILE *validFout,
180 FILE *sigmaFout, FILE *spdFout, FILE *aggFout, int aggType,
181 int startYear, int startDay, int numDays) {
182 char line[MAX_LINE];
183 int loc, year, day;
184 int currYear, currDay;
185 int ndays; // number of days we've read so far (including partial days)
186 int stepsThisDay;
187 int done;
188 char *status;
189
190 fprintf(spdFout, "%d %d ", startYear, startDay);
191
192 // read next line from climate file:
193 status = readClimLine(climFin, line, MAX_LINE, &loc, &year, &day);
194 checkStatus(status,
195 "ERROR reading climate file: no data found after start date\n");
196
197 ndays = 1;
198 stepsThisDay = 1;
199 done = 0;
200 while (!done) {
201 currYear = year;
202 currDay = day;
203
204 fputs(line, climFout);
205
206 // read & copy line from data file
207 status = fgets(line, MAX_LINE, dataFin);
208 checkStatus(status, "ERROR reading data file: Unexpected EOF or error\n");
209 fputs(line, dataFout);
210
211 // read & copy line from valid file
212 status = fgets(line, MAX_LINE, validFin);
213 checkStatus(status, "ERROR reading valid file: Unexpected EOF or error\n");
214 fputs(line, validFout);
215
216 // read & copy line from sigma file
217 status = fgets(line, MAX_LINE, sigmaFin);
218 checkStatus(status, "ERROR reading sigma file: Unexpected EOF or error\n");
219 fputs(line, sigmaFout);
220
221 // read next line from climate file:
222 // note that we copy it to output file at TOP of loop
223 status = readClimLine(climFin, line, MAX_LINE, &loc, &year, &day);
224
225 if (status == NULL) { // EOF or error
226 // do final aggregation and spd file output:
227 switch (aggType) {
228 case NO_AGGREGATION:
229 fprintf(aggFout, "1 -1\n");
230 break;
231 case DAILY_AGGREGATION:
232 fprintf(aggFout, "%d -1\n", stepsThisDay);
233 break;
234 default:
235 printf("Invalid aggregation type: %d\n", aggType);
236 exit(1);
237 }
238 fprintf(spdFout, "%d -1\n", stepsThisDay);
239
240 if ((numDays <= 0) || (ndays == numDays)) {
241 // If we have hit end of file or read enough days
242 done = 1;
243 } else { // we haven't read enough days
244 printf("ERROR reading climate file: reached EOF or error after reading "
245 "only %d of %d days\n",
246 ndays, numDays);
247 printf("Try choosing an earlier start date, or a smaller number of "
248 "days\n");
249 exit(1);
250 }
251 }
252
253 else { // not EOF or error
254 if ((year - currYear > 1) ||
255 ((year - currYear == 1) && (currDay < 365 || day > 1)) ||
256 ((year == currYear) && (day - currDay) > 1)) {
257 // if we (1) advanced by more than 1 year, or (2) advanced by one year,
258 // but we either (a) weren't on day 365 (or 366), or (b) now aren't on
259 // day 1, or (3) advanced by more than one day (note: we don't
260 // explicitly check for climate going backwards: we assume this doesn't
261 // happen)
262 printf("ERROR: subsetData only implemented for input files with at "
263 "least one step per day\n");
264 exit(1);
265 }
266
267 // possibly do aggregation file output:
268 switch (aggType) {
269 case NO_AGGREGATION:
270 fprintf(aggFout, "1 ");
271 break;
272 case DAILY_AGGREGATION:
273 if (day != currDay) {
274 fprintf(aggFout, "%d ", stepsThisDay);
275 }
276 break;
277 default:
278 printf("Invalid aggregation type: %d\n", aggType);
279 exit(1);
280 }
281
282 if (day != currDay) {
283 fprintf(spdFout, "%d ", stepsThisDay);
284 ndays++;
285 stepsThisDay = 1;
286 // we set new currDay and currYear at TOP of loop
287
288 if ((numDays > 0) && (ndays > numDays)) { // we've read numDays days
289 done = 1;
290
291 /* output end-of-line terminators to aggregation and spd files
292 Note that, because the only possible aggregations are
293 NO_AGGREGATION and DAILY_AGGREGATION, we will definitely have
294 already output the last agg. step to aggregation file. If we add
295 another aggregation type (e.g. annual), we may need to output the
296 last value here before terminating the line
297 */
298 fprintf(aggFout, "-1\n");
299 fprintf(spdFout, "-1\n");
300 }
301 } else { // day == currDay
302 stepsThisDay++;
303 }
304 } // else (not EOF or error)
305 } // while !done
306} // copySubset

308// copy contents of <inFilename> to <outFilename>
309void copyFile(char *inFilename, char *outFilename) {
310 FILE *in, *out;
311 char c;
312
313 in = openFile(inFilename, "r");
314 out = openFile(outFilename, "w");
315
316 while ((c = fgetc(in)) != EOF) {
317 fputc(c, out);
318 }
319
320 fclose(out);
321 fclose(in);

352 NamelistInputs *namelistInputs;

354 char outFilename[MAX_FILENAME]; // base output file name
355 int startYear, startDay, numDays;
356 char aggTypeString[MAX_NAME];

359 *validFout, *sigmaFout, *spdFout, *aggFout;
360
361 // clang-format off
362 // setup namelist input:
363 namelistInputs = newNamelistInputs();
364 addNamelistInputItem(namelistInputs, "INPUT_FILENAME", STRING_TYPE, inFilename, MAX_FILENAME);
365 addNamelistInputItem(namelistInputs, "OUTPUT_FILENAME", STRING_TYPE, outFilename, MAX_FILENAME);
366 addNamelistInputItem(namelistInputs, "START_YEAR", INT_TYPE, &startYear, 0);
367 addNamelistInputItem(namelistInputs, "START_DAY", INT_TYPE, &startDay, 0);
368 addNamelistInputItem(namelistInputs, "NUM_DAYS", INT_TYPE, &numDays, 0);
369 addNamelistInputItem(namelistInputs, "AGGREGATION", STRING_TYPE, &aggTypeString, MAX_NAME);
370 // clang-format on
371
372 // read from namelist file:
373 readNamelistInputs(namelistInputs, NAMELIST_FILE);
374
375 // and make sure we read everything:
376 dieIfNotRead(namelistInputs, "INPUT_FILENAME");
377 dieIfNotRead(namelistInputs, "OUTPUT_FILENAME");
378 dieIfNotRead(namelistInputs, "START_YEAR");
379 dieIfNotRead(namelistInputs, "START_DAY");
380 dieIfNotRead(namelistInputs, "NUM_DAYS");
381 dieIfNotRead(namelistInputs, "AGGREGATION");
382
383 if (strcmp(aggTypeString, "") == 0) {
384 aggType = NO_AGGREGATION;
385 } else if (strcmpIgnoreCase(aggTypeString, "daily") == 0) {
386 aggType = DAILY_AGGREGATION;
387 } else {
388 printf("Unrecognized aggregation: '%s'\n", aggTypeString);
389 exit(1);
390 }
391
392 // open files:
393 // input files:
394 climFin = openFileExt(inFilename, "clim", "r");
395 dataFin = openFileExt(inFilename, "dat", "r");
396 validFin = openFileExt(inFilename, "valid", "r");
397 sigmaFin = openFileExt(inFilename, "sigma", "r");
398
399 // output files:
400 climFout = openFileExt(outFilename, "clim", "w");
401 dataFout = openFileExt(outFilename, "dat", "w");
402 validFout = openFileExt(outFilename, "valid", "w");
403 sigmaFout = openFileExt(outFilename, "sigma", "w");
404 spdFout = openFileExt(outFilename, "spd", "w");
405 aggFout = openFileExt(outFilename, "agg", "w");
406
407 // do the work:
408 checkLocations(climFin); // make sure the climate file only contains one
409 // location
410 gotoStart(climFin, dataFin, validFin, sigmaFin, &startYear, &startDay);
411 copySubset(climFin, dataFin, validFin, sigmaFin, climFout, dataFout,
412 validFout, sigmaFout, spdFout, aggFout, aggType, startYear,
413 startDay, numDays);
414 copyOtherFiles(inFilename, outFilename);
415
416 // close files:
417 fclose(climFin);
418 fclose(dataFin);
419 fclose(validFin);
420 fclose(sigmaFin);
421 fclose(climFout);
422 fclose(dataFout);
423 fclose(validFout);
424 fclose(sigmaFout);
425 fclose(spdFout);
426 fclose(aggFout);
427
428 return 0;
429}
