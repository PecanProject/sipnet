22void checkRuntype(const char *runType) {
23 if (strcasecmp(runType, "standard") != 0) {

33void readInputFile(void) {
34 // First, make sure the filename is valid
35 validateFilename();
36 logInfo("Reading config from file %s\n", ctx.inputFile);
37
38 const char *SEPARATORS = " \t=:"; // characters that can separate names from
39 // values in input file
40 const char *COMMENT_CHARS = "!"; // comment characters (ignore everything
41 // after this on a line)
42 const char *EMPTY_STRING = "none"; // if this is the value of a string type,
43 // we take it to mean the empty string
44
45 FILE *infile;
46
47 char line[1024];
48 char allSeparators[16];
49 char *inputName; // name of input item just read in
50 char *inputValue; // value of input item just read in, as a string
51 char *errc = "";
52 int isComment;
53
54 strcpy(allSeparators, SEPARATORS);
55 strcat(allSeparators, "\n\r");
56 infile = openFile(ctx.inputFile, "r");
57
58 while (fgets(line, sizeof(line), infile) != NULL) { // while not EOF or error
59 // remove trailing comments:
60 isComment = stripComment(line, COMMENT_CHARS);
61
62 if (!isComment) { // if this isn't just a comment line or blank line
63 // tokenize line:
64 inputName = strtok(line, SEPARATORS);
65 inputValue = strtok(NULL, allSeparators);
66
67 // Handle RUNTYPE as an obsolete param; if it isn't set, consider it to be
68 // set to "standard"; and make sure it is that if set
69 if (strcasecmp(inputName, "runtype") == 0) {
70 checkRuntype(inputValue);
71 continue;
72 }
73
74 // Find the metadata so we know what to do with this param
75 struct context_metadata *ctx_meta = getContextMetadata(inputName);
76 if (ctx_meta == NULL) {
77 logWarning("ignoring input file parameter %s\n", inputName);
78 continue;
79 }
80
81 if (inputValue == NULL) {
82 logError("Error in input file: No value given for input item %s\n",
83 inputName);
84 logError("Please fix %s and re-run\n", ctx.inputFile);
85 exit(EXIT_CODE_BAD_PARAMETER_VALUE);
86 }
87
88 switch (ctx_meta->type) {
89 case CTX_INT: {
90 int intVal = strtol(inputValue, &errc, 0); // NOLINT
91 if (strlen(errc) > 0) { // invalid character(s) in input string
92 logError("ERROR in input file: Invalid value for %s: %s\n",
93 inputName, inputValue);
94 logError("Please fix %s and re-run\n", ctx.inputFile);
95 exit(EXIT_CODE_BAD_PARAMETER_VALUE);
96 }
97 updateIntContext(inputName, intVal, CTX_CONTEXT_FILE);
98 } break;
99 case CTX_CHAR: {
100 if (strcmp(inputValue, EMPTY_STRING) == 0) {
101 // if the value is specified as the value which signifies the
102 // empty string
103 updateCharContext(inputName, "", CTX_CONTEXT_FILE);
104 } else {
105 if (strlen(inputValue) >= CONTEXT_CHAR_MAXLEN) {
106 logError("ERROR in input file: value '%s' exceeds maximum "
107 "length for %s (%d)\n",
108 inputValue, inputName, CONTEXT_CHAR_MAXLEN);
109 logError(
110 "Please fix %s, or change the maximum length, and re-run\n",
111 ctx.inputFile);
112 exit(EXIT_CODE_BAD_PARAMETER_VALUE);
113 }
114 updateCharContext(inputName, inputValue, CTX_CONTEXT_FILE);
115 }
116 } break;
117 default:
118 logError("ERROR in readInputFile: Unrecognized type for %s: %d\n",
119 inputName, ctx_meta->type);
120 exit(EXIT_CODE_INTERNAL_ERROR);
121 }
122 } // if (!isComment)
123 } // while not EOF or error
124
125 fclose(infile);
126}

128int main(int argc, char *argv[]) {
129
130 FILE *out, *outConfig;
131
132 ModelParams *modelParams; // the parameters used in the model
133 OutputItems *outputItems; // structure to hold information for output to

139 char paramFile[FILENAME_MAXLEN], climFile[FILENAME_MAXLEN];
140
141 // 1. Initialize Context with default values
142 initContext();
143 // Also need to verify the CLI name map, now that the context has been init'd
144 checkCLINameMap();
145
146 // 2. Parse command line args
147 parseCommandLineArgs(argc, argv);
148
149 // 3. Read input config file
150 // Note: command-line args have precedence
151 // read from input file
152 readInputFile();
153
154 // 4. Run some checks
155 validateContext();
156
157 // 5. Set calculated parameters
158 strcpy(paramFile, ctx.fileName);
159 strcat(paramFile, ".param");
160 updateCharContext("paramFile", paramFile, CTX_CALCULATED);
161 strcpy(climFile, ctx.fileName);
162 strcat(climFile, ".clim");
163 updateCharContext("climFile", climFile, CTX_CALCULATED);
164 if (ctx.doMainOutput) {
165 strcpy(outFile, ctx.fileName);
166 strcat(outFile, ".out");
167 updateCharContext("outFile", outFile, CTX_CALCULATED);
168 out = openFile(outFile, "w");
169 } else {
170 out = NULL;
171 }
172
173 // Lastly - do after all other config processing
174 if (ctx.dumpConfig) {
175 strcpy(outConfigFile, ctx.fileName);
176 strcat(outConfigFile, ".config");
177 updateCharContext("outConfigFile", outConfigFile, CTX_CALCULATED);
178 outConfig = openFile(outConfigFile, "w");
179 printConfig(outConfig);
180 fclose(outConfig);
181 } else {
182 outConfig = NULL;
183 }
184
185 // 6. Initialize model, events, outputItems
186 initModel(&modelParams, paramFile, climFile);
187
188 if (ctx.events) {
189 initEvents(EVENT_IN_FILE, ctx.printHeader);
190 }
191
192 if (ctx.doSingleOutputs) {
193 outputItems = newOutputItems(ctx.fileName, ' ');
194 setupOutputItems(outputItems);
195 } else {
196 outputItems = NULL;
197 }
198
199 // 7. Do the run!
200 runModelOutput(out, outputItems, ctx.printHeader);
201
202 // 8. Cleanup
203 if (ctx.doMainOutput) {
204 fclose(out);
205 }
206
207 cleanupModel();
208 if (outputItems != NULL) {
209 deleteOutputItems(outputItems);
210 }
211
212 deleteModelParams(modelParams);
213
214 return EXIT_CODE_SUCCESS;
215}
