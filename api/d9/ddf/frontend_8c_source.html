33void checkRuntype(NamelistInputs *namelist, const char *inputFile) {
34 NamelistInputItem *inputItem = locateNamelistInputItem(namelist, "RUNTYPE");
35 if (inputItem == NULL) {
36 // Well, something really broke
37 printf("ERROR: did not find NameListInputItem RUNTYPE\n");
38 exit(EXIT_CODE_INTERNAL_ERROR);
39 }
40 if (!(inputItem->wasRead)) {
41 strcpy((char *)inputItem->ptr, "standard");
42 return;
43 }
44 if (strcmpIgnoreCase((char *)inputItem->ptr, "standard") != 0) {
45 // Make sure this is not an old config with a different RUNTYPE set
46 printf("SIPNET only supports the standard runtype mode; other options are "
47 "obsolete and were last supported in v1.3.0\n");
48 printf("Please fix %s and re-run\n", inputFile);
49 exit(EXIT_CODE_BAD_PARAMETER_VALUE);
50 }
51}

60int main(int argc, char *argv[]) {

62 NamelistInputs *namelistInputs;
63
64 FILE *out;
65 int option; // reading in optional arguments

67 SpatialParams *spatialParams; // the parameters used in the model (possibly

75 int doMainOutput = DO_MAIN_OUTPUT; // do we do main outputting of all

77 int doSingleOutputs = DO_SINGLE_OUTPUTS; // do we do extra outputting of
78 // single-variable files?
79 int loc = LOC; // location to run at (set through optional -l argument)
80 int numLocs; // read in initModel
81 int *steps; // number of time steps in each location
82
83 int printHeader = HEADER;

85 char fileName[FILE_MAXNAME];
86 char outFile[FILE_MAXNAME + 24];
87 char paramFile[FILE_MAXNAME + 24], climFile[FILE_MAXNAME + 24];
88
89 // get command-line arguments:
90 while ((option = getopt(argc, argv, "hi:")) != -1) {
91 // we have another optional argument
92 switch (option) {
93 case 'h':
94 usage(argv[0]);
95 exit(1);
96 case 'i':
97 if (strlen(optarg) >= INPUT_MAXNAME) {
98 printf("ERROR: input filename %s exceeds maximum length of %d\n",
99 optarg, INPUT_MAXNAME);
100 printf("Either change the name or increase INPUT_MAXNAME in "
101 "frontend.c\n");
102 exit(1);
103 }
104 strcpy(inputFile, optarg);
105 break;
106 default:
107 usage(argv[0]);
108 exit(1);
109 }
110 }
111
112 // clang-format off
113 // setup namelist input:
114 namelistInputs = newNamelistInputs();
115 addNamelistInputItem(namelistInputs, "RUNTYPE", STRING_TYPE, runtype, RUNTYPE_MAXNAME);
116 addNamelistInputItem(namelistInputs, "FILENAME", STRING_TYPE, fileName, FILE_MAXNAME);
117 addNamelistInputItem(namelistInputs, "LOCATION", INT_TYPE, &loc,0);
118 addNamelistInputItem(namelistInputs, "DO_MAIN_OUTPUT", INT_TYPE, &doMainOutput, 0);
119 addNamelistInputItem(namelistInputs, "DO_SINGLE_OUTPUTS", INT_TYPE, &doSingleOutputs, 0);
120 addNamelistInputItem(namelistInputs, "PRINT_HEADER", INT_TYPE, &printHeader,0);
121 // clang-format on
122
123 // read from input file:
124 readNamelistInputs(namelistInputs, inputFile);
125
126 // Make sure FILENAME is set; everything else is optional (not necessary or
127 // has a default)
128 dieIfNotSet(namelistInputs, "FILENAME");
129
130 // Handle RUNTYPE as an obsolete param; if it isn't set, consider it to be
131 // set to "standard"; and make sure it is that if set
132 checkRuntype(namelistInputs, inputFile);
133
134 strcpy(paramFile, fileName);
135 strcat(paramFile, ".param");
136 strcpy(climFile, fileName);
137 strcat(climFile, ".clim");
138 numLocs = initModel(&spatialParams, &steps, paramFile, climFile);
139
140#if EVENT_HANDLER
141 initEvents(EVENT_IN_FILE, numLocs, printHeader);
142#endif
143
144 if (doSingleOutputs) {
145 outputItems = newOutputItems(fileName, ' ');
146 setupOutputItems(outputItems);
147 } else {
148 outputItems = NULL;
149 }
150
151 // Do the run!
152 if (doMainOutput) {
153 strcpy(outFile, fileName);
154 strcat(outFile, ".out");
155 out = openFile(outFile, "w");
156 } else {
157 out = NULL;
158 }
159
160 runModelOutput(out, outputItems, printHeader, spatialParams, loc);
161
162 if (doMainOutput) {
163 fclose(out);
164 }
165
166 cleanupModel(numLocs);
167 deleteSpatialParams(spatialParams);
168 if (outputItems != NULL) {
169 deleteOutputItems(outputItems);
170 }
171 free(steps);
172
173 return 0;
174}
