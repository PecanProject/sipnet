22// Allocate space for a new namelistInputItem structure, return a pointer to it
23// Name, type, ptr and maxlen will have the given values; nextItem will be null
24// Note that if type is anything other than STRING_TYPE, maxlen is unused (but
25// still must be supplied)
26NamelistInputItem *newNamelistInputItem(char *name, int type, void *ptr,
27 int maxlen) {
28 NamelistInputItem *namelistInputItem;
29
30 if (strlen(name) >= NAMELIST_INPUT_MAXNAME) {
31 printf("ERROR in newNamelistInputItem: parameter name '%s' exceeds maximum "
32 "length of %d\n",
33 name, NAMELIST_INPUT_MAXNAME);
34 exit(1);
35 }
36
37 namelistInputItem = (NamelistInputItem *)malloc(sizeof(NamelistInputItem));
38
39 strcpy(namelistInputItem->name, name);
40 namelistInputItem->type = type;
41 namelistInputItem->ptr = ptr;
42 namelistInputItem->maxlen = maxlen;
43 namelistInputItem->wasRead = 0;
44
45 namelistInputItem->nextItem = NULL;
46
47 return namelistInputItem;
48}

106/* Read all namelist inputs from file given by fileName, put them in the
107 locations pointed to by each inputs' ptr variable
108
109 Structure of input file: Each line contains:
110 name value
111 where the following are valid separators: space, tab, =, :
112
113 A value of 'none' for a string type is translated into the empty string
114
115 The order of input items in the input file does not matter
116
117 ! is a comment character: anything after a ! on a line is ignored
118 */
119void readNamelistInputs(NamelistInputs *namelistInputs, const char *fileName) {
120 const char *SEPARATORS = " \t=:"; // characters that can separate names from
121 // values in input file
122 const char *COMMENT_CHARS = "!"; // comment characters (ignore everything
123 // after this on a line)
124 const char *EMPTY_STRING = "none"; // if this is the value of a string type,
125 // we take it to mean the empty string
126
127 FILE *infile;
128
129 char line[1024];
130 char allSeparators[16];
131 char *inputName; // name of input item just read in
132 char *inputValue; // value of input item just read in, as a string
133 char *errc = "";
134 int isComment;
135 NamelistInputItem *namelistInputItem;
136
137 strcpy(allSeparators, SEPARATORS);
138 strcat(allSeparators, "\n\r");
139 infile = openFile(fileName, "r");
140
141 while (fgets(line, sizeof(line), infile) != NULL) { // while not EOF or error
142 // remove trailing comments:
143 isComment = stripComment(line, COMMENT_CHARS);
144
145 if (!isComment) { // if this isn't just a comment line or blank line
146 // tokenize line:
147 inputName = strtok(line, SEPARATORS); // make inputName point to first
148 // token
149 inputValue = strtok(NULL, allSeparators); // make inputValue point to
150 // next token (e.g. after the
151 // '=')
152
153 // printf("%s: <%s>\n", inputName, inputValue);
154
155 namelistInputItem = locateNamelistInputItem(namelistInputs, inputName);
156 if (namelistInputItem == NULL) {
157 printf("ERROR in readNamelistInputs: Read unexpected input item: %s\n",
158 inputName);
159 printf("Please fix %s and re-run\n", fileName);
160 exit(1);
161 }
162 // otherwise, we have found the item with this name
163
164 if (inputValue == NULL) {
165 printf(
166 "ERROR in readNamelistInputs: No value given for input item %s\n",
167 inputName);
168 printf("Please fix %s and re-run\n", fileName);
169 exit(1);
170 }
171
172 switch (namelistInputItem->type) {
173 case INT_TYPE:
174 *(int *)(namelistInputItem->ptr) = strtol(inputValue, &errc, 0);
175 if (strlen(errc) > 0) { // invalid character(s) in input string
176 printf("ERROR in readNamelistInputs: Invalid value for %s: %s\n",
177 inputName, inputValue);
178 printf("Please fix %s and re-run\n", fileName);
179 // printf("%d <%s>\n", (int)strlen(errc), errc);
180 exit(1);
181 }
182 break;
183 case LONG_TYPE:
184 *(long *)(namelistInputItem->ptr) = strtol(inputValue, &errc, 0);
185 if (strlen(errc) > 0) { // invalid character(s) in input string
186 printf("ERROR in readNamelistInputs: Invalid value for %s: %s\n",
187 inputName, inputValue);
188 printf("Please fix %s and re-run\n", fileName);
189 // printf("%d <%s>\n", (int)strlen(errc), errc);
190 exit(1);
191 }
192 break;
193 case DOUBLE_TYPE:
194 *(double *)(namelistInputItem->ptr) = strtod(inputValue, &errc);
195 if (strlen(errc) > 0) { // invalid character(s) in input string
196 printf("ERROR in readNamelistInputs: Invalid value for %s: %s\n",
197 inputName, inputValue);
198 printf("Please fix %s and re-run\n", fileName);
199 // printf("%d <%s>\n", (int)strlen(errc), errc);
200 exit(1);
201 }
202 break;
203 case STRING_TYPE:
204 if (strcmp(inputValue, EMPTY_STRING) == 0) {
205 // if the value is specified as the value which signifies the empty
206 // string
207 strcpy((char *)(namelistInputItem->ptr), "");
208 } else {
209 if (strlen(inputValue) >= namelistInputItem->maxlen) {
210 printf("ERROR in readNamelistInputs: value '%s' exceeds maximum "
211 "length for %s (%d)\n",
212 inputValue, inputName, namelistInputItem->maxlen);
213 printf(
214 "Please fix %s, or change the maximum length, and re-run\n",
215 fileName);
216 exit(1);
217 }
218 strcpy((char *)(namelistInputItem->ptr), inputValue);
219 }
220 break;
221 default:
222 printf("ERROR in readNamelistInputs: Unrecognized type for %s: %d\n",
223 inputName, namelistInputItem->type);
224 exit(1);
225 }
226
227 namelistInputItem->wasRead = 1;
228
229 } // if (!isComment)
230 } // while not EOF or error
231
232 fclose(infile);
233}

253/* For string types only:
254 Check to make sure the given item:
255 (1) was read from the input file
256 (2) was set to a non-empty value (i.e. wasn't set to 'none')
257 Kill the program and print an error message if either of these conditions are
258 not met
259 */
260void dieIfNotSet(NamelistInputs *namelistInputs, char *name) {
261 NamelistInputItem *namelistInputItem;
262
263 namelistInputItem = locateNamelistInputItem(namelistInputs, name);
264 if (namelistInputItem == NULL) {
265 printf("ERROR in dieIfNotSet: Can't find input item '%s'\n", name);
266 exit(1);
267 }
268 if (namelistInputItem->type != STRING_TYPE) {
269 printf("ERROR in dieIfNotSet: %s not of string_type\n", name);
270 printf(" Use dieIfNotRead instead\n");
271 exit(1);
272 }
273
274 dieIfNotRead(namelistInputs, name);
275 if (strcmp((char *)namelistInputItem->ptr, "") == 0) {
276 printf("ERROR: '%s' must not be set to 'none' or the empty string\n", name);
277 exit(1);
278 }
279}
