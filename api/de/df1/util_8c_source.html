19// our own openFile method, which exits gracefully if there's an error
20FILE *openFile(const char *name, const char *mode) {
21 FILE *f;
22
23 if ((f = fopen(name, mode)) == NULL) {
24 const char *mode_word =
25 (!strcmp(mode, "r") || !strcmp(mode, "rb")) ? "reading" : "writing";
26 fprintf(stderr, "Error %s '%s': %s\n", mode_word, name, strerror(errno));
27 exit(EXIT_CODE_FILE_OPEN_OR_READ_ERROR);
28 }
29
30 return f;
31}

33// If line contains any character in the string commentChars,
34// strip the comment off the line (i.e. replace first occurrence of
35// commentChars with '\0')
36// Return 1 if line contains only a comment (or only blanks), 0 otherwise
37int stripComment(char *line, const char *commentChars) {
38 char *commentCharLoc;
39 int lenTrim;
40
41 // strip trailing comment:
42 commentCharLoc = strpbrk(line, commentChars);
43 if (commentCharLoc != NULL) {
44 commentCharLoc[0] = '\0';
45 }
46
47 // determine length without any leading blanks
48 lenTrim = strlen(line) - strspn(line, " \t\n\r");
49 return (lenTrim == 0);
50}

52// count number of fields in a string separated by delimiter 'sep'
53int countFields(const char *line, const char *sep) {
54 // strtok modifies string, so we need a copy
55 char lineCopy[256];
56 strcpy(lineCopy, line);
57 int numParams = 0;
58 char *par = strtok(lineCopy, sep);
59 while (par != NULL) {
60 ++numParams;
61 par = strtok(NULL, sep);
62 }
63 return numParams;
64}
