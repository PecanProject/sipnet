88/* Dynamically allocate space for a 2-d array of doubles of given size,
89 return pointer to start of array
90
91 Ensures that array elements are continuous to increase efficiency
92
93 To free array, first free(arr[0]), then free(arr)
94*/
95double **make2DArray(int nrows, int ncols) {
96 double **arr;
97 int i;
98
99 arr = (double **)malloc(nrows * sizeof(double *));
100
101 // assure that all space is allocated continuously:
102 arr[0] = (double *)malloc(nrows * ncols * sizeof(double));
103 for (i = 1; i < nrows; i++) {
104 arr[i] = arr[0] + i * ncols;
105 }
106
107 return arr;
108}

154// do an strcmp on s1 and s2, ignoring case
155// (convert both to lower case before comparing)
156// return value is the same as for strcmp
157int strcmpIgnoreCase(const char *s1, const char *s2) {
158 char *s1Lower;
159 char *s2Lower;
160 int i;
161 int result;
162
163 // allocate space for copies
164 // note that we need one more than strlen to allow room for termination
165 // character
166 s1Lower = (char *)malloc((strlen(s1) + 1) * sizeof(char));
167 s2Lower = (char *)malloc((strlen(s2) + 1) * sizeof(char));
168
169 for (i = 0; i <= strlen(s1); i++) {
170 s1Lower[i] = tolower(s1[i]);
171 }
172 for (i = 0; i <= strlen(s2); i++) {
173 s2Lower[i] = tolower(s2[i]);
174 }
175
176 result = strcmp(s1Lower, s2Lower);
177
178 free(s1Lower);
179 free(s2Lower);
180
181 return result;
182}

184// If line contains any character in the string commentChars,
185// strip the comment off the line (i.e. replace first occurrence of
186// commentChars with '\0')
187// Return 1 if line contains only a comment (or only blanks), 0 otherwise
188int stripComment(char *line, const char *commentChars) {
189 char *commentCharLoc;
190 int lenTrim;
191
192 // strip trailing comment:
193 commentCharLoc = strpbrk(line, commentChars);
194 if (commentCharLoc != NULL) {
195 commentCharLoc[0] = '\0';
196 }
197
198 // determine length without any leading blanks
199 lenTrim = strlen(line) - strspn(line, " \t\n\r");
200 return (lenTrim == 0);
201}
