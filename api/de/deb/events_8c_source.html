23EventNode *createEventNode(int year, int day, int eventType,
24 const char *eventParamsStr) {
25 EventNode *event = (EventNode *)malloc(sizeof(EventNode));
26 event->year = year;
27 event->day = day;
28 event->type = eventType;
29
30 switch (eventType) {
31 case HARVEST: {
32 double fracRA, fracRB, fracTA, fracTB;
33 HarvestParams *params = (HarvestParams *)malloc(sizeof(HarvestParams));
34 int numRead =
35 sscanf(eventParamsStr, // NOLINT
36 "%lf %lf %lf %lf", &fracRA, &fracRB, &fracTA, &fracTB);
37 if (numRead != NUM_HARVEST_PARAMS) {
38 printf("Error parsing Harvest params for year %d day %d\n", year, day);
39 exit(EXIT_CODE_INPUT_FILE_ERROR);
40 }
41 // Validate the params
42 if ((fracRA + fracTA > 1) || (fracRB + fracTB > 1)) {
43 printf("Invalid harvest event for year %d day %d; above and below must "
44 "each add to 1 or less",
45 year, day);
46 exit(EXIT_CODE_BAD_PARAMETER_VALUE);
47 }
48 params->fractionRemovedAbove = fracRA;
49 params->fractionRemovedBelow = fracRB;
50 params->fractionTransferredAbove = fracTA;
51 params->fractionTransferredBelow = fracTB;
52 event->eventParams = params;
53 } break;
54 case IRRIGATION: {
55 double amountAdded;
56 int method;
57 IrrigationParams *params =
58 (IrrigationParams *)malloc(sizeof(IrrigationParams));
59 int numRead = sscanf(eventParamsStr, // NOLINT
60 "%lf %d", &amountAdded, &method);
61 if (numRead != NUM_IRRIGATION_PARAMS) {
62 printf("Error parsing Irrigation params for year %d day %d\n", year,
63 day);
64 exit(EXIT_CODE_INPUT_FILE_ERROR);
65 }
66 params->amountAdded = amountAdded;
67 params->method = method;
68 event->eventParams = params;
69 } break;
70 case FERTILIZATION: {
71 // If/when we try two N pools, enable the additional nh4_no3_frac param
72 // (likely via compiler switch)
73 double orgN, orgC, minN;
74 // double nh4_no3_frac;
75 FertilizationParams *params =
76 (FertilizationParams *)malloc(sizeof(FertilizationParams));
77 int numRead = sscanf(eventParamsStr, // NOLINT
78 "%lf %lf %lf", &orgN, &orgC, &minN);
79 if (numRead != NUM_FERTILIZATION_PARAMS) {
80 printf("Error parsing Fertilization params for year %d day %d\n", year,
81 day);
82 exit(EXIT_CODE_INPUT_FILE_ERROR);
83 }
84 // scanf(eventParamsStr, "%lf %lf %lf %lf", &org_N, &org_C, &min_N,
85 // &nh4_no3_frac);
86 params->orgN = orgN;
87 params->orgC = orgC;
88 params->minN = minN;
89 // params->nh4_no3_frac = nh4_nos_frac;
90 event->eventParams = params;
91 } break;
92 case PLANTING: {
93 double leafC, woodC, fineRootC, coarseRootC;
94 PlantingParams *params = (PlantingParams *)malloc(sizeof(PlantingParams));
95 int numRead =
96 sscanf(eventParamsStr, // NOLINT
97 "%lf %lf %lf %lf", &leafC, &woodC, &fineRootC, &coarseRootC);
98 if (numRead != NUM_PLANTING_PARAMS) {
99 printf("Error parsing Planting params for year %d day %d\n", year, day);
100 exit(EXIT_CODE_INPUT_FILE_ERROR);
101 }
102 params->leafC = leafC;
103 params->woodC = woodC;
104 params->fineRootC = fineRootC;
105 params->coarseRootC = coarseRootC;
106 event->eventParams = params;
107 } break;
108 case TILLAGE: {
109 double fracLT, somDM, litterDM;
110 TillageParams *params = (TillageParams *)malloc(sizeof(TillageParams));
111 int numRead = sscanf(eventParamsStr, // NOLINT
112 "%lf %lf %lf", &fracLT, &somDM, &litterDM);
113 if (numRead != NUM_TILLAGE_PARAMS) {
114 printf("Error parsing Tillage params for year %d day %d\n", year, day);
115 exit(EXIT_CODE_INPUT_FILE_ERROR);
116 }
117 params->fractionLitterTransferred = fracLT;
118 params->somDecompModifier = somDM;
119 params->litterDecompModifier = litterDM;
120 event->eventParams = params;
121 } break;
122 default:
123 // Unknown type, error and exit
124 printf("Error reading event file: unknown event type %d\n", eventType);
125 exit(1);
126 }
127
128 event->nextEvent = NULL;
129 return event;
130}

132event_type_t getEventType(const char *eventTypeStr) {
133 if (strcmp(eventTypeStr, "irrig") == 0) {
134 return IRRIGATION;
135 } else if (strcmp(eventTypeStr, "fert") == 0) {
136 return FERTILIZATION;
137 } else if (strcmp(eventTypeStr, "plant") == 0) {
138 return PLANTING;
139 } else if (strcmp(eventTypeStr, "till") == 0) {
140 return TILLAGE;
141 } else if (strcmp(eventTypeStr, "harv") == 0) {
142 return HARVEST;
143 }
144 return UNKNOWN_EVENT;
145}

147EventNode *readEventData(char *eventFile) {
148 int year, day, eventType;
149 int currYear, currDay;
150 int EVENT_LINE_SIZE = 1024;
151 int numBytes;
152 char *eventParamsStr;
153 char eventTypeStr[20];
154 char line[EVENT_LINE_SIZE];
155 EventNode *curr, *next;
156 EventNode *events = NULL;
157
158 // Check for a non-empty file
159 if (access(eventFile, F_OK) != 0) {
160 // no file found, which is fine; we're done, a vector of NULL is what we
161 // want for events
162 printf("No event file found, assuming no events\n");
163 return events;
164 }
165
166 printf("Begin reading event data from file %s\n", eventFile);
167
168 FILE *in = openFile(eventFile, "r");
169
170 if (fgets(line, EVENT_LINE_SIZE, in) == NULL) {
171 // Again, this is fine - just return the empty events array
172 return events;
173 }
174
175 int numRead = sscanf(line, // NOLINT
176 "%d %d %s %n", &year, &day, eventTypeStr, &numBytes);
177 if (numRead != NUM_EVENT_CORE_PARAMS) {
178 printf("Error reading event file: bad data on first line\n");
179 exit(EXIT_CODE_INPUT_FILE_ERROR);
180 }
181 eventParamsStr = line + numBytes;
182
183 eventType = getEventType(eventTypeStr);
184 if (eventType == UNKNOWN_EVENT) {
185 printf("Error reading event file: unknown event type %s\n", eventTypeStr);
186 exit(EXIT_CODE_UNKNOWN_EVENT_TYPE_OR_PARAM);
187 }
188
189 events = createEventNode(year, day, eventType, eventParamsStr);
190 next = events;
191 currYear = year;
192 currDay = day;
193
194 while (fgets(line, EVENT_LINE_SIZE, in) != NULL) {
195 // We have another event
196 curr = next;
197 numRead = sscanf(line, "%d %d %s %n", // NOLINT
198 &year, &day, eventTypeStr, &numBytes);
199 if (numRead != NUM_EVENT_CORE_PARAMS) {
200 printf(
201 "Error reading event file: bad data on line after year %d day %d\n",
202 currYear, currDay);
203 exit(EXIT_CODE_INPUT_FILE_ERROR);
204 }
205 eventParamsStr = line + numBytes;
206
207 eventType = getEventType(eventTypeStr);
208 if (eventType == UNKNOWN_EVENT) {
209 printf("Error reading event file: unknown event type %s\n", eventTypeStr);
210 exit(EXIT_CODE_UNKNOWN_EVENT_TYPE_OR_PARAM);
211 }
212
213 if ((year < currYear) || ((year == currYear) && (day < currDay))) {
214 printf("Error reading event file: last event was at (%d, %d) ", currYear,
215 currDay);
216 printf("next event is at (%d, %d)\n", year, day);
217 printf("Event records must be in time-ascending order\n");
218 exit(EXIT_CODE_INPUT_FILE_ERROR);
219 }
220
221 next = createEventNode(year, day, eventType, eventParamsStr);
222 curr->nextEvent = next;
223 currYear = year;
224 currDay = day;
225 }
226
227 fclose(in);
228 return events;
229}

232 if (event == NULL) {

249 break;

251 printf("PLANTING on %d %d, ", year, day);
252 PlantingParams *const pParams = (PlantingParams *)event->eventParams;
253 printf("with params: leaf C %4.2f, wood C %4.2f, fine root C %4.2f, "

255 pParams->leafC, pParams->woodC, pParams->fineRootC,
256 pParams->coarseRootC);
257 break;
258 case TILLAGE:
259 printf("TILLAGE at on %d %d, ", year, day);
260 TillageParams *const tParams = (TillageParams *)event->eventParams;
261 printf("with params: frac litter transferred %4.2f, som decomp modifier "
262 "%4.2f, litter decomp modifier %4.2f\n",
263 tParams->fractionLitterTransferred, tParams->somDecompModifier,
264 tParams->litterDecompModifier);
265 break;
266 case HARVEST:
267 printf("HARVEST on %d %d, ", year, day);
268 HarvestParams *const hParams = (HarvestParams *)event->eventParams;
269 printf("with params: frac removed above %4.2f, frac removed below %4.2f, "
270 "frac transferred above %4.2f, frac transferred below %4.2f\n",
271 hParams->fractionRemovedAbove, hParams->fractionRemovedBelow,
272 hParams->fractionTransferredAbove,
273 hParams->fractionTransferredBelow);
274 break;
275 default:
276 printf("Error printing event: unknown type %d\n", event->type);

309void writeEventOut(FILE *out, EventNode *event, int numParams, ...) {
310 va_list args;
311 int ind = 0;
312
313 // Spec:
314 // year day event_type <param name=delta>[,<param name>=<delta>,...]
315
316 // Standard prefix for all
317 fprintf(out, "%4d %3d %-7s ", event->year, event->day,
318 eventTypeToString(event->type));
319 // Variable output per event type
320 va_start(args, numParams);
321 for (ind = 0; ind < numParams - 1; ind++) {
322 fprintf(out, "%s=%-.2f,", va_arg(args, char *), va_arg(args, double));
323 }
324 fprintf(out, "%s=%-.2f\n", va_arg(args, char *), va_arg(args, double));
325 va_end(args);
326}
