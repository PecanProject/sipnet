25EventNode *createEventNode(int year, int day, int eventType,
26 const char *eventParamsStr) {
27 EventNode *event = (EventNode *)malloc(sizeof(EventNode));
28 event->year = year;
29 event->day = day;
30 event->type = eventType;
31
32 switch (eventType) {
33 case HARVEST: {
34 double fracRA, fracRB, fracTA, fracTB;
35 HarvestParams *params = (HarvestParams *)malloc(sizeof(HarvestParams));
36 int numRead =
37 sscanf(eventParamsStr, // NOLINT
38 "%lf %lf %lf %lf", &fracRA, &fracRB, &fracTA, &fracTB);
39 if (numRead != NUM_HARVEST_PARAMS) {
40 logError("parsing Harvest params for year %d day %d\n", year, day);
41 exit(EXIT_CODE_INPUT_FILE_ERROR);
42 }
43 // Validate the params
44 if ((fracRA + fracTA > 1) || (fracRB + fracTB > 1)) {
45 logError("nvalid harvest event for year %d day %d; above and below "
46 "must each add to 1 or less",
47 year, day);
48 exit(EXIT_CODE_BAD_PARAMETER_VALUE);
49 }
50 params->fractionRemovedAbove = fracRA;
51 params->fractionRemovedBelow = fracRB;
52 params->fractionTransferredAbove = fracTA;
53 params->fractionTransferredBelow = fracTB;
54 event->eventParams = params;
55 } break;
56 case IRRIGATION: {
57 double amountAdded;
58 int method;
59 IrrigationParams *params =
60 (IrrigationParams *)malloc(sizeof(IrrigationParams));
61 int numRead = sscanf(eventParamsStr, // NOLINT
62 "%lf %d", &amountAdded, &method);
63 if (numRead != NUM_IRRIGATION_PARAMS) {
64 logError("parsing Irrigation params for year %d day %d\n", year, day);
65 exit(EXIT_CODE_INPUT_FILE_ERROR);
66 }
67 params->amountAdded = amountAdded;
68 params->method = method;
69 event->eventParams = params;
70 } break;
71 case FERTILIZATION: {
72 // If/when we try two N pools, enable the additional nh4_no3_frac param
73 // (likely via compiler switch)
74 double orgN, orgC, minN;
75 // double nh4_no3_frac;
76 FertilizationParams *params =
77 (FertilizationParams *)malloc(sizeof(FertilizationParams));
78 int numRead = sscanf(eventParamsStr, // NOLINT
79 "%lf %lf %lf", &orgN, &orgC, &minN);
80 if (numRead != NUM_FERTILIZATION_PARAMS) {
81 logError("parsing Fertilization params for year %d day %d\n", year,
82 day);
83 exit(EXIT_CODE_INPUT_FILE_ERROR);
84 }
85 // scanf(eventParamsStr, "%lf %lf %lf %lf", &org_N, &org_C, &min_N,
86 // &nh4_no3_frac);
87 params->orgN = orgN;
88 params->orgC = orgC;
89 params->minN = minN;
90 // params->nh4_no3_frac = nh4_nos_frac;
91 event->eventParams = params;
92 } break;
93 case PLANTING: {
94 double leafC, woodC, fineRootC, coarseRootC;
95 PlantingParams *params = (PlantingParams *)malloc(sizeof(PlantingParams));
96 int numRead =
97 sscanf(eventParamsStr, // NOLINT
98 "%lf %lf %lf %lf", &leafC, &woodC, &fineRootC, &coarseRootC);
99 if (numRead != NUM_PLANTING_PARAMS) {
100 logError("parsing Planting params for year %d day %d\n", year, day);
101 exit(EXIT_CODE_INPUT_FILE_ERROR);
102 }
103 params->leafC = leafC;
104 params->woodC = woodC;
105 params->fineRootC = fineRootC;
106 params->coarseRootC = coarseRootC;
107 event->eventParams = params;
108 } break;
109 case TILLAGE: {
110 double fracLT, somDM, litterDM;
111 TillageParams *params = (TillageParams *)malloc(sizeof(TillageParams));
112 int numRead = sscanf(eventParamsStr, // NOLINT
113 "%lf %lf %lf", &fracLT, &somDM, &litterDM);
114 if (numRead != NUM_TILLAGE_PARAMS) {
115 logError("parsing Tillage params for year %d day %d\n", year, day);
116 exit(EXIT_CODE_INPUT_FILE_ERROR);
117 }
118 params->fractionLitterTransferred = fracLT;
119 params->somDecompModifier = somDM;
120 params->litterDecompModifier = litterDM;
121 event->eventParams = params;
122 } break;
123 default:
124 // Unknown type, error and exit
125 logError("reading event file: unknown event type %d\n", eventType);
126 exit(1);
127 }
128
129 event->nextEvent = NULL;
130 return event;
131}

133event_type_t getEventType(const char *eventTypeStr) {
134 if (strcmp(eventTypeStr, "irrig") == 0) {
135 return IRRIGATION;
136 } else if (strcmp(eventTypeStr, "fert") == 0) {
137 return FERTILIZATION;
138 } else if (strcmp(eventTypeStr, "plant") == 0) {
139 return PLANTING;
140 } else if (strcmp(eventTypeStr, "till") == 0) {
141 return TILLAGE;
142 } else if (strcmp(eventTypeStr, "harv") == 0) {
143 return HARVEST;
144 }
145 return UNKNOWN_EVENT;
146}

148EventNode *readEventData(char *eventFile) {
149 int year, day, eventType;
150 int currYear, currDay;
151 int EVENT_LINE_SIZE = 1024;
152 int numBytes;
153 char *eventParamsStr;
154 char eventTypeStr[20];
155 char line[EVENT_LINE_SIZE];
156 EventNode *curr, *next;
157 EventNode *events = NULL;
158
159 // Check for a non-empty file
160 if (access(eventFile, F_OK) != 0) {
161 // no file found, which is fine; we're done, a vector of NULL is what we
162 // want for events
163 logInfo("No event file found, assuming no events\n");
164 return events;
165 }
166
167 logInfo("Begin reading event data from file %s\n", eventFile);
168
169 FILE *in = openFile(eventFile, "r");
170
171 if (fgets(line, EVENT_LINE_SIZE, in) == NULL) {
172 // Again, this is fine - just return the empty events array
173 return events;
174 }
175
176 int numRead = sscanf(line, // NOLINT
177 "%d %d %s %n", &year, &day, eventTypeStr, &numBytes);
178 if (numRead != NUM_EVENT_CORE_PARAMS) {
179 logError("reading event file: bad data on first line\n");
180 exit(EXIT_CODE_INPUT_FILE_ERROR);
181 }
182 eventParamsStr = line + numBytes;
183
184 eventType = getEventType(eventTypeStr);
185 if (eventType == UNKNOWN_EVENT) {
186 logError("reading event file: unknown event type %s\n", eventTypeStr);
187 exit(EXIT_CODE_UNKNOWN_EVENT_TYPE_OR_PARAM);
188 }
189
190 events = createEventNode(year, day, eventType, eventParamsStr);
191 next = events;
192 currYear = year;
193 currDay = day;
194
195 while (fgets(line, EVENT_LINE_SIZE, in) != NULL) {
196 // We have another event
197 curr = next;
198 numRead = sscanf(line, "%d %d %s %n", // NOLINT
199 &year, &day, eventTypeStr, &numBytes);
200 if (numRead != NUM_EVENT_CORE_PARAMS) {
201 logError("reading event file: bad data on line after year %d day %d\n",
202 currYear, currDay);
203 exit(EXIT_CODE_INPUT_FILE_ERROR);
204 }
205 eventParamsStr = line + numBytes;
206
207 eventType = getEventType(eventTypeStr);
208 if (eventType == UNKNOWN_EVENT) {
209 logError("reading event file: unknown event type %s\n", eventTypeStr);
210 exit(EXIT_CODE_UNKNOWN_EVENT_TYPE_OR_PARAM);
211 }
212
213 if ((year < currYear) || ((year == currYear) && (day < currDay))) {
214 // clang-format off
215 logError("reading event file: last event was at (%d, %d), next event is "
216 "at (%d, %d)\n", currDay, year, day);
217 logError("event records must be in time-ascending order\n", currYear);
218 // clang-format on
219 exit(EXIT_CODE_INPUT_FILE_ERROR);
220 }
221
222 next = createEventNode(year, day, eventType, eventParamsStr);
223 curr->nextEvent = next;
224 currYear = year;
225 currDay = day;
226 }
227
228 fclose(in);
229 return events;
230}

258 break;
259 case TILLAGE:
260 printf("TILLAGE at on %d %d, ", year, day);
261 TillageParams *const tParams = (TillageParams *)event->eventParams;

265 tParams->litterDecompModifier);
266 break;
267 case HARVEST:
268 printf("HARVEST on %d %d, ", year, day);
269 HarvestParams *const hParams = (HarvestParams *)event->eventParams;

272 hParams->fractionRemovedAbove, hParams->fractionRemovedBelow,

274 hParams->fractionTransferredBelow);
275 break;

310void writeEventOut(FILE *out, EventNode *event, int numParams, ...) {
311 va_list args;
312 int ind = 0;
313
314 // Spec:
315 // year day event_type <param name=delta>[,<param name>=<delta>,...]
316
317 // Standard prefix for all
318 fprintf(out, "%4d %3d %-7s ", event->year, event->day,
319 eventTypeToString(event->type));
320 // Variable output per event type
321 va_start(args, numParams);
322 for (ind = 0; ind < numParams - 1; ind++) {
323 fprintf(out, "%s=%-.2f,", va_arg(args, char *), va_arg(args, double));
324 }
325 fprintf(out, "%s=%-.2f\n", va_arg(args, char *), va_arg(args, double));
326 va_end(args);
327}
