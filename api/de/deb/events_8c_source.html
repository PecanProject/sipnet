33EventNode *createEventNode(int year, int day, int eventType,
34 const char *eventParamsStr) {
35 EventNode *newEvent = (EventNode *)malloc(sizeof(EventNode));
36 newEvent->year = year;
37 newEvent->day = day;
38 newEvent->type = eventType;
39
40 switch (eventType) {
41 case HARVEST: {
42 double fracRA, fracRB, fracTA, fracTB;
43 HarvestParams *hParams = (HarvestParams *)malloc(sizeof(HarvestParams));
44 int numRead =
45 sscanf(eventParamsStr, // NOLINT
46 "%lf %lf %lf %lf", &fracRA, &fracRB, &fracTA, &fracTB);
47 if (numRead != NUM_HARVEST_PARAMS) {
48 logError("parsing Harvest params for year %d day %d\n", year, day);
49 exit(EXIT_CODE_INPUT_FILE_ERROR);
50 }
51 // Validate the params
52 if ((fracRA + fracTA > 1) || (fracRB + fracTB > 1)) {
53 logError("invalid harvest newEvent for year %d day %d; above and below "
54 "must each add to 1 or less",
55 year, day);
56 exit(EXIT_CODE_BAD_PARAMETER_VALUE);
57 }
58 hParams->fractionRemovedAbove = fracRA;
59 hParams->fractionRemovedBelow = fracRB;
60 hParams->fractionTransferredAbove = fracTA;
61 hParams->fractionTransferredBelow = fracTB;
62 newEvent->eventParams = hParams;
63 } break;
64 case IRRIGATION: {
65 double amountAdded;
66 int method;
67 IrrigationParams *iParams =
68 (IrrigationParams *)malloc(sizeof(IrrigationParams));
69 int numRead = sscanf(eventParamsStr, // NOLINT
70 "%lf %d", &amountAdded, &method);
71 if (numRead != NUM_IRRIGATION_PARAMS) {
72 logError("parsing Irrigation params for year %d day %d\n", year, day);
73 exit(EXIT_CODE_INPUT_FILE_ERROR);
74 }
75 iParams->amountAdded = amountAdded;
76 iParams->method = method;
77 newEvent->eventParams = iParams;
78 } break;
79 case FERTILIZATION: {
80 // If/when we try two N pools, enable the additional nh4_no3_frac param
81 // (likely via compiler switch)
82 double orgN, orgC, minN;
83 // double nh4_no3_frac;
84 FertilizationParams *fParams =
85 (FertilizationParams *)malloc(sizeof(FertilizationParams));
86 int numRead = sscanf(eventParamsStr, // NOLINT
87 "%lf %lf %lf", &orgN, &orgC, &minN);
88 if (numRead != NUM_FERTILIZATION_PARAMS) {
89 logError("parsing Fertilization params for year %d day %d\n", year,
90 day);
91 exit(EXIT_CODE_INPUT_FILE_ERROR);
92 }
93 // scanf(eventParamsStr, "%lf %lf %lf %lf", &org_N, &org_C, &min_N,
94 // &nh4_no3_frac);
95 fParams->orgN = orgN;
96 fParams->orgC = orgC;
97 fParams->minN = minN;
98 // params->nh4_no3_frac = nh4_nos_frac;
99 newEvent->eventParams = fParams;
100 } break;
101 case PLANTING: {
102 double leafC, woodC, fineRootC, coarseRootC;
103 PlantingParams *pParams =
104 (PlantingParams *)malloc(sizeof(PlantingParams));
105 int numRead =
106 sscanf(eventParamsStr, // NOLINT
107 "%lf %lf %lf %lf", &leafC, &woodC, &fineRootC, &coarseRootC);
108 if (numRead != NUM_PLANTING_PARAMS) {
109 logError("parsing Planting params for year %d day %d\n", year, day);
110 exit(EXIT_CODE_INPUT_FILE_ERROR);
111 }
112 pParams->leafC = leafC;
113 pParams->woodC = woodC;
114 pParams->fineRootC = fineRootC;
115 pParams->coarseRootC = coarseRootC;
116 newEvent->eventParams = pParams;
117 } break;
118 case TILLAGE: {
119 double tillEffect;
120 TillageParams *tParams = (TillageParams *)malloc(sizeof(TillageParams));
121 int numRead = sscanf(eventParamsStr, // NOLINT
122 "%lf", &tillEffect);
123 if (numRead != NUM_TILLAGE_PARAMS) {
124 logError("parsing Tillage params for year %d day %d\n", year, day);
125 exit(EXIT_CODE_INPUT_FILE_ERROR);
126 }
127 tParams->tillageEffect = tillEffect;
128 newEvent->eventParams = tParams;
129 } break;
130 default:
131 // Unknown type, error and exit
132 logError("reading newEvent file: unknown newEvent type %d\n", eventType);
133 exit(1);
134 }
135
136 newEvent->nextEvent = NULL;
137 return newEvent;
138}

158event_type_t eventStringToType(const char *eventTypeStr) {
159 if (strcmp(eventTypeStr, "irrig") == 0) {
160 return IRRIGATION;
161 } else if (strcmp(eventTypeStr, "fert") == 0) {
162 return FERTILIZATION;
163 } else if (strcmp(eventTypeStr, "plant") == 0) {
164 return PLANTING;
165 } else if (strcmp(eventTypeStr, "till") == 0) {

168 return HARVEST;

173EventNode *readEventData(char *eventFile) {
174 int year, day, eventType;
175 int currYear, currDay;
176 int EVENT_LINE_SIZE = 1024;
177 int numBytes;
178 char *eventParamsStr;
179 char eventTypeStr[20];
180 char line[EVENT_LINE_SIZE];
181 EventNode *curr, *next;
182 EventNode *newEvents = NULL;
183
184 // Check for a non-empty file
185 if (access(eventFile, F_OK) != 0) {
186 // no file found, which is fine; we're done, a vector of NULL is what we
187 // want for newEvents
188 logInfo("No event file found, assuming no newEvents\n");
189 return newEvents;
190 }
191
192 logInfo("Begin reading event data from file %s\n", eventFile);
193
194 FILE *in = openFile(eventFile, "r");
195
196 if (fgets(line, EVENT_LINE_SIZE, in) == NULL) {
197 // Again, this is fine - just return the empty newEvents array
198 return newEvents;
199 }
200
201 int numRead = sscanf(line, // NOLINT
202 "%d %d %s %n", &year, &day, eventTypeStr, &numBytes);
203 if (numRead != NUM_EVENT_CORE_PARAMS) {
204 logError("reading event file: bad data on first line\n");
205 exit(EXIT_CODE_INPUT_FILE_ERROR);
206 }
207 eventParamsStr = line + numBytes;
208
209 eventType = eventStringToType(eventTypeStr);
210 if (eventType == UNKNOWN_EVENT) {
211 logError("reading event file: unknown event type %s\n", eventTypeStr);
212 exit(EXIT_CODE_UNKNOWN_EVENT_TYPE_OR_PARAM);
213 }
214
215 newEvents = createEventNode(year, day, eventType, eventParamsStr);
216 next = newEvents;
217 currYear = year;
218 currDay = day;
219
220 while (fgets(line, EVENT_LINE_SIZE, in) != NULL) {
221 // We have another event
222 curr = next;
223 numRead = sscanf(line, "%d %d %s %n", // NOLINT
224 &year, &day, eventTypeStr, &numBytes);
225 if (numRead != NUM_EVENT_CORE_PARAMS) {
226 logError("reading event file: bad data on line after year %d day %d\n",
227 currYear, currDay);
228 exit(EXIT_CODE_INPUT_FILE_ERROR);
229 }
230 eventParamsStr = line + numBytes;
231
232 eventType = eventStringToType(eventTypeStr);
233 if (eventType == UNKNOWN_EVENT) {
234 logError("reading event file: unknown event type %s\n", eventTypeStr);
235 exit(EXIT_CODE_UNKNOWN_EVENT_TYPE_OR_PARAM);
236 }
237
238 if ((year < currYear) || ((year == currYear) && (day < currDay))) {
239 // clang-format off
240 logError("reading event file: last event was at (%d, %d), next event is "
241 "at (%d, %d)\n", currYear, currDay, year, day);
242 logError("event records must be in time-ascending order\n");
243 // clang-format on
244 exit(EXIT_CODE_INPUT_FILE_ERROR);
245 }
246
247 next = createEventNode(year, day, eventType, eventParamsStr);
248 curr->nextEvent = next;
249 currYear = year;
250 currDay = day;
251 }
252
253 fclose(in);
254 return newEvents;
255}

267void writeEventOut(EventNode *oneEvent, int numParams, ...) {
268 va_list args;
269 int ind = 0;
270
271 // Spec:
272 // year day event_type <param name=delta>[,<param name>=<delta>,...]
273
274 // Standard prefix for all
275 fprintf(eventOutFile, "%4d %3d %-7s ", oneEvent->year, oneEvent->day,

277 // Variable output per oneEvent type

284 va_arg(args, double));

313void processEvents(void) {
314 // This should be in events.h/c, but with all the global state defined in
315 // this file, let's leave it here for now. Maybe someday we will factor that
316 // out.
317
318 // Set all event fluxes to zero, as these have no memory from one step to
319 // the next
320 resetEventFluxes();
321
322 // If event starts off NULL, this function will just fall through, as it
323 // should.
324 const int climYear = climate->year;
325 const int climDay = climate->day;
326 const double climLen = climate->length;
327
328 // As this is used as a divisor in many places, let's make sure it's >0
329 if (climLen <= 0) {
330 logError("climate length (%f) on year %d day %d is non-positive; please "
331 "fix and re-run",
332 climLen, climYear, climDay);
333 exit(EXIT_CODE_BAD_PARAMETER_VALUE);
334 }
335
336 // The events file has been tested on read, so we know this event list
337 // should be in chrono order. However, we need to check to make sure the
338 // current event is not in the past, as that would indicate an event that
339 // did not have a corresponding climate file record.
340 while (gEvent != NULL && gEvent->year <= climYear && gEvent->day <= climDay) {
341 if (gEvent->year < climYear || gEvent->day < climDay) {
342 logError("Agronomic event found for year: %d day: %d that does not "
343 "have a corresponding record in the climate file\n",
344 gEvent->year, gEvent->day);
345 exit(EXIT_CODE_INPUT_FILE_ERROR);
346 }
347 switch (gEvent->type) {
348 case IRRIGATION: {
349 const IrrigationParams *irrParams = gEvent->eventParams;
350 const double amount = irrParams->amountAdded;
351 double soilAmount, evapAmount;
352 if (irrParams->method == CANOPY) {
353 // Part of the irrigation evaporates, and the rest makes it to the
354 // soil. Evaporated fraction:
355 evapAmount = params.immedEvapFrac * amount;
356 // Soil fraction:
357 soilAmount = amount - evapAmount;
358 } else if (irrParams->method == SOIL) {
359 // All goes to the soil
360 evapAmount = 0.0;
361 soilAmount = amount;
362 } else {
363 logError("Unknown irrigation method type: %d\n", irrParams->method);
364 exit(EXIT_CODE_UNKNOWN_EVENT_TYPE_OR_PARAM);
365 }
366 fluxes.eventEvap += evapAmount / climLen;
367 fluxes.eventSoilWater += soilAmount / climLen;
368 writeEventOut(gEvent, 2, "fluxes.eventSoilWater", soilAmount / climLen,
369 "fluxes.eventEvap", evapAmount / climLen);
370 } break;
371 case PLANTING: {
372 const PlantingParams *plantParams = gEvent->eventParams;
373 const double leafC = plantParams->leafC;
374 const double woodC = plantParams->woodC;
375 const double fineRootC = plantParams->fineRootC;
376 const double coarseRootC = plantParams->coarseRootC;
377
378 // Update the fluxes
379 fluxes.eventLeafC += leafC / climLen;
380 fluxes.eventWoodC += woodC / climLen;
381 fluxes.eventFineRootC += fineRootC / climLen;
382 fluxes.eventCoarseRootC += coarseRootC / climLen;
383
384 // FUTURE: allocate to N pools
385
386 writeEventOut(gEvent, 4, "fluxes.eventLeafC", leafC / climLen,
387 "fluxes.eventWoodC", woodC / climLen,
388 "fluxes.eventFineRootC", fineRootC / climLen,
389 "fluxes.eventCoarseRootC", coarseRootC / climLen);
390 } break;
391 case HARVEST: {
392 // Harvest can both remove biomass and move biomass to the litter pool
393 const HarvestParams *harvParams = gEvent->eventParams;
394 const double fracRA = harvParams->fractionRemovedAbove;
395 const double fracTA = harvParams->fractionTransferredAbove;
396 const double fracRB = harvParams->fractionRemovedBelow;
397 const double fracTB = harvParams->fractionTransferredBelow;
398
399 // Litter increase
400 const double litterAdd = fracTA * (envi.plantLeafC + envi.plantWoodC) +
401 fracTB * (envi.fineRootC + envi.coarseRootC);
402 // Pool reductions, counting both mass moved to litter and removed by
403 // the harvest itself. Above-ground changes:
404 const double leafDelta = -envi.plantLeafC * (fracRA + fracTA);
405 const double woodDelta = -envi.plantWoodC * (fracRA + fracTA);
406 // Below-ground changes:
407 const double fineDelta = -envi.fineRootC * (fracRB + fracTB);
408 const double coarseDelta = -envi.coarseRootC * (fracRB + fracTB);
409
410 // Pool updates:
411 fluxes.eventLitterC += litterAdd / climLen;
412 fluxes.eventLeafC += leafDelta / climLen;
413 fluxes.eventWoodC += woodDelta / climLen;
414 fluxes.eventFineRootC += fineDelta / climLen;
415 fluxes.eventCoarseRootC += coarseDelta / climLen;
416
417 // FUTURE: move/remove biomass in N pools
418 writeEventOut(gEvent, 5, "fluxes.eventLitterC", litterAdd / climLen,
419 "fluxes.eventLeafC", leafDelta / climLen,
420 "fluxes.eventWoodC", woodDelta / climLen,
421 "fluxes.eventFineRootC", fineDelta / climLen,
422 "fluxes.eventCoarseRootC", coarseDelta / climLen);
423 } break;
424 case TILLAGE: {
425 // BIG NOTE: this is the one event type that is NOT modeled as a flux;
426 // see updateEventTrackers() for more
427 const TillageParams *tillParams = gEvent->eventParams;
428 // Update the tillage mod for R_H calculations; this will be slowly
429 // reduced by an exponential decay function. Note we add here, not set,
430 // as there may be lingering effects from a prior tillage.
431 eventTrackers.d_till_mod += tillParams->tillageEffect;
432 writeEventOut(gEvent, 1, "eventTrackers.d_till_mod",
433 tillParams->tillageEffect);
434 } break;
435 case FERTILIZATION: {
436 const FertilizationParams *fertParams = gEvent->eventParams;
437 // const double orgN = fertParams->orgN;
438 const double orgC = fertParams->orgC;
439 // const double minN = fertParams->minN;
440
441 fluxes.eventLitterC += orgC / climLen;
442
443 // FUTURE: allocate to N pools
444
445 // This will (likely) be 3 params eventually
446 writeEventOut(gEvent, 1, "fluxes.eventLitterC", orgC / climLen);
447 } break;
448 default:
449 logError("Unknown event type (%d) in processEvents()\n", gEvent->type);
450 exit(EXIT_CODE_UNKNOWN_EVENT_TYPE_OR_PARAM);
451 }
452
453 gEvent = gEvent->nextEvent;
454 }
455}

457void updatePoolsForEvents(void) {
458 // Harvest and planting events
459 envi.plantWoodC += fluxes.eventWoodC * climate->length;
460 envi.plantLeafC += fluxes.eventLeafC * climate->length;
461
462 // Irrigation events
463 envi.soilWater += fluxes.eventSoilWater * climate->length;
464
465 // Harvest and fertilization events
466 if (ctx.litterPool) {
467 envi.litter += fluxes.eventLitterC * climate->length;
468 } else {
469 envi.soil += fluxes.eventLitterC * climate->length;
470 }
471
472 // Harvest and planting events
473 envi.coarseRootC += fluxes.eventCoarseRootC * climate->length;
474 envi.fineRootC += fluxes.eventFineRootC * climate->length;
475}
