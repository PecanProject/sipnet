36EventNode *createEventNode(int loc, int year, int day, int eventType,
37 const char *eventParamsStr) {
38 EventNode *event = (EventNode *)malloc(sizeof(EventNode));
39 event->loc = loc;
40 event->year = year;
41 event->day = day;
42 event->type = eventType;
43
44 switch (eventType) {
45 case HARVEST: {
46 double fracRA, fracRB, fracTA, fracTB;
47 HarvestParams *params = (HarvestParams *)malloc(sizeof(HarvestParams));
48 int numRead = sscanf(eventParamsStr, "%lf %lf %lf %lf", &fracRA, &fracRB,
49 &fracTA, &fracTB);
50 if (numRead != NUM_HARVEST_PARAMS) {
51 printf("Error parsing Harvest params for loc %d year %d day %d\n", loc,
52 year, day);
53 exit(EXIT_CODE_INPUT_FILE_ERROR);
54 }
55 // Validate the params
56 if ((fracRA + fracTA > 1) || (fracRB + fracTB > 1)) {
57 printf("Invalid harvest event for loc %d year %d day %d; above and "
58 "below must each add to 1 or less",
59 loc, year, day);
60 exit(EXIT_CODE_BAD_PARAMETER_VALUE);

62 params->fractionRemovedAbove = fracRA;
63 params->fractionRemovedBelow = fracRB;

138 exit(1);

141 event->nextEvent = NULL;

145event_type_t getEventType(const char *eventTypeStr) {
146 if (strcmp(eventTypeStr, "irrig") == 0) {
147 return IRRIGATION;
148 } else if (strcmp(eventTypeStr, "fert") == 0) {
149 return FERTILIZATION;
150 } else if (strcmp(eventTypeStr, "plant") == 0) {
151 return PLANTING;
152 } else if (strcmp(eventTypeStr, "till") == 0) {
153 return TILLAGE;
154 } else if (strcmp(eventTypeStr, "harv") == 0) {
155 return HARVEST;
156 }
157 return UNKNOWN_EVENT;
158}

160EventNode **readEventData(char *eventFile, int numLocs) {
161 int loc, year, day, eventType;
162 int currLoc, currYear, currDay;
163 int EVENT_LINE_SIZE = 1024;
164 int numBytes;
165 char *eventParamsStr;
166 char eventTypeStr[20];
167 char line[EVENT_LINE_SIZE];
168 EventNode *curr, *next;
169
170 EventNode **events =
171 (EventNode **)calloc(sizeof(EventNode *), numLocs * sizeof(EventNode *));
172
173 // Check for a non-empty file
174 if (access(eventFile, F_OK) != 0) {
175 // no file found, which is fine; we're done, a vector of NULL is what we
176 // want for events
177 printf("No event file found, assuming no events\n");
178 return events;
179 }
180
181 printf("Begin reading event data from file %s\n", eventFile);
182
183 FILE *in = openFile(eventFile, "r");
184
185 if (fgets(line, EVENT_LINE_SIZE, in) == NULL) {
186 // Again, this is fine - just return the empty events array
187 return events;
188 }
189
190 int numRead = sscanf(line, "%d %d %d %s %n", &loc, &year, &day, eventTypeStr,
191 &numBytes);
192 if (numRead != NUM_EVENT_CORE_PARAMS) {
193 printf("Error reading event file: bad data on first line\n");
194 exit(EXIT_CODE_INPUT_FILE_ERROR);
195 }
196 eventParamsStr = line + numBytes;
197
198 eventType = getEventType(eventTypeStr);
199 if (eventType == UNKNOWN_EVENT) {
200 printf("Error reading event file: unknown event type %s\n", eventTypeStr);
201 exit(EXIT_CODE_UNKNOWN_EVENT_TYPE_OR_PARAM);
202 }
203
204 next = createEventNode(loc, year, day, eventType, eventParamsStr);
205 events[loc] = next;
206 currLoc = loc;
207 currYear = year;
208 currDay = day;
209
210 while (fgets(line, EVENT_LINE_SIZE, in) != NULL) {
211 // We have another event
212 curr = next;
213 numRead = sscanf(line, "%d %d %d %s %n", &loc, &year, &day, eventTypeStr,
214 &numBytes);
215 if (numRead != 4) {
216 printf("Error reading event file: bad data on first line\n");
217 exit(EXIT_CODE_INPUT_FILE_ERROR);
218 }
219 eventParamsStr = line + numBytes;
220
221 eventType = getEventType(eventTypeStr);
222 if (eventType == UNKNOWN_EVENT) {
223 printf("Error reading event file: unknown event type %s\n", eventTypeStr);
224 exit(EXIT_CODE_UNKNOWN_EVENT_TYPE_OR_PARAM);
225 }
226
227 // make sure location and time are non-decreasing
228 if (loc < currLoc) {
229 printf("Error reading event file: was reading location %d, trying to "
230 "read location %d\n",
231 currLoc, loc);
232 printf("Event records for a given location should be contiguous, and "
233 "locations should be in ascending order\n");
234 exit(EXIT_CODE_INPUT_FILE_ERROR);
235 }
236 if ((loc == currLoc) && ((year < currYear) || (day < currDay))) {
237 printf("Error reading event file: for location %d, last event was at "
238 "(%d, %d) ",
239 currLoc, currYear, currDay);
240 printf("next event is at (%d, %d)\n", year, day);
241 printf("Event records for a given location should be in time-ascending "
242 "order\n");
243 exit(EXIT_CODE_INPUT_FILE_ERROR);
244 }
245
246 next = createEventNode(loc, year, day, eventType, eventParamsStr);
247 if (currLoc == loc) {
248 // Same location, add the new event to this location's list
249 curr->nextEvent = next;
250 } else {
251 // New location, update location and start a new list
252 currLoc = loc;
253 events[currLoc] = next;
254 }
255 }
256
257 fclose(in);
258 return events;
259}
