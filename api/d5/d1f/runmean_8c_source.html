22MeanTracker *newMeanTracker(double initMean, double totWeight, int maxEntries) {
23 MeanTracker *tracker;
24
25 // error check:
26 if (totWeight <= 0 || maxEntries < 1) {
27 return NULL;

40/* PRE: tracker has been created using newMeanTracker
41 so has malloced values and weights arrays, and has length and totWeight
42 set reset tracker to have a single entry, with mean initMean
43*/
44void resetMeanTracker(MeanTracker *tracker, double initMean) {
45 // start with only one entry, with appropriate mean and total weight
46 // don't bother resetting other entries - we'll set them as we come to them
47 tracker->start = tracker->last = 0;
48 tracker->values[tracker->start] = initMean;
49 tracker->weights[tracker->start] = tracker->totWeight;
50 tracker->sum = initMean * tracker->totWeight;
51}

53/* PRE: weight > 0
54 Add current value as most recent value in tracker
55
56 Return value:
57 0 if all okay
58 -1 if error in inputs
59 -2 if error because array is too short
60*/
61int addValueToMeanTracker(MeanTracker *tracker, double value, double weight) {
62 double weightLeft;
63 int i;
64
65 if (weight <= 0) { // error
66 return -1; // don't change tracker at all
67 } else if (weight >= tracker->totWeight) { // current value knocks out all
68 // previous values
69 resetMeanTracker(tracker, value);
70 } else { // 0 < weight < totWeight: remove oldest values to make room for new
71 // (keeping total weight the same)
72 weightLeft = weight; // how much more do we have to subtract to make room?
73 i = tracker->start; // current position
74
75 while (weightLeft > 0) { // while we still have to subtract more
76 if (tracker->weights[i] > weightLeft) { // subtract a portion of this one
77 // and we'll be done
78 tracker->weights[i] -= weightLeft; // subtract correct portion from
79 // current weight
80 tracker->sum -= weightLeft * tracker->values[i]; // subtract correct
81 // portion from sum
82 weightLeft = 0; // we're done
83 } else { // remove this value entirely to make room
84 tracker->sum -= tracker->weights[i] * tracker->values[i]; // update sum
85 weightLeft -= tracker->weights[i]; // we now have this much less weight
86 // to subtract
87 i = (i + 1) % tracker->length; // forget this value
88 // note: we don't have to worry about going past last value, since 1) we
89 // check for weight >= tracker->totWeight, and 2) before starting this
90 // function, we know that sum(tracker->weights[start..last] ==
91 // tracker->totWeight)
92 }
93 } // end while: now weightLeft = 0
94
95 tracker->start = i; // move start position along to correct new position
96 i = (tracker->last + 1) % tracker->length; // point to position of new
97 // value
98 if (i == tracker->start) { // uh-oh: we've wrapped around and are out of
99 // space!
100 // restore previous state and return error value
101 // note: this will only happen if we only executed the while loop once,
102 // and entered the first if block we can take advantage of this fact to
103 // restore the old state
104 tracker->weights[i] += weight; // restore old weight here
105 tracker->sum += weight * tracker->values[i]; // restore old sum
106 return -2; // error
107 } else { // there's enough space for the new value
108 tracker->last = i; // increment tracker->last to point to new value
109 tracker->values[i] = value;
110 tracker->weights[i] = weight;
111 tracker->sum += value * weight; // update sum appropriately
112 }
113 } // end else (0 < weight < totWeight)
114
115 return 0; // all okay
116}
