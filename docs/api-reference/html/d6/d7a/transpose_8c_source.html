25/* Count and return # of lines in f

27 - includes extra character to hold '\0'

30int numlines(FILE *f, int *longestLine) {
31 int nl;
32 char c;
33 int length; // length of this line
34 int longLine;
35
36 nl = 0;
37 length = 0;
38 longLine = 0;
