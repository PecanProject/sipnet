92/* Dynamically allocate space for a 2-d array of doubles of given size,
93 return pointer to start of array
94
95 Ensures that array elements are continuous to increase efficiency
96
97 To free array, first free(arr[0]), then free(arr)
98*/
99double **make2DArray(int nrows, int ncols) {
100 double **arr;
101 int i;
102
103 arr = (double **)malloc(nrows * sizeof(double *));
104
105 // assure that all space is allocated continuously:
106 arr[0] = (double *)malloc(nrows * ncols * sizeof(double));
107 for (i = 1; i < nrows; i++)
108 arr[i] = arr[0] + i*ncols;
109
110 return arr;
111}

158// do an strcmp on s1 and s2, ignoring case
159// (convert both to lower case before comparing)
160// return value is the same as for strcmp
161int strcmpIgnoreCase(const char *s1, const char *s2) {
162 char *s1Lower;
163 char *s2Lower;
164 int i;
165 int result;
166
167 // allocate space for copies
168 // note that we need one more than strlen to allow room for termination character
169 s1Lower = (char *)malloc((strlen(s1) + 1) * sizeof(char));
170 s2Lower = (char *)malloc((strlen(s2) + 1) * sizeof(char));
171
172 for (i = 0; i <= strlen(s1); i++)
173 s1Lower[i] = tolower(s1[i]);
174 for (i = 0; i <= strlen(s2); i++)
175 s2Lower[i] = tolower(s2[i]);
176
177 result = strcmp(s1Lower, s2Lower);
178
179 free(s1Lower);
180 free(s2Lower);
181
182 return result;
183}
