100/* Read all namelist inputs from file given by fileName, put them in the locations pointed to by each inputs' ptr variable
101
102 Structure of input file: Each line contains:
103 name value
104 where the following are valid separators: space, tab, =, :
105
106 A value of 'none' for a string type is translated into the empty string
107
108 The order of input items in the input file does not matter
109
110 ! is a comment character: anything after a ! on a line is ignored
111 */
112void readNamelistInputs(NamelistInputs *namelistInputs, const char *fileName) {
113 const char *SEPARATORS = " \t=:"; // characters that can separate names from values in input file
114 const char *COMMENT_CHARS = "!"; // comment characters (ignore everything after this on a line)
115 const char *EMPTY_STRING = "none"; // if this is the value of a string type, we take it to mean the empty string
116
117 FILE *infile;
118
119 char line[1024];
120 char allSeparators[16];
121 char *inputName; // name of input item just read in
122 char *inputValue; // value of input item just read in, as a string
123 char *errc = "";
124 int isComment;
125 NamelistInputItem *namelistInputItem;
126
127
128 strcpy(allSeparators, SEPARATORS);
129 strcat(allSeparators, "\n\r");
130 infile = openFile(fileName, "r");
131
132 while(fgets(line, sizeof(line), infile) != NULL) { // while not EOF or error
133 // remove trailing comments:
134 isComment = stripComment(line, COMMENT_CHARS);
135
136 if (!isComment) { // if this isn't just a comment line or blank line
137 // tokenize line:
138 inputName = strtok(line, SEPARATORS); // make inputName point to first token
139 inputValue = strtok(NULL, allSeparators); // make inputValue point to next token (e.g. after the '=')
140
141 // printf("%s: <%s>\n", inputName, inputValue);
142
143 namelistInputItem = locateNamelistInputItem(namelistInputs, inputName);
144 if (namelistInputItem == NULL) {
145 printf("ERROR in readNamelistInputs: Read unexpected input item: %s\n", inputName);
146 printf("Please fix %s and re-run\n", fileName);
147 exit(1);
148 }
149 // otherwise, we have found the item with this name
150
151 if (inputValue == NULL) {
152 printf("ERROR in readNamelistInputs: No value given for input item %s\n", inputName);
153 printf("Please fix %s and re-run\n", fileName);
154 exit(1);
155 }
156
157 switch (namelistInputItem->type) {
158 case INT_TYPE:
159 *(int *)(namelistInputItem->ptr) = strtol(inputValue, &errc, 0);
160 if (strlen(errc) > 0) { // invalid character(s) in input string
161 printf("ERROR in readNamelistInputs: Invalid value for %s: %s\n", inputName, inputValue);
162 printf("Please fix %s and re-run\n", fileName);
163 // printf("%d <%s>\n", (int)strlen(errc), errc);
164 exit(1);
165 }
166 break;
167 case LONG_TYPE:
168 *(long *)(namelistInputItem->ptr) = strtol(inputValue, &errc, 0);
169 if (strlen(errc) > 0) { // invalid character(s) in input string
170 printf("ERROR in readNamelistInputs: Invalid value for %s: %s\n", inputName, inputValue);
171 printf("Please fix %s and re-run\n", fileName);
172 // printf("%d <%s>\n", (int)strlen(errc), errc);
173 exit(1);
174 }
175 break;
176 case DOUBLE_TYPE:
177 *(double *)(namelistInputItem->ptr) = strtod(inputValue, &errc);
178 if (strlen(errc) > 0) { // invalid character(s) in input string
179 printf("ERROR in readNamelistInputs: Invalid value for %s: %s\n", inputName, inputValue);
180 printf("Please fix %s and re-run\n", fileName);
181 // printf("%d <%s>\n", (int)strlen(errc), errc);
182 exit(1);
183 }
184 break;
185 case STRING_TYPE:
186 if (strcmp(inputValue, EMPTY_STRING) == 0) {
187 // if the value is specified as the value which signifies the empty string
188 strcpy((char *)(namelistInputItem->ptr), "");
189 }
190 else {
191 if (strlen(inputValue) >= namelistInputItem->maxlen) {
192 printf("ERROR in readNamelistInputs: value '%s' exceeds maximum length for %s (%d)\n", inputValue, inputName, namelistInputItem->maxlen);
193 printf("Please fix %s, or change the maximum length, and re-run\n", fileName);
194 exit(1);
195 }
196 strcpy((char *)(namelistInputItem->ptr), inputValue);
197 }
198 break;
199 default:
200 printf("ERROR in readNamelistInputs: Unrecognized type for %s: %d\n", inputName, namelistInputItem->type);
201 exit(1);
202 }
203
204 namelistInputItem->wasRead = 1;
205
206 } // if (!isComment)
207 } // while not EOF or error
208
209 fclose(infile);
210}
