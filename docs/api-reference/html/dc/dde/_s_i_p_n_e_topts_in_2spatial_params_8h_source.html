7// struct to hold a single (possibly) spatially-varying param
8typedef struct OneSpatialParamStruct {
9 char name[64]; // name of parameter
10 int numLocs; // 0 if this parameter is non-spatial
11 int isChangeable; // 0 or 1
12 double *value; /* if non-spatial, points to a single double
13 if spatial, points to a vector of doubles
14 before it's set, value = NULL */
15 double *externalLoc; // a pointer to a (non-spatial) copy of this parameter value elsewhere (e.g. local to the model)
16
17 // items that we care about when doing parameter optimization:
18 // parameter info read from file:
19 double *guess; // possibly spatial; if non-changeable, ptr will be void
20 double min; // left-hand side of allowable range
21 double max; // right-hand side of allowable range
22 double sigma; // parameter uncertainty
23
24 // variables related to optimization:
25 double *best; // best value found so far; possibly patial; if non-changeable, ptr will be void
26 double *knob; /* a "knob" that can be twisted to adjust this parameter's optimization (can be used however the individual optimization scheme wants)
27 possibly spatial; if non-changeable, ptr will be void */
28
29} OneSpatialParam;
