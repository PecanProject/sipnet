105/* set array of data types to only optimize on a restricted set of data types
106 note: this is an interactive function
107 NOTE2: THIS FUNCTION IS CURRENTLY UNUSED (as of 10/21/03)

109 returns number of data types
110*/
111int setDataTypeIndices(int dataTypeIndices[]) {
112 int numDataTypes;
113 int i, j, index;
114 int usedArray[MAX_DATA_TYPES]; // has the user already chosen a given data type?
115 char **dataTypeNames = getDataTypeNames();
116
117 // initialize usedArray:
118 for (i = 0; i < MAX_DATA_TYPES; i++)
119 usedArray[i] = 0; // false
120
121 printf("Data types available for optimization:\n");
122 for (i = 0; i < MAX_DATA_TYPES; i++)
123 printf("%d: %s\n", i, dataTypeNames[i]);
124
125 printf("\n");
126
127 // prompt for number of types:
128 do {
129 printf("Enter number of data types to use in optimization: ");
130 scanf("%d", &numDataTypes);
131
132 if (numDataTypes < 1 || numDataTypes > MAX_DATA_TYPES)
133 printf("Must enter a number between 1 and %d\n", MAX_DATA_TYPES);
134 } while (numDataTypes < 1 || numDataTypes > MAX_DATA_TYPES);
135
136 // prompt for each index:
137 i = 0;
138 while (i < numDataTypes) {
139 printf("\nData type %d of %d: Please choose one of the following:\n", i+1, numDataTypes);
140 for (j = 0; j < MAX_DATA_TYPES; j++) {
141 if (!usedArray[j]) // this data type hasn't been selected yet
142 printf("%d: %s\n", j, dataTypeNames[j]);
143 }
144 printf("Your choice: ");
145 scanf("%d", &index);
146
147 if (index < 0 || index >= MAX_DATA_TYPES || usedArray[index]) // out of bounds or already selected
148 printf("Invalid choice.\n");
149 else {
150 dataTypeIndices[i] = index;
151 usedArray[index] = 1; // this index is used
152 i++;
153 }
154 } // end while (i < numDataTypes)
155
156 return numDataTypes;
157}

159/* PRE: dataTypeIndices is at least big enough to hold MAX_DATA_TYPES ints
160 argv is vector of command-line arguments
161 startPos is index in argv where data type indices start
162 argc is total argument count (i.e. number of elements in argv[])
163 all arguments in argv from startPos to end are integers between 0 and (MAX_DATA_TYPES - 1), with no repeats
164
165 POST: fills dataTypeIndices with given integer arguments
166 returns number of data types
167*/
168int parseDataTypeIndices(int dataTypeIndices[], char *argv[], int startPos, int argc) {
169 int numDataTypes;
170 int i, pos;
171 int thisType;
172 char *errc = "";
173 int usedArray[MAX_DATA_TYPES]; // has a given data type already been chosen?
174
175 numDataTypes = (argc - startPos); // how many arguments are left? they should all be data type indices
176
177 // error checking:
178 if (numDataTypes <= 0) { // uh-oh: no more arguments left
179 usage(argv[0]);
180 exit(1);
181 }
182 else if (numDataTypes > MAX_DATA_TYPES) { // uh-oh: too many arguments left
183 printf("Error: more than %d data types specified\n", MAX_DATA_TYPES);
184 exit(1);
185 }
186
187 // initialize usedArray:
188 for (i = 0; i < MAX_DATA_TYPES; i++)
189 usedArray[i] = 0; // false
190
191 // read each data type:
192 i = 0;
193 pos = startPos;
194 while (i < numDataTypes) {
195 thisType = strtol(argv[pos], &errc, 0); // convert string argument to integer
196
197 // error check:
198 if (thisType < 0 || thisType >= MAX_DATA_TYPES) {
199 printf("Error: data type %d out of range [0..%d]\n", thisType, MAX_DATA_TYPES - 1);
200 exit(1);
201 }
202 else if (usedArray[thisType]) {
203 printf("Error: data type %d specified twice\n", thisType);
204 exit(1);
205 }
206
207 // okay, thisType is valid
208 dataTypeIndices[i] = thisType;
209 usedArray[thisType] = 1; // this index is used
210
211 i++;
212 pos++;
213 } // end while (i < numDataTypes)
214
215 return numDataTypes;
216}
