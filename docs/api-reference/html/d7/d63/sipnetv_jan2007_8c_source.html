345/* Read climate file into linked lists,
346 make firstClimates be a vector where each element is a pointer to the head of a list corresponding to one spatial location
347
348 return an array containing the number of time steps in each location (dynamically allocated with malloc)
349
350 numLocs = number of spatial locations: there should be one set of entries in climFile for each location,
351 or a location can be skipped, and we'll use climate at location 0 for that location, too
352
353 format of climFile:
354 each line represents one time step, with the following format:
355 location year day time intervalLength tair tsoil par precip vpd vpdSoil vPress wspd soilWetness netrad
356 Note: there should be NO blank lines in file, even between different spatial locations;
357 all entries for a given location should be contiguous, and locations should be in ascending order
358 There may be locations for which there is no climate data, in which case we use climate from location 0
359 If intervalLength is negative, then it is interpreted as being in seconds rather than days (e.g. -1800 for 1/2 hour time steps)
360 - In this case, it is multiplied by -1 and converted to days before being stored in the climate linked list
361*/
362int * readClimData(char *climFile, int numLocs) {
363 FILE *in;
364 ClimateNode *curr, *next;
365 int loc, year, day;
366 int lastYear;
367 double time, length; // time in hours, length in days (or fraction of day)
368 double tair, tsoil, par, precip, vpd, vpdSoil, vPress, wspd, soilWetness, netrad, soilheat;
369 int currLoc;
370 int count;
371 int i;
372 int *steps; // # of time steps in each location
373
374#if GDD
375 double thisGdd; // growing degree days of this time step
376 double gdd = 0.0; // growing degree days since the last Jan. 1
377#endif
378
379 int status; // status of the read
380
381 steps = (int *)malloc(numLocs * sizeof(int));
382 for (i = 0; i < numLocs; i++)
383 steps[i] = 0; // 0 will denote nothing read
384
385 in = openFile(climFile, "r");
386
387 status = fscanf(in, "%d %d %d %lf %lf %lf %lf %lf %lf %lf %lf %lf %lf %lf %lf %lf", &loc, &year, &day, &time, &length, &tair, &tsoil, &par, &precip, &vpd, &vpdSoil, &vPress, &wspd, &soilWetness, &netrad, &soilheat);
388 if (status == EOF) {
389 printf("Error: no climate data in %s\n", climFile);
390 exit(1);
391 }
392
393 if (loc != 0) {
394 printf("Error reading from climate file: first location must be loc. 0\n");
395 exit(1);
396 }
397
398 firstClimates = (ClimateNode **)malloc(numLocs * sizeof(ClimateNode *)); // vector of heads
399 for (currLoc = 0; currLoc < numLocs; currLoc++)
400 firstClimates[currLoc] = NULL; // default is null
401 // if firstClimates[i] stays null for some i, that means we didn't read any climate data for location i, so use climate from location 0
402
403 currLoc = loc;
404 firstClimates[currLoc] = (ClimateNode *)malloc(sizeof(ClimateNode)); // allocate space for head
405 curr = next = firstClimates[currLoc];
406 count = 0;
407 while (status != EOF) {
408 // we have another day's climate
409 curr = next;
410 count++; // # of time steps in this location
411
412 curr->year = year;
413 curr->day = day;
414 curr->time = time;
415
416 if (length < 0) // parse as seconds
417 length = length/-86400.; // convert to days
418 curr->length = length;
419
420 curr->tair = tair;
421 curr->tsoil = tsoil;
422 curr->par = par * (1.0/length);
423 // convert par from Einsteins * m^-2 to Eisteins * m^-2 * day^-1
424 curr->precip = precip * 0.1; // convert from mm to cm
425 curr->vpd = vpd * 0.001; // convert from Pa to kPa
426 if (curr->vpd < TINY)
427 curr->vpd = TINY; // avoid divide by zero
428 curr->vpdSoil = vpdSoil * 0.001; // convert from Pa to kPa
429 curr->vPress = vPress * 0.001; // convert from Pa to kPa
430 curr->wspd = wspd;
431 if (curr->wspd < TINY)
432 curr->wspd = TINY; // avoid divide by zero
433 curr->soilWetness = soilWetness;
434 curr->netrad = netrad;
435 curr->soilheat = soilheat;
436#if GDD
437 if (year != lastYear) // HAPPY NEW YEAR!
438 gdd = 0; // reset growing degree days
439 thisGdd = tair * length;
440 if (thisGdd < 0) // can't have negative growing degree days
441 thisGdd = 0;
442 gdd += thisGdd;
443 curr->gdd = gdd;
444#endif
445
446 lastYear = year;
447
448 status = fscanf(in, "%d %d %d %lf %lf %lf %lf %lf %lf %lf %lf %lf %lf %lf %lf %lf", &loc, &year, &day, &time, &length, &tair, &tsoil, &par, &precip, &vpd, &vpdSoil, &vPress, &wspd, &soilWetness, &netrad, &soilheat);
449
450 if (status != EOF) { // we have another climate record
451 // check new location, compare with old location (currLoc), make sure new location is valid, and act accordingly:
452 if (loc == currLoc) { // still reading climate records from the same place: add a node at end of linked list
453 next = (ClimateNode *)malloc(sizeof(ClimateNode));
454 curr->nextClim = next;
455 // set this down here rather than at top of loop so head treated the same as rest of list
456 }
457 else if (loc >= numLocs) { // (loc == numLocs is an error since we use 0-indexing)
458 printf("Error reading climate file: trying to read location %d, but numLocs = %d\n", loc, numLocs);
459 exit(1);
460 }
461 else if (loc > currLoc) { // we've advanced to the next location (note: possible that we skipped some locations: this is OK)
462 curr->nextClim = NULL; // terminate last linked list
463 steps[currLoc] = count; // record the number of time steps for last location
464 count = 0; // reset count of # of time steps
465#if GDD
466 gdd = 0; // reset growing degree days
467#endif
468 currLoc = loc;
469 firstClimates[currLoc] = (ClimateNode *)malloc(sizeof(ClimateNode)); // allocate space for head
470 curr = next = firstClimates[currLoc]; // we'll start writing to next linked list
471 }
472 else { // loc < currLoc
473 printf("Error reading climate file: was reading location %d, trying to read location %d\n", currLoc, loc);
474 printf("Climate records for a given location should be contiguous, and locations should be in ascending order\n");
475 exit(1);
476 }
477 }
478 else { // status == EOF - no more records
479 curr->nextClim = NULL; // terminate this last linked list
480 steps[currLoc] = count; // record the number of time steps for last location
481 }
482
483 } // end while
484
485 fclose(in);
486
487 for (i = 0; i < numLocs; i++)
488 if (steps[i] == 0) // nothing read for this location
489 steps[i] = steps[0]; // this location will duplicate location 0
490
491 return steps;
492}

622// calculate amount of light absorbed
623// using light attenuation (as in PnET)
624// CURRENTLY UNUSED (see calcLightEff3)
625void calcLightEff2 (double *lightEff, double lai, double par) {
626 static const int NUM_LAYERS = 50;
627
628 int layer; // counter
629 double cumLai = 0.0; // lai from this layer up
630 double lightIntensity;
631 double cumLightEff = 0.0;
632
633 if (lai > 0 && par > 0) { // must have at least some leaves and some light; DANGER: sensitive to round-off errors: e.g. lai = -0.00000001 will give a very different lightEff from lai = 0.000000001; but since we later multiply by lai (in potPsn), this shouldn't lead to any major problems
634 for (layer = 1; layer <= NUM_LAYERS; layer++) {
635 cumLai = lai * ((double)layer / NUM_LAYERS); // lai from this layer up
636 lightIntensity = par * exp(-1.0 * params.attenuation * cumLai); // between 0 and par
637 cumLightEff += (1 - pow(2, (-1.0 * lightIntensity/params.halfSatPar))); // add between 0 and 1
638 // when lightIntensity = halfSatPar, add 1/2
639 }
640
641 *lightEff = cumLightEff/(double)NUM_LAYERS; // find average light effect
642 }
643 else // no leaves or no light!
644 *lightEff = 0;
645}

648// another method to calculate amount of light absorbed
649// using light attenuation (as in PnET)
650// difference between this and calcLightEff2 is that here we use Simpson's method to approximate the integral
651void calcLightEff3 (double *lightEff, double lai, double par) {
652 /*
653 We are essentially integrating over the canopy, from top to bottom,
654 but it's an ugly integral, so we'll approximate it numerically.
655 Here we use Simpson's method to approximate the integral, so we need an odd number of points.
656 This means that NUM_LAYERS must be EVEN because we loop from layer = 0 to layer = NUM_LAYERS
657
658 As a reminder, Simpson's rule approximates the integral as:
659 (h/3) * (y(0) + 4y(1) + 2y(2) + 4y(3) + ... + 2y(n-2) + 4y(n-1) + y(n)),
660 where h is the distance between each x value (here h = 1/NUM_LAYERS).
661 We keep track of the running sum in cumLightEff.
662 */
663
664 static const int NUM_LAYERS = 6;
665 // believe it or not, 6 layers gives approximately the same result as 100 layers
666
667 int layer; // counter
668 double cumLai; // lai from this layer up
669 double lightIntensity;
670 double currLightEff, cumLightEff;
671 int coeff; // current coefficient in Simpson's rule
672
673 if (lai > 0 && par > 0) { // must have at least some leaves and some light; DANGER: sensitive to round-off errors: e.g. lai = -0.00000001 will give a very different lightEff from lai = 0.000000001; but since we later multiply by lai (in potPsn), this shouldn't lead to any major problems
674 cumLai = 0.0;
675 cumLightEff = 0.0; // the running sum
676 layer = 0;
677 coeff = 1;
678
679 while (layer <= NUM_LAYERS) {
680 cumLai = lai * ((double)layer / NUM_LAYERS); // lai from this layer up
681 lightIntensity = par * exp(-1.0 * params.attenuation * cumLai); // between 0 and par
682 currLightEff = (1 - pow(2, (-1.0 * lightIntensity/params.halfSatPar))); // between 0 and 1
683 // when lightIntensity = halfSatPar, currLightEff = 1/2
684 cumLightEff += coeff * currLightEff;
685
686 // now move to the next layer:
687 layer++;
688 coeff = 2*(1 + layer%2); // coeff. goes 1, 4, 2, 4, ..., 2, 4, 2
689 }
690 // last value should have had a coefficient of 1, but actually had a coefficient of 2,
691 // so subtract 1:
692 cumLightEff -= currLightEff;
693
694 *lightEff = cumLightEff/(3.0*NUM_LAYERS); // multiplying by (h/3) in Simpson's rule
695 }
696 else // no leaves or no light!
697 *lightEff = 0;
698}

702// calculate gross photosynthesis without water effect (g C * m^-2 ground area * day^-1)
703// and base foliar respiration without temp, water, etc. (g C * m^-2 ground area * day^-1)
704void potPsn(double *potGrossPsn, double *baseFolResp, double lai, double tair, double vpd, double par, int day) {
705 double grossAMax; // maximum possible gross respiration (nmol CO2 * g^-1 leaf * sec^-1)
706 double dTemp, dVpd, lightEff; // decrease in photosynth. due to temp, vpd and amt. of light absorbed
707 double respPerGram; // base foliar respiration in nmol CO2 * g^-1 leaf * sec^-1
708 double conversion; /* convert from (nmol CO2 * g^-1 leaf * sec^-1)
709 to (g C * m^-2 ground area * day^-1) */
710
711 respPerGram = params.baseFolRespFrac * params.aMax;
712 // foliar respiration, unmodified by temp, etc.
713 grossAMax = params.aMax * params.aMaxFrac + respPerGram;
714
715 dTemp = (params.psnTMax - tair)*(tair - params.psnTMin)/pow((params.psnTMax - params.psnTMin)/2.0, 2);
716 if (dTemp < 0)
717 dTemp = 0.0;
718
719 dVpd = 1.0 - params.dVpdSlope * pow(vpd, params.dVpdExp);
720
721 calcLightEff3(&lightEff, lai, par);
722
723 conversion = C_WEIGHT * (1.0/TEN_9) * (params.leafCSpWt/params.cFracLeaf) * lai * SEC_PER_DAY; // to convert units
724 *potGrossPsn = grossAMax * dTemp * dVpd * lightEff * conversion;
725 *baseFolResp = respPerGram * conversion; // do foliar resp. even if no photosynthesis in this time step
726
727 // printf("%f %f %f %f %f %f ", climate->length, grossAMax, conversion, dTemp, dVpd, lightEff);
728}

733void moisture_bwb(double *trans, double *dWater, double potGrossPsn, double vpd, double vPress, double plantLeafC, double leafCSpWt, double soilWater) {
734/*moisture_bwb: Calculates moisture use by using a Ball Woodrow Berry
735 * Instead of A (net photosynthesis) we are driving the Ball Woodrow Berry
736 * equation with potGrossPsn units: g C * m^-2 ground area * day^-1
737 * formulation which estimates gs from A */
738double potTrans; // potential transpiration in the absense of plant water stress (cm H20 * day^-1)
739 double removableWater;
740 double gs_canopy; //gs stomatal conductance... canopy level
741 double CO2_stom = 380;
742 double vPress_sat; // Saturating Vapor Press (sum of VPD and vPress)
743 double RH_pcent; // Relative Humidity - the ratio of vPress to saturating vapor pressure
744 double lai_int; // calculate lai from current plantLeafC and the leafCSpwt */
745 lai_int = envi.plantLeafC / params.leafCSpWt;
746 /* We do not require lai since potGrossPsn is in units of
747 * g C * m^-2 ground area * day^-1 */
748 vPress_sat = climate->vPress + climate->vpd; //calculate saturating vapor pressure
749 RH_pcent = climate->vPress / vPress_sat; // calculate relative humidity for ball berry
750
751// double wue; // water use efficiency, in mg CO2 fixed * g^-1 H20 transpired
752
753 if (potGrossPsn < TINY) { // avoid divide by 0
754 *trans = 0.0; // no photosynthesis -> no transpiration
755 *dWater = 1; // dWater doesn't matter, since we don't have any photosynthesis
756 }
757 else {
758 /*Ball Berry Equation
759 * unit issues:
760 * potGrossPsn should be converted to
761 * micro mol c02 per m^2 leaf area
762 * there are 1/12 mol of carbon in a gram
763 * potGrossPsn/12*1000*LAI = umol carbon dioxide per
764 * m^2
765 * */;
766 gs_canopy = params.m_ballBerry * potGrossPsn/12*1000*lai_int * (RH_pcent/CO2_stom) ;
767 //mol / m^2 leaf area
768
769 potTrans = (gs_canopy *climate->vpd/climate->vPress)/lai_int*20/1000;
770 /* gs_canopy/climate_>vpd is the transpiration rate mol/m^2 leaf area per day
771 * dividing by lai_init converts to mol per m^2 ground area
772 * there are 20g water per mol
773 * multiply by the density of water 1cm^3/g
774 * multipy by 100cm and divide by 10^6cm^3
775 *
776 * must convert to cm3 per cm2 land area.
777 */
778 removableWater = soilWater * params.waterRemoveFrac;
779 if (climate->tsoil < params.frozenSoilThreshold) // frozen soil - less or no water available
780 removableWater *= params.frozenSoilEff;
781 /* frozen soil effect: fraction of water available if soil is frozen
782 (assume amt. of water avail. w/ frozen soil scales linearly with amt. of
783 water avail. in thawed soil) */
784 if (removableWater >= potTrans)
785 *trans = potTrans;
786 else
787 *trans = removableWater;
788
789#if WATER_PSN // we're modeling water stress
790 *dWater = *trans/potTrans; // from PnET: equivalent to setting DWATER_MAX = 1
791#else // WATER_PSN = 0
792 if (climate->tsoil < params.frozenSoilThreshold && params.frozenSoilEff == 0)
793 // (note: can't have partial shutdown of psn with frozen soil if WATER_PSN = 0)
794 *dWater = 0; // still allow total shut down of psn. if soil is frozen
795 else // either soil is thawed, or frozenSoilEff > 0
796 *dWater = 1; // no water stress, even if *trans/potTrans < 1
797#endif // WATER_PSN
798//printf("Remove %f potT %f dW %f vpd %f \n", removableWater, potTrans, *dWater, climate->vpd);
799 }

961 double npp; // temporal mean of recent npp (g C * m^-2 ground * day^-1)
962
963 npp = getMeanTrackerMean(meanNPP);
964 // first determine the fluxes that happen at every time step, not just start & end of growing season:
965 if (npp > 0)
966 *leafCreation = npp * params.leafAllocation; // a fraction of NPP is allocated to leaf growth
967 else // net loss of C in this time step - no C left for growth
968 *leafCreation = 0;
969 *leafLitter = plantLeafC * params.leafTurnoverRate; // a constant fraction of leaves fall in each time step
970
971 // now add add'l fluxes at start/end of growing season:
972
973 // first check for new year; if new year, reset trackers (since we haven't done leaf growth or fall yet in this new year):
974 if (climate->year > phenologyTrackers.lastYear) { // HAPPY NEW YEAR!
975 phenologyTrackers.didLeafGrowth = 0;
976 phenologyTrackers.didLeafFall = 0;
977 phenologyTrackers.lastYear = climate->year;
978 }
979
980 // check for start of growing season:
981 if (!phenologyTrackers.didLeafGrowth && pastLeafGrowth()) { // we just reached the start of the growing season
982 *leafCreation += (params.leafGrowth / climate->length);
983 phenologyTrackers.didLeafGrowth = 1;
984 }
985
986 // check for end of growing season:
987 if (!phenologyTrackers.didLeafFall && pastLeafFall()) { // we just reached the end of the growing season
988 *leafLitter += (plantLeafC * params.fracLeafFall) / climate->length;
989 phenologyTrackers.didLeafFall = 1;
990 }
991}

998// this is the simplified water flow function, which has only one soil moisture layer

1009 *rain = 0;
1010 *snowMelt = 0;
1011 }
1012 else { // above freezing

1042// calculate total rain and snowfall (cm water equiv./day)
1043// also, immediate evaporation (from interception) (cm/day)
1044void calcPrecip(double *rain, double *snowFall, double *immedEvap)
1045{
1046 // below freezing -> precip falls as snow
1047 if (climate->tair <= 0) {
1048 *snowFall = climate->precip/climate->length;
1049 *rain = 0;
1050 }
1051
1052 // above freezing -> precip falls as rain
1053 else {
1054 *snowFall = 0;
1055 *rain = climate->precip/climate->length;
1056 }
1057
1058 /* Immediate evaporation is a sum of evaporation from canopy interception
1059 and evaporation from pools on the ground
1060 Higher LAI will mean more canopy evap. but less evap. from pools on the ground
1061 For now we'll assume that these two effects cancel, and immediate evap. is a constant fraction
1062 Note that we don't evaporate snow here (we'll let sublimation take care of that)
1063 */
1064 *immedEvap = (*rain) * params.immedEvapFrac;
1065}
1066
1067// snowpack dynamics:
1068// calculate snow melt (cm water equiv./day) & sublimation (cm water equiv./day)
1069// ensure we don't overdrain the snowpack (so that it becomes negative)
1070// snowFall in cm/day
1071void snowPack(double *snowMelt, double *sublimation, double snowFall)
1072{
1073 // conversion factor for sublimation
1074 static const double CONVERSION = (RHO * CP)/GAMMA * (1./LAMBDA_S)
1075 * 1000. * 1000. * (1./10000) * SEC_PER_DAY;
1076 // 1000 converts kg to g, 1000 converts kPa to Pa, 1/10000 converts m^2 to cm^2
1077
1078 double rd; // aerodynamic resistance between ground and canopy air space (sec/m)
1079 double snowRemaining; // to make sure we don't get rid of more than there is
1080
1081 /* If no snow, set fluxes to 0
1082 Note that this formulation means that there could be big differences in the fluxes depending on whether the snow pack is 0 or slightly greater than 0
1083 e.g. if there is snow fall in timestep i but no snow pack, then there will be no sublimation or snowMelt, but if there is a small snow pack, then there can be sublimation
1084 This could have the unrealistic result that, with no snow pack in timestep i, there is actually a larger snow pack in timestep (i+1) than there would be if there had been a snow pack in timestep i
1085 However, if we allow sublimation when there is snowfall but no snow pack, then we could have sublimation and evaporation from the soil in the same time step, which may be even worse
1086 Note that there is, in principle, no reason why we should set snowMelt = 0 if there is snow fall but no snow pack,
1087 in practice (the way the model is written now), snowMelt will always = 0 in this case anyway, since snow fall only happens when tair < 0 and snow melt only when tair > 0
1088 Note that if we wanted to allow sublimation when there is snow fall but no snow pack, then the check here could be rewritten as if (snowRemaining <= 0) rather than envi.snow <= 0 (and snowRemaining could be calculated before the check)
1089 */
1090 if (envi.snow <= 0) {
1091 *snowMelt = 0;
1092 *sublimation = 0;
1093 }
1094
1095 // else: there is snow
1096 else {
1097 // first calculate sublimation, then snow melt
1098 // (if there's not enough snow to do both, priority given to sublimation)
1099 rd = (params.rdConst)/(climate->wspd); // aerodynamic resistance (sec/m)
1100 *sublimation = CONVERSION * (E_STAR_SNOW - climate->vPress)/rd;
1101
1102 snowRemaining = envi.snow + (snowFall * climate->length);
1103
1104 // remove to allow sublimation of a negative amount of snow
1105 // right now we can't sublime a negative amount of snow
1106 if (*sublimation < 0)
1107 *sublimation = 0;
1108
1109 // make sure we don't sublime more than there is to sublime:
1110 if (snowRemaining - (*sublimation * climate->length) < 0) {
1111 *sublimation = snowRemaining/climate->length;
1112 snowRemaining = 0;
1113 }
1114
1115 else
1116 snowRemaining -= (*sublimation * climate->length);
1117
1118
1119 // below freezing: no snow melt
1120 if (climate->tair <= 0)
1121 *snowMelt = 0;
1122
1123 // above freezing: melt snow
1124 else {
1125 *snowMelt = params.snowMelt * climate->tair; // snow melt proportional to temp.
1126
1127 // make sure we don't melt more than there is to melt:
1128 if (snowRemaining - (*snowMelt * climate->length) < 0)
1129 *snowMelt = snowRemaining/climate->length;
1130 } // end else above freezing
1131 } // end else there is snow
1132
1133 // printf("%d %f %f %f %f %f %f\n", climate->day, climate->tair, climate->precip, envi.snow, *snowMelt, *sublimation, snowFall);
1134} // end snowPack

1137// water calculations relating to the top (litter/evaporative) layer of soil
1138// (if only modeling one soil water layer, these relate to that layer)
1139// calculates fastFlow (cm/day), evaporation (cm/day) and drainage to lower layer (cm/day)
1140// water is amount of water in evaporative layer (cm)
1141// whc is water holding capacity of evaporative layer (cm)
1142// (water and whc are used rather than envi variables to allow use of either litterWater or soilWater)
1143// net rain (cm/day) is (rain - immedEvap) - i.e. the amount available to enter the soil
1144// snowMelt in cm water equiv./day
1145// fluxesOut is the sum of any fluxes out of this layer that have already been calculated
1146// (e.g. transpiration if we're just using one layer) (for calculating remaining water/drainage) (cm/day)
1147void evapSoilFluxes(double *fastFlow, double *evaporation, double *drainage,
1148 double water, double whc, double netRain, double snowMelt, double fluxesOut)
1149{
1150 // conversion factor for evaporation
1151 static const double CONVERSION = (RHO * CP)/GAMMA * (1./LAMBDA)
1152 * 1000. * 1000. * (1./10000) * SEC_PER_DAY;
1153 // 1000 converts kg to g, 1000 converts kPa to Pa, 1/10000 converts m^2 to cm^2
1154
1155 double waterRemaining; /* keep running total of water remaining, in cm
1156 (to make sure we don't evap. or drain too much, and so we can drain any overflow) */
1157 double netIn; // keep track of net water into soil, in cm/day
1158 double rd; // aerodynamic resistance between ground and canopy air space (sec/m)
1159 double rsoil; // bare soil surface resistance (sec/m)
1160
1161 netIn = netRain + snowMelt;
1162
1163 // fast flow: fraction that goes directly to drainage
1164 *fastFlow = netIn * params.fastFlowFrac;
1165 netIn -= *fastFlow;
1166
1167 // calculate evaporation:
1168 // first calculate how much water is left to evaporate (used later)
1169 waterRemaining = water + netIn * climate->length - fluxesOut * climate->length;
1170
1171 // if there's a snow pack, don't evaporate from soil:
1172 if (envi.snow > 0)
1173 *evaporation = 0;
1174
1175 // else no snow pack:
1176 else {
1177 rd = (params.rdConst)/(climate->wspd); // aerodynamic resistance (sec/m)
1178 rsoil = exp(params.rSoilConst1 - params.rSoilConst2 * (water/whc));
1179 *evaporation = CONVERSION * climate->vpdSoil/(rd + rsoil);
1180 // by using vpd we assume that relative humidity of soil pore space is 1
1181 // (when this isn't true, there won't be much water evaporated anyway)
1182
1183 // remove to allow negative evaporation (i.e. condensation):
1184 if (*evaporation < 0)
1185 *evaporation = 0;
1186
1187 // make sure we don't evaporate more than we have:
1188 if (waterRemaining - (*evaporation * climate->length) < 0) {
1189 *evaporation = waterRemaining/climate->length;
1190 waterRemaining = 0;
1191 }
1192 else
1193 waterRemaining -= (*evaporation * climate->length);
1194 }
1195
1196#if LITTER_WATER_DRAINAGE // we're calculating drainage even when evap. layer is not overflowing
1197 *drainage = params.litWaterDrainRate * (water/whc); // drainage rate is proportional to fractional soil moisture
1198 // make sure we don't drain more than we have:
1199 if (waterRemaining - (*drainage * climate->length) < 0) {
1200 *drainage = waterRemaining/climate->length;
1201 waterRemaining = 0;
1202 }
1203 else
1204 waterRemaining -= (*drainage * climate->length);
1205#else // LITTER_WATER_DRAINAGE = 0
1206 *drainage = 0;
1207#endif
1208
1209 // drain any water that remains beyond water holding capacity:
1210 if (waterRemaining > whc)
1211 *drainage += (waterRemaining - whc)/(climate->length);
1212}

1217void transSoilDrainage(double *bottomDrainage, double topDrainage, double trans, double soilWater) {

1223 *bottomDrainage = 0;

1228// net rain (cm/day) is (rain - immedEvap) - i.e. the amount available to enter the soil
1229// snowMelt in cm water equiv./day
1230// litterWater and soilWater in cm

1285// general function to calculate soil C broken down, as a function of tsoil and soil moisture:
1286// (g C * m^-2 * day^-1)
1287// based on function in PnET-CN
1288// poolC is amt. of C in this soil pool (g C * m^-2),
1289// baseRate is breakdown rate at 0 degrees C, max soil moisture (g respired/g/day)
1290// water and whc (both in cm) refer to this specific soil layer
1291double soilBreakdown(double poolC, double baseRate, double water, double whc, double tsoil, double Q10) {
1292 double tempEffect;
1293 double moistEffect;
1294
1295#if DAYCENT_WATER_HRESP
1296 // calculate here so only have to do this ugly calculation once
1297 static const double daycentWaterExp = 3.22 * (1.7 - 0.55)/(0.55 + 0.007);
1298#endif
1299
1300
1301#if LLOYD_TAYLOR // we're using the Lloyd-Taylor model for temperature effect
1302 tempEffect = exp(params.E0 * (-1/params.T0 - 1/(tsoil - params.T0)));
1303#else // using standard Q10
1304 tempEffect = pow(Q10, tsoil/10.0);
1305#endif
1306
1307#if WATER_HRESP // if soil moisture affects heterotrophic resp.
1308
1309#if DAYCENT_WATER_HRESP // using DAYCENT formulation
1310 // NOTE: would probably be best to make all these constants parameters that are read in, but for now we have them hard-coded
1311 moistEffect = pow(((water/whc - 1.7)/(0.55 - 1.7)), daycentWaterExp)
1312 * pow((water/whc + 0.007)/(0.55 + 0.007), 3.22);
1313#else // using PnET formulation
1314 moistEffect = pow((water/whc), params.soilRespMoistEffect);
1315#endif // DAYCENT_WATER_HRESP
1316
1317#else // no WATER_HRESP
1318 moistEffect = 1;
1319#endif // WATER_HRESP
1320
1321 return poolC * baseRate * tempEffect * moistEffect;
1322}
