15/* PRE: totWeight > 0, maxEntries >= 1
16 allocate space for and return a pointer to a new MeanTracker
17 New MeanTracker is initialized to have mean = initMean and appropriate total weight
18 Total weight will remain fixed once the MeanTracker is created
19 maxEntries gives the maximum number of individual values that can be stored at once

21MeanTracker *newMeanTracker(double initMean, double totWeight, int maxEntries) {
22 MeanTracker *tracker;
23
24 // error check:
25 if (totWeight <= 0 || maxEntries < 1)
26 return NULL;
27
28 tracker = (MeanTracker *)malloc(sizeof(MeanTracker));
29 tracker->values = (double *)malloc(maxEntries * sizeof(double));
30 tracker->weights = (double *)malloc(maxEntries * sizeof(double));
31 tracker->length = maxEntries;
32 tracker->totWeight = totWeight;
33 resetMeanTracker(tracker, initMean);
34
35 return tracker;
36}

39/* PRE: tracker has been created using newMeanTracker
40 so has malloced values and weights arrays, and has length and totWeight set
41 reset tracker to have a single entry, with mean initMean
42*/
43void resetMeanTracker(MeanTracker *tracker, double initMean) {
44 // start with only one entry, with appropriate mean and total weight
45 // don't bother resetting other entries - we'll set them as we come to them
46 tracker->start = tracker->last = 0;
47 tracker->values[tracker->start] = initMean;
48 tracker->weights[tracker->start] = tracker->totWeight;
49 tracker->sum = initMean * tracker->totWeight;
50}

53/* PRE: weight > 0
54 Add current value as most recent value in tracker
55
56 Return value:
57 0 if all okay
58 -1 if error in inputs
59 -2 if error because array is too short
60*/
61int addValueToMeanTracker(MeanTracker *tracker, double value, double weight) {
62 double weightLeft;
63 int i;
64
65 if (weight <= 0) // error
66 return -1; // don't change tracker at all
67 else if (weight >= tracker->totWeight) // current value knocks out all previous values
68 resetMeanTracker(tracker, value);
69 else { // 0 < weight < totWeight: remove oldest values to make room for new (keeping total weight the same)
70 weightLeft = weight; // how much more do we have to subtract to make room?
71 i = tracker->start; // current position
72
73 while (weightLeft > 0) { // while we still have to subtract more
74 if (tracker->weights[i] > weightLeft) { // subtract a portion of this one and we'll be done
75 tracker->weights[i] -= weightLeft; // subtract correct portion from current weight
76 tracker->sum -= weightLeft * tracker->values[i]; // subtract correct portion from sum
77 weightLeft = 0; // we're done
78 }
79 else { // remove this value entirely to make room
80 tracker->sum -= tracker->weights[i] * tracker->values[i]; // update sum
81 weightLeft -= tracker->weights[i]; // we now have this much less weight to subtract
82 i = (i + 1) % tracker->length; // forget this value
83 // note: we don't have to worry about going past last value, since 1) we check for weight >= tracker->totWeight,
84 // and 2) before starting this function, we know that sum(tracker->weights[start..last] == tracker->totWeight)
85 }
86 } // end while: now weightLeft = 0
87
88 tracker->start = i; // move start position along to correct new position
89 i = (tracker->last + 1) % tracker->length; // point to position of new value
90 if (i == tracker->start) { // uh-oh: we've wrapped around and are out of space!
91 // restore previous state and return error value
92 // note: this will only happen if we only executed the while loop once, and entered the first if block
93 // we can take advantage of this fact to restore the old state
94 tracker->weights[i] += weight; // restore old weight here
95 tracker->sum += weight * tracker->values[i]; // restore old sum
96 return -2; // error
97 }
98 else { // there's enough space for the new value
99 tracker->last = i; // increment tracker->last to point to new value
100 tracker->values[i] = value;
101 tracker->weights[i] = weight;
102 tracker->sum += value * weight; // update sum appropriately
103 }
104 } // end else (0 < weight < totWeight)
105
106 return 0; // all okay
107}
