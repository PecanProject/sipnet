139void readModelParams(ModelParams *modelParams, FILE *paramFile) {
140 const char *SEPARATORS = " \t\n\r"; // characters that can separate values
141 // in parameter files
142 const char *COMMENT_CHARS = "!"; // comment characters (ignore everything
143 // after this on a line)
144
145 char line[256];
146 char pName[MODEL_PARAM_MAXNAME]; // parameter name
147 int paramIndex;
148 OneModelParam *param; // a pointer to a single parameter, for easier access
149 char strValue[32]; // before we know whether value is a number or "*"
150 double value;
151 char *errc;
152 int isComment;
153 char unknownParams[2048] = "";
154 char hasUnknownParams = 0;
155
156 // Check for old-style (spatial param) format on first line containing params
157 int formatChecked = 0;
158
159 while (fgets(line, sizeof(line), paramFile) != NULL) { // while not EOF or
160 // error
161 // remove trailing comments:
162 isComment = stripComment(line, COMMENT_CHARS);
163
164 if (!isComment) { // if this isn't just a comment line or blank line
165 // Check for old spatial-param format and warn if appropriate
166 if (!formatChecked) {
167 checkParamFormat(line, SEPARATORS);
168 formatChecked = 1;
169 }
170
171 // tokenize line:
172 strcpy(pName, strtok(line, SEPARATORS)); // copy first token into pName
173
174 strcpy(strValue, strtok(NULL, SEPARATORS));
175
176 if (strcmp(strValue, "*") == 0) {
177 // This used to mean "spatially varying" when sipnet supported multiple
178 // sites, but now this is an error
179 logError("reading parameter %s; '*' is no longer supported\n", pName);
180 exit(EXIT_CODE_BAD_PARAMETER_VALUE);
181 }
182 value = strtod(strValue, &errc);
183 paramIndex = locateParam(modelParams, pName);
184
185 if (paramIndex == -1) { // not found
186 if (hasUnknownParams) {
187 if (strlen(unknownParams) + strlen(pName) > 2048) {
188 logError("Too many unknown params; please remove some from %s and "
189 "rerun\n",
190 paramFile);
191 logError("Unknown params include: %s\n", unknownParams);
192 exit(EXIT_CODE_INPUT_FILE_ERROR);
193 }
194 strcat(unknownParams, ", ");
195 strcat(unknownParams, pName);
196 } else {
197 strcat(unknownParams, pName);
198 hasUnknownParams = 1;
199 }
200 } else if (valueSet(modelParams, paramIndex)) {
201 logError("reading parameter file: read %s, but this parameter has "
202 "already been set\n",
203 pName);
204 exit(EXIT_CODE_INPUT_FILE_ERROR);
205 } else { // otherwise, we're good to go
206 // set param to point to the appropriate parameter, for easier access
207 param = &(modelParams->params[paramIndex]);
208 *(param->value) = value;
209 param->isRead = 1;
210 // mark this as the next parameter read from file:
211 modelParams->readIndices[modelParams->numParamsRead] = paramIndex;
212 modelParams->numParamsRead++;
213 } // else (no errors in reading this line from parameter file)
214 } // if !isComment
215 } // while not EOF or error
216
217 // Warn if unknown params were found
218 if (hasUnknownParams) {
219 logWarning("Unknown param(s) found (and ignored): %s\n", unknownParams);
220 }
221
222 // check for error in reading:
223 if (ferror(paramFile)) {
224 logError("reading file in readModelParams\n");
225 logError("ferror = %d\n", ferror(paramFile));
226 exit(1);
227 }
228
229 checkAllRead(modelParams); // terminate program if some required parameters
230 // weren't read
231} // readModelParams
